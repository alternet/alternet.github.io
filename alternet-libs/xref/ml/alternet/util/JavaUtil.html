3   import java.util.Arrays;
4   import java.util.Collections;
5   import java.util.HashSet;
6   import java.util.Set;

9   /**
10   * Utilities related to the Java language.
11   *
12   * @author Philippe Poulard
13   */
14  @Util
15  public final class JavaUtil {
16  
17      private JavaUtil() { }
18  
19      private static Set<String> KW = new HashSet<>(Arrays.asList(
20              "abstract", "continue", "for",        "new",       "switch",
21              "assert",   "default",  "goto",       "package",   "synchronized",
22              "boolean",  "do",       "if",         "private",   "this",
23              "break",    "double",   "implements", "protected", "throw",
24              "byte",     "else",     "import",     "public",    "throws",
25              "case",     "enum",     "instanceof", "return",    "transient",
26              "catch",    "extends",  "int",        "short",     "try",
27              "char",     "final",    "interface",  "static",    "void",
28              "class",    "finally",  "long",       "strictfp",  "volatile",
29              "const",    "float",    "native",     "super",     "while"));
30  
31      /**
32       * Return the set of Java keywords.
33       *
34       * @return The set of Java keywords.
35       */
36      public static Set<String> keywords() {
37          return Collections.unmodifiableSet(KW);
38      }
39  
40      /**
41       * Indicates whether a word is a Java keyword or not.
42       *
43       * @param s The word to check.
44       *
45       * @return <code>true</code> if the word is a Java keyword,
46       *      <code>false</code> otherwise.
47       */
48      public static boolean isKeyword(String s) {
49          return KW.contains(s);
50      }
51  
52      /**
53       * Remove the characters from a string that are
54       * not legal for a Java name.
55       * NOTE : this method doesn't check whether the
56       * result is a Java keyword or not.
57       *
58       * @param javaName The candidate Java name.
59       * @return The cleaned Java name
60       *
61       * @throws IllegalArgumentException If the candidate Java name
62       *      contains only illegal characters.
63       *
64       * @see Character#isJavaIdentifierStart(int)
65       * @see Character#isJavaIdentifierPart(int)
66       */
67      public static String asJavaName(String javaName) {
68          int[] n = { 0 }; // expect to be one but increase
69                           // at each wrong first character found
70          int first = javaName.codePoints().filter(cp -> {
71              n[0]++;
72              return Character.isJavaIdentifierStart(cp);
73          }).findFirst().orElseThrow(() ->
74              new IllegalArgumentException(javaName + " is not a Java name."));
75          IntStream parts = javaName.codePoints().skip(n[0]).filter(
76              c -> Character.isJavaIdentifierPart(c));
77          int[] codepoints = IntStream.concat(
78              IntStream.of(first),
79              parts
80          ).toArray();
81          return new String(codepoints, 0, codepoints.length);
82      }
