1   package ml.alternet.scan;
2   
3   import java.io.IOException;
4   import java.io.Reader;
5   import java.util.Optional;
6   
7   import ml.alternet.facet.Trackable;
8   
9   /**
10   * A scanner that tracks the current line, column, and offset.
11   *
12   * @see Scanner#asTrackable()
13   *
14   * @author Philippe Poulard
15   */
16  public class TrackableScanner extends Scanner implements Trackable {
17  
18      static class Position extends Scanner.Cursor { // also hold a mark
19  
20          // track the current position
21          private long line = 1;
22          private long column = 0;
23          private long offset = 0;
24          // and the next position (read in advance)
25          private long nextLine = 1;
26          private long nextColumn = 0;
27          // allow tracking \r\n
28          private boolean previsousWasReturnChar = false;
29  
30          Position(int cursor, Position position) {
31              super(cursor);
32              this.line = position.line;
33              this.column = position.column;
34              this.offset = position.offset;
35              this.nextLine = position.nextLine;
36              this.nextColumn = position.nextColumn;
37              this.previsousWasReturnChar = position.previsousWasReturnChar;
38          }
39  
40          Position() {
41              super(-1); // current position is not marked
42          }
43  
44      }
45  
46      // the current position doesn't hold a mark
47      private Position position = new Position();
48  
49      // the underlying concrete scanner
50      private Scanner scanner;
51  
52      /**
53       * Create a new trackable scanner.
54       * The internal state of this trackable scanner is
55       * the one which is wrapped.
56       *
57       * @param scanner The scanner to wrap.
58       *
59       * @throws IOException When an I/O error occur.
60       */
61      public TrackableScanner(Scanner scanner) throws IOException {
62          this.scanner = scanner;
63          this.state = scanner.state;
64          this.state.source = this; // loop back to this allow trackable methods invokation
65          setNextLocation();
66      }
67  
68      private void setNextLocation() {
69          if ( hasNext() ) {
70              char c = (char) lookAhead();
71              if ( c == '\r' ) {
72                  this.position.nextLine++;
73                  this.position.nextColumn = 0;
74                  this.position.previsousWasReturnChar = true;
75              } else {
76                  if ( c == '\n' ) {
77                      if ( ! this.position.previsousWasReturnChar) {
78                          this.position.nextLine++;
79                          this.position.nextColumn = 0;
80                      }
81                  } else {
82                      this.position.nextColumn++;
83                  }
84                  this.position.previsousWasReturnChar = false;
85              }
86              // next offset is always offset + 1
87          }
88      }
89  
90      /**
91       * Create a new scanner with an initial position.
92       *
93       * @param scanner The scanner to wrap.
94       * @param line The initial line number.
95       * @param column The initial column number.
96       * @param offset The initial offset number.
97       *
98       * @throws IOException When an I/O error occur.
99       */
100     public TrackableScanner(Scanner scanner, long line, long column, long offset) throws IOException {
101         this(scanner);
102         this.position.line = line;
103         this.position.column = column;
104         this.position.offset = offset;
105     }
106 
107     @Override
108     public Optional<ml.alternet.misc.Position> getPosition() {
109         return Optional.of(ml.alternet.misc.Position
110             .$(this.position.column, this.position.line, this.position.offset));
111     }
112 
113     /**
114      * Read a character.
115      * The column and line are updated.
116      *
117      * @throws IOException When an I/O error occur.
118      */
119     @Override
120     public void read() throws IOException {
121         this.position.line = this.position.nextLine;
122         this.position.column = this.position.nextColumn;
123         this.scanner.read();
124         if (hasNext()) {
125             this.position.offset++;
126         }
127         setNextLocation();
128     }
129 
130     /**
131      * Mark the current position.
132      * The current column and line are saved.
133      *
134      * @see Reader#mark(int)
135      */
136     @Override
137     public void mark() {
138         this.scanner.mark();
139     }
140 
141     /**
142      * Cancel the current mark.
143      * The previous column and line are restored.
144      *
145      * @throws IllegalStateException When no mark have been previously saved.
146      */
147     @Override
148     public void cancel() throws IllegalStateException {
149         if (this.state.cursors.isEmpty()) {
150             throw new IllegalStateException( "Can't cancel the reading since no position was marked." );
151         } else {
152             this.position = (Position) this.state.cursors.peek();
153             // we should set this.position.mark = -1 but it is useless
154         }
155         this.scanner.cancel();
156     }
157 
158     /**
159      * Consume the characters read so far.
160      * The previous column and line are removed.
161      */
162     @Override
163     public void consume() {
164         this.scanner.consume(); // just drop the last saved position
165     }
166 
167     @Override
168     protected void push( int cursor ) {
169         // copy the current position and mark it with the cursor
170         Position pos = new Position(cursor, this.position);
171         this.state.cursors.push(pos);
172     }
173 
174     @Override
175     public Optional<Reader> getRemainder() throws IOException {
176         return this.scanner.getRemainder();
177     }
178 
179     @Override
180     public Optional<String> getRemainderString() throws IOException {
181         return this.scanner.getRemainderString();
182     }
183 
184 }
