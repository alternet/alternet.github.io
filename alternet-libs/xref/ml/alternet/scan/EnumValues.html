22  /**
23   * Wraps an enum class or a list of string values to a hierarchy
24   * of enum or string values, partitioned by characters (or common
25   * sequence of characters).
26   *
27   * This allow to look for an enum or string value without testing
28   * the same character several times (when it occurs).
29   *
30   * @author Philippe Poulard
31   *
32   * @param <T> The type of values : <code>String</code> or an enum class.
33   *
34   * @see EnumValues#nextValue(Scanner)
35   * @see EnumValues#from(Class)
36   * @see EnumValues#from(String...)
37   * @see EnumValues#from(Stream)
38   */
39  public class EnumValues<T> implements Readable<T>, Presentable { // is a set of EnumValues
40  
41      // the start index in the string value (val and values are complete strings, char may be partial)
42      int start = 0;
43      // a single terminal value
44      T val;
45      // a subset of values
46      Set<EnumValues<T>> values;
47      // the chars from index "start" common to all values
48      CharArray chars;
49  
50      /**
51       * Create a hierarchy of enum values by char.
52       *
53       * @param values The enum class.
54       *
55       * @return The enum values that holds all the hierarchy.
56       */
57      @SuppressWarnings({ "rawtypes" })
58      public static EnumValues<? extends Enum> from(Class<? extends Enum> values) {
59          Set<EnumValues<? extends Enum>> enumValues = Arrays.stream(values.getEnumConstants())
60                  .map(e -> new EnumValues<>(e))
61                  .collect(Collectors.toSet());
62          // dispatch by chars, start at index 0
63          @SuppressWarnings("unchecked")
64          EnumValues<? extends Enum> result = new EnumValues<>((Set<EnumValues<?>>) (Object)
65                                                enumValues); // odd cast to ensure to call the right constructor
66          return result.dispatch(0);
67      }

87      public static EnumValues<String> from(Stream<String> values) {
88          Set<EnumValues<String>> enumValues = values
89                  .map(e -> new EnumValues<>(e))
90                  .collect(Collectors.toSet());
91          // dispatch by chars, start at index 0
92          @SuppressWarnings("unchecked")
93          EnumValues<String> result = new EnumValues<>((Set<EnumValues<?>>) (Object)
94                                                enumValues); // odd cast to ensure to call the right constructor
95          return result.dispatch(0);

98      // constructor for the root
99      @SuppressWarnings("unchecked")
100     EnumValues(Set<EnumValues<?>> enumValues) { // doesn't work with Set<EnumValues<T>>
101         if (enumValues.size() == 1) {
102             // single value => set the terminal value
103             this.val = (T) enumValues.stream().findAny().get().val;
104         } else {
105             this.values = (Set<EnumValues<T>>) (Object) enumValues; // odd cast
106         }
107     }
108 
109     // constructor for terminal values
110     EnumValues(T val) {
111         // just wrap an enum value
112         this.val = val;
113     }
114 
115     // constructor for set of values to dispatch by char
116     EnumValues(char c, Set<EnumValues<T>> enumValues, int start) {
117         this.start = start; // start index of the character to take into account
118         if (enumValues.size() == 1) {
119             // single value => set the terminal value
120             this.val = enumValues.stream().findAny().get().val;
121         } else {
122             this.values = enumValues;
123             // all the values are starting by the same character
124             this.chars = new CharArray(new char[1], 0, 0); // allocate a single char
125             this.chars.append(c);
126             // additional chars may be appended later
127             // if it appears that the next chars are
128             // also common to all values -> see push() below
129         }
130     }
131 
132     // push a char in the current char sequence
133     void push(char c) {
134         // additional characters are common to the set of values
135         this.chars.append(c);
136     }

138     @Override
139     public String toString() {
140         return this.start == 0 ? "" : ("(" + this.start + ") ")
141              + this.chars == null ? "" : ("\"" + this.chars + "\"=")
142              + "\'" + (this.val == null ? "" : this.val) + "\'"
143              + ("" + this.values == null ? "[]" : (this.values + ""));
144     }
145 
146     // do we have the next chars from the input here ?
147     boolean hasNextChars(Scanner scanner) {
148         try {
149             if (this.chars == null) {
150                 // we have the terminal value
151                 if (this.start == 0) {
152                     return scanner.hasNextString(this.val.toString(), true);
153                 } else {
154                     // when previous chars were already tested...
155                     String nextChars = this.val.toString().substring(this.start);
156                     // ...we check the rest of the sequence
157                     return scanner.hasNextString(nextChars, true);
158                 }
159             } else if (this.chars.length() == 1) {
160                 // we have the set of values with just a common char
161                 return scanner.lookAhead() == this.chars.charAt(0);
162             } else {
163                 // we have the set of values with several common chars
164                 return scanner.hasNextString(this.chars, false);
165                             // "false" because it is partial and consumed by the caller
166             }
167         } catch (IOException e) {
168             return Thrower.doThrow(e);
169         }
170     }

172     /**
173      * Read the next enum value that belongs to this set of values.
174      *
175      * @param scanner The input to read.
176      *
177      * @return <code>empty</code> if not found in the input,
178      *      the actual enum value otherwise.
179      *
180      * @throws IOException When an I/O error occurs.
181      */
182     @Override
183     public Optional<T> nextValue(Scanner scanner) throws IOException {
184         return nextValuePart(scanner, 0); // part is complete from 0
185     }
186 
187     // look for the next enum value part in this (sub)tree at the given char pos
188     Optional<T> nextValuePart(Scanner scanner, int charPos) throws IOException {
189         Optional<T> result = Optional.empty();
190         Optional<EnumValues<T>> value = this.values.stream()
191             .filter(e -> e.hasNextChars(scanner))
192             .findFirst(); // because we may find 1 or 0
193         if (value.isPresent()) {
194             EnumValues<T> enumValue = value.get();
195             // try to find the longest match
196             if (enumValue.values == null) {
197                 // we have it
198                 result = Optional.of(enumValue.val);
199             } else {
200                 scanner.mark();
201                 // read as many chars as in the common chars (matched so far)...
202                 scanner.nextString(new StringConstraint.ReadLength(enumValue.chars.length()));
203                 // ...and go on parsing
204                 result = enumValue.nextValuePart(scanner, charPos + 1);
205                 if (result.isPresent()) {
206                     scanner.consume();
207                 } else {
208                     scanner.cancel();
209                 }
210             }
211             // if no longest match and a val is present use it
212             if (! result.isPresent() && enumValue.val != null) {
213                 String strValue = enumValue.val.toString();
214                 if (strValue.length() == charPos // terminal value, all characters were consumed
215                         || scanner.hasNextString(strValue.substring(charPos), true))
216                 {
217                     result = Optional.of(enumValue.val);
218                 }
219             }
220         }
221         return result;
222     }

224     // Dispatch enum value string by char at index.
225     //
226     // This method is called recursively at the
227     // initialization phase.
228     //
229     // No longer used when using the scanner
230     EnumValues<T> dispatch(int index) {
231         if (this.values == null) {
232             return this;
233         } else if (this.values.size() == 1) {
234             // a single enum for this set
235             return this.values.stream().findAny().get();
236         } else {
237             Map<Boolean, Set<EnumValues<T>>> dispatchable;
238             Map<Character, Set<EnumValues<T>>> enumByChar;
239             boolean sameString;
240             do {
241                 int i = index;
242                 dispatchable = this.values.stream().collect(
243                     Collectors.partitioningBy(e -> e.val.toString().length() > i, Collectors.toSet())
244                     // true -> can be dispatch
245                     // false -> can't be dispatch (too small string) -> at most one item in this partition
246                 );
247                 enumByChar = dispatchable.get(true).stream()
248                     .collect(
249                         // map by char at index
250                         Collectors.groupingBy(e -> e.val.toString().charAt(i),
251                             TreeMap::new,
252                             Collectors.toSet()
253                     )
254                 );
255                 // if all values are starting by the same char
256                 // and everything was dispatched, we can push that char and loop
257                 sameString = enumByChar.size() == 1 && dispatchable.get(false).isEmpty();
258                 if (sameString) {
259                     // forEach = forSingle
260                     enumByChar.forEach((c, e) -> this.push(c));
261                     index++;
262                 }
263             } while (sameString); // exit => we have a clean separation by char / or not dispatchable
264             int i = index;
265             // try to dispatch recursively each set-by-char
266             Set<EnumValues<T>> enumValues = enumByChar.entrySet().stream()
267                 //                            EnumValue<>(char,       Set<EnumValue>)
268                 .<EnumValues<T>> map(e -> new EnumValues<>(e.getKey(), e.getValue(), i))
269                 .map(e -> e.dispatch(i + 1)) // dispatch Set<EnumValue> for the next index
270                 .collect(Collectors.toSet());
271             if (dispatchable.get(false).isEmpty()) {
272                 if (enumValues.size() == 1) {
273                     // a single enum for this set
274                     return enumValues.stream().findAny().get();
275                 }
276             } else {
277                 // the non-dispatchable is the terminal value
278                 this.val = dispatchable.get(false).stream().findAny().get().val;
279             }
280             // it's a tree of enum by char
281             this.values = enumValues; // replace by what was dispatched
282             return this;
283             // note that this.val and this.values can be set together
284         }
285     }

287     @Override
288     public StringBuilder toPrettyString(StringBuilder buf) {
289         return values()
290             .map(o -> "'" + o.toString() + "'")
291             .sorted(Comparator.comparing(String::length).thenComparing(s -> s))
292             .collect(StringBuilderUtil.collectorOf("( ", " | ", " )", buf));
293     }

295     private static int FLAGS =
296               Spliterator.CONCURRENT | Spliterator.DISTINCT
297             | Spliterator.IMMUTABLE | Spliterator.NONNULL
298             | Spliterator.ORDERED;
299 
300     /**
301      * Return all the values inside.
302      *
303      * @return The values, in alphabetical order.
304      */
305     public Stream<T> values() {
306         // looks heavy, but it's lazy !
307         return Stream.concat(
308             // concat this.val if it exist...
309             Optional.ofNullable(this.val)
310                 .map(Stream::of)
311                 .orElseGet(Stream::empty),
312             // ...with this.values.forEach(EnumValues::values) if it exist
313             StreamSupport.stream(
314                 new Spliterators.AbstractSpliterator<T>(Long.MAX_VALUE, FLAGS) {
315                     Spliterator<T> stream = Optional.ofNullable(EnumValues.this.values)
316                         .map(values -> values.stream().flatMap(v -> v.values()))
317                         .orElseGet(() -> Stream.<T>empty())
318                         .spliterator();
319                     @Override
320                     public boolean tryAdvance(Consumer<? super T> action) {
321                         return stream.tryAdvance(action);
322                     }
323                 },
324                 false)
325         );
326     }
