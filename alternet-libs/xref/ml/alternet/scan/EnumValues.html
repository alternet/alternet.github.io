23  /**
24   * Wraps an enum class or a list of string values to a hierarchy
25   * of enum or string values, partitioned by characters (or common
26   * sequence of characters).
27   *
28   * This allow to look for an enum or string value without testing
29   * the same character several times (when it occurs).
30   *
31   * @author Philippe Poulard
32   *
33   * @param <T> The type of values : <code>String</code> or an enum class.
34   *
35   * @see EnumValues#nextValue(Scanner)
36   * @see EnumValues#from(Class)
37   * @see EnumValues#from(String...)
38   * @see EnumValues#from(Stream)
39   */
40  public class EnumValues<T> implements Readable<T>, Presentable { // is a set of EnumValues
41  
42      // the start index in the string value (val and values are complete strings, char may be partial)
43      int start = 0;
44      // a single terminal value
45      T val;
46      // a subset of values
47      Set<EnumValues<T>> values;
48      // the chars from index "start" common to all values
49      CharArray chars;
50  
51      /**
52       * Create a hierarchy of enum values by char.
53       *
54       * @param values The enum class.
55       * @param <T> The enum type.
56       *
57       * @return The enum values that holds all the hierarchy.
58       */
59      public static <T extends Enum<T>> EnumValues<T> from(Class<T> values) {
60          Set<EnumValues<T>> enumValues = Arrays.stream(values.getEnumConstants())
61              .map(e -> new EnumValues<>(e))
62              .collect(Collectors.toSet());
63          // dispatch by chars, start at index 0
64          @SuppressWarnings("unchecked")
65          EnumValues<T> result = new EnumValues<>((Set<EnumValues<?>>) (Object)
66                                     enumValues); // odd cast to ensure to call the right constructor
67          return result.dispatch(0).wrap();
68      }

99      EnumValues<T> wrap() {
100         if (this.chars != null        // when all values start by the same String
101             || this.values == null)  // or when we have a single value
102         {
103             // we are wrapping this in a set because when
104             // using nextValue() we are starting with the set
105             Set<EnumValues<T>> set = new HashSet<>();
106             set.add(this);
107             EnumValues<T> wrapper = new EnumValues<T>((T) null);
108             wrapper.values = set;
109             return wrapper;
110         } else {
111             return this;
112         }
113     }

115     // constructor for the root
116     @SuppressWarnings("unchecked")
117     EnumValues(Set<EnumValues<?>> enumValues) { // doesn't work with Set<EnumValues<T>>
118         if (enumValues.size() == 1) {
119             // single value => set the terminal value
120             this.val = (T) enumValues.stream().findAny().get().val;
121         } else {
122             this.values = (Set<EnumValues<T>>) (Object) enumValues; // odd cast
123         }
124     }

126     // constructor for terminal values
127     EnumValues(T val) {
128         // just wrap an enum value
129         this.val = val;
130     }
131 
132     // constructor for set of values to dispatch by char
133     EnumValues(char c, Set<EnumValues<T>> enumValues, int start) {
134         this.start = start; // start index of the character to take into account
135         if (enumValues.size() == 1) {
136             // single value => set the terminal value
137             this.val = enumValues.stream().findAny().get().val;
138         } else {
139             this.values = enumValues;
140             // all the values are starting by the same character
141             this.chars = new CharArray(new char[1], 0, 0); // allocate a single char
142             this.chars.append(c);
143             // additional chars may be appended later
144             // if it appears that the next chars are
145             // also common to all values -> see push() below
146         }
147     }
148 
149     // push a char in the current char sequence
150     void push(char c) {
151         if (this.chars == null) {
152             this.chars = new CharArray(new char[1], 0, 0); // allocate a single char
153         }
154         // additional characters are common to the set of values
155         this.chars.append(c);
156     }

158     @Override
159     public String toString() {
160         return (this.start == 0 ? "" : ("(" + this.start + ") "))
161              + (this.chars == null ? "" : ("\'" + this.chars + "\' -> "))
162              + (this.val == null ? "" : "\'" + this.val + "\'")
163              + (this.values == null ? "" : (this.values + ""));
164     }

166     // do we have the next chars from the input here ?
167     boolean hasNextChars(Scanner scanner) {
168         try {
169             if (this.chars == null) {
170                 // we have the terminal value
171                 if (this.start == 0) {
172                     return scanner.hasNextString(this.val.toString(), true);
173                 } else {
174                     // when previous chars were already tested...
175                     String nextChars = this.val.toString().substring(this.start);
176                     // ...we check the rest of the sequence
177                     return scanner.hasNextString(nextChars, true);
178                 }
179             } else if (this.chars.length() == 1) {
180                 // we have the set of values with just a common char
181                 return scanner.lookAhead() == this.chars.charAt(0);
182             } else {
183                 // we have the set of values with several common chars
184                 return scanner.hasNextString(this.chars, false);
185                             // "false" because it is partial and consumed by the caller
186             }
187         } catch (IOException e) {
188             return Thrower.doThrow(e);
189         }
190     }

192     /**
193      * Read the next enum value that belongs to this set of values.
194      *
195      * @param scanner The input to read.
196      *
197      * @return <code>empty</code> if not found in the input,
198      *      the actual enum value otherwise.
199      *
200      * @throws IOException When an I/O error occurs.
201      */
202     @Override
203     public Optional<T> nextValue(Scanner scanner) throws IOException {
204         return nextValuePart(scanner, 0); // part is complete from 0
205     }
206 
207     // look for the next enum value part in this (sub)tree at the given char pos
208     Optional<T> nextValuePart(Scanner scanner, int charPos) throws IOException {
209         Optional<T> result = Optional.empty();
210         Optional<EnumValues<T>> value = this.values.stream()
211             .filter(e -> e.hasNextChars(scanner))
212             .findFirst(); // because we may find 1 or 0
213         if (value.isPresent()) {
214             EnumValues<T> enumValue = value.get();
215             // try to find the longest match
216             if (enumValue.values == null) {
217                 // we have it
218                 result = Optional.of(enumValue.val);
219             } else {
220                 scanner.mark();
221                 // read as many chars as in the common chars (matched so far)...
222                 int l = scanner.skipNextString(new StringConstraint.ReadLength(enumValue.chars.length()));
223                 // ...and go on parsing
224                 result = enumValue.nextValuePart(scanner, charPos + l);
225                 if (result.isPresent()) {
226                     scanner.consume();
227                 } else {
228                     scanner.cancel();
229                 }
230             }
231             // if no longest match and a val is present use it
232             if (! result.isPresent() && enumValue.val != null) {
233                 String strValue = enumValue.val.toString();
234                 if (strValue.length() == charPos // terminal value, all characters were consumed
235                         || scanner.hasNextString(strValue.substring(charPos), true))
236                 {
237                     result = Optional.of(enumValue.val);
238                 }
239             }
240         }
241         return result;
242     }

244     // Dispatch enum value string by char at index.
245     //
246     // This method is called recursively at the
247     // initialization phase.
248     //
249     // No longer used when using the scanner
250     EnumValues<T> dispatch(int index) {
251         if (this.values == null) {
252             return this;
253         } else if (this.values.size() == 1) {
254             // a single enum for this set
255             return this.values.stream().findAny().get();
256         } else {
257             Map<Boolean, Set<EnumValues<T>>> dispatchable;
258             Map<Character, Set<EnumValues<T>>> enumByChar;
259             boolean sameString;
260             do {
261                 int i = index;
262                 dispatchable = this.values.stream().collect(
263                     Collectors.partitioningBy(e -> e.val.toString().length() > i, Collectors.toSet())
264                     // true -> can be dispatch
265                     // false -> can't be dispatch (too small string) -> at most one item in this partition
266                 );
267                 enumByChar = dispatchable.get(true).stream()
268                     .collect(
269                         // map by char at index
270                         Collectors.groupingBy(e -> e.val.toString().charAt(i),
271                             TreeMap::new,
272                             Collectors.toSet()
273                     )
274                 );
275                 // if all values are starting by the same char
276                 // and everything was dispatched, we can push that char and loop
277                 sameString = enumByChar.size() == 1 && dispatchable.get(false).isEmpty();
278                 if (sameString) {
279                     // forEach = forSingle
280                     enumByChar.forEach((c, e) -> this.push(c));
281                     index++;
282                 }
283             } while (sameString); // exit => we have a clean separation by char / or not dispatchable
284             int i = index;
285             // try to dispatch recursively each set-by-char
286             Set<EnumValues<T>> enumValues = enumByChar.entrySet().stream()
287                 //                            EnumValue<>(char,       Set<EnumValue>)
288                 .<EnumValues<T>> map(e -> new EnumValues<>(e.getKey(), e.getValue(), i))
289                 .map(e -> e.dispatch(i + 1)) // dispatch Set<EnumValue> for the next index
290                 .collect(Collectors.toSet());
291             if (dispatchable.get(false).isEmpty()) {
292                 if (enumValues.size() == 1) {
293                     // a single enum for this set
294                     return enumValues.stream().findAny().get();
295                 }
296             } else {
297                 // the non-dispatchable is the terminal value
298                 this.val = dispatchable.get(false).stream().findAny().get().val;
299             }
300             // it's a tree of enum by char
301             this.values = enumValues; // replace by what was dispatched
302             return this;
303             // note that this.val and this.values can be set together
304         }
305     }

307     @Override
308     public StringBuilder toPrettyString(StringBuilder buf) {
309         return values()
310             .map(o -> "'" + o.toString() + "'")
311             .sorted(Comparator.comparing(String::length).thenComparing(s -> s))
312             .collect(StringBuilderUtil.collectorOf("( ", " | ", " )", buf));
313     }

315     private static int FLAGS =
316               Spliterator.CONCURRENT | Spliterator.DISTINCT
317             | Spliterator.IMMUTABLE | Spliterator.NONNULL
318             | Spliterator.ORDERED;
319 
320     /**
321      * Return all the values inside.
322      *
323      * @return The values, in alphabetical order.
324      */
325     public Stream<T> values() {
326         // looks heavy, but it's lazy !
327         return Stream.concat(
328             // concat this.val if it exist...
329             Optional.ofNullable(this.val)
330                 .map(Stream::of)
331                 .orElseGet(Stream::empty),
332             // ...with this.values.forEach(EnumValues::values) if it exist
333             StreamSupport.stream(
334                 new Spliterators.AbstractSpliterator<T>(Long.MAX_VALUE, FLAGS) {
335                     Spliterator<T> stream = Optional.ofNullable(EnumValues.this.values)
336                         .map(values -> values.stream().flatMap(v -> v.values()))
337                         .orElseGet(() -> Stream.<T>empty())
338                         .spliterator();
339                     @Override
340                     public boolean tryAdvance(Consumer<? super T> action) {
341                         return stream.tryAdvance(action);
342                     }
343                 },
344                 false)
345         );
346     }
