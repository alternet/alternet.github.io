7   /**
8    * A number constraint is used by scanners to read numbers
9    * under conditions, such as limiting the number of total
10   * digits to read, or accepting negative numbers, forcing
11   * a number to be an integer, etc.
12   *
13   * @author Philippe Poulard
14   */
15  public interface NumberConstraint extends Constraint {

17      /**
18       * When there is no constraint, there is no stop condition
19       * (that always return true).
20       *
21       * <p>A number under no constraint is parsed until a character
22       * is unexpected in the input sequence.</p>
23       */
24      NumberConstraint NO_CONSTRAINT = new NumberConstraint() {
25          public boolean stopCondition(StringBuffer buf, int sourceIndex,
26                  int dotIndex, int exponentIndex, Scanner scanner)
27                  throws IOException {
28              return false;
29          }
30          public Class<? extends Number> getNumberType() {
31              return null;
32          }
33      };

185     /**
186      * Base class for checking constraints on number types.
187      *
188      * @author Philippe Poulard
189      */
190     abstract class NumberClassConstraint implements NumberConstraint {
191         public boolean checkInteger( Scanner scanner ) throws IOException {
192             return scanner.hasNextChar( ".Ee", false );
193 //            return scanner.lookAhead() == '.';
194         }
195         public boolean checkExponent( Scanner scanner ) throws IOException {
196             return scanner.hasNextChar( "Ee", false );
197         }
198         /**
199          * Check the digits of a buffer.
200          * 
201          * @param buf The buffer
202          * @param scanner The scanner
203          * @param min For negative values only. Inclusive.
204          * @param max For positive values only. Inclusive.
205          * @return <tt>true</tt> if the number is in the expected range, <tt>false</tt> otherwise.
206          * @throws IOException When an I/O error occurs.
207          */
208         public boolean checkDigits(StringBuffer buf, Scanner scanner, String min, String max) throws IOException {
209             int bl = buf.length();
210             boolean neg = bl > 0 && buf.charAt( 0 ) == '-';
211             if ( neg && min == null || ! neg && max == null ) {
212                 return false; // nothing to check
213             }
214             String number = "" + scanner.lookAhead();
215             // skip sign and leading zeroes
216             for (int i = neg?1:0; i < bl; i++ ) {
217                 if ( buf.charAt( i ) != '0' ) {
218                     number = buf.substring( i ) + scanner.lookAhead();
219                     break;
220                 }
221             }
222             int l = number.length();
223             if ( neg ) {
224                 // negative values, use min
225                 max = min.substring( 1 );
226             }
227             if ( l > max.length() ) {
228                 // too much digits
229                 return true;
230             } else if ( l < max.length() ) {
231                 // not too much digits
232                 return false;
233             } else {
234                 // same digits, tests them one by one until one is different
235                 for ( int i = 0 ; i < max.length() ; i++ ) {
236                     char nc = number.charAt( i );
237                     char mc = max.charAt( i );
238                     if ( nc > mc ) {
239                         return true;
240                     } else if ( nc < mc ) {
241                         return false;
242                     } // else continue
243                 }
244                 // all digits are eligible
245                 return false; // because inclusive
246             }
247         }
248     };

250     /**
251      * Evaluate the stop condition from the given parameters.
252      *
253      * <p>This condition is evaluated by the scanner to check
254      * if it has to append characters to the current buffer.</p>
255      *
256      * @param buf The buffer that receive the input characters
257      * 		so far. It can be used by this method to evaluate
258      * 		the stop condition.
259      * 		For example if negative numbers are not allowed, the
260      * 		first character of the buffer can't be "-".
261      * @param sourceIndex The number of characters read so far ;
262      * 		might be useful in certain stop conditions.
263      * @param dotIndex The index where the dot character was
264      * 		encountered, or -1 if not found so far.
265      * @param exponentIndex The index where the exponent character
266      * 		was encountered, or -1 if not found so far.
267      * @param scanner The scanner that reads the input.
268      * 		According to the parsing strategy, if the sequence
269      * 		of characters involved in the stop condition (if any)
270      * 		don't have to be consumed, the relevant methods of
271      * 		the scanner should be involved.
272      *
273      * @return <code>true</code> to indicate that the current scan
274      * 		must stop, <code>false</code> if more characters have
275      * 		to be read.
276      *
277      * @throws IOException When the scanner cause an error.
278      */
279     boolean stopCondition(StringBuffer buf, int sourceIndex, int dotIndex, int exponentIndex, Scanner scanner) throws IOException;

289     /**
290      * A constraint on numbers : after parsing the number, it will
291      * be of the type given.
292      *
293      * @author Philippe Poulard
294      */
295     class Type implements NumberConstraint {
296 
297         /** The number class. */
298         private Class<? extends Number> clazz;
299 
300         /**
301          *
302          * @param numberClass Used to enforce the number to be an
303          * 		instance of the class given. The class must implement
304          * 		{@link Number} and belong either to <tt>java.lang</tt>
305          * 		or to <tt>java.math</tt>.
306          */
307         public Type( Class<? extends Number> numberClass ) {
308             this.clazz = numberClass;
309         }
310 
311         /**
312          * Return the type of the number.
313          */
314         public Class<? extends Number> getNumberType() {
315             return this.clazz;
316         }
317 
318         /**
319          * No stop condition here : the string supplied must be
320          * of the type expected.
321          *
322          * @return <code>false</code>
323          */
324         public boolean stopCondition(StringBuffer buf, int sourceIndex,
325                 int dotIndex, int exponentIndex, Scanner scanner)
326                 throws IOException {
327             return false;
328         }
329     }
