3   import java.io.IOException;

5   /**
6    * A constraint used by scanners to read specific objects
7    * under conditions.
8    *
9    * @author Philippe Poulard
10   *
11   * @see UserData
12   */
13  public interface DataConstraint<T> extends Constraint {
14  
15      /**
16       * Evaluate the stop condition from the given parameters.
17       *
18       * <p>This condition is evaluated by the scanner to check
19       * if it has to append characters to the current buffer.</p>
20       *
21       * @param sourceIndex The number of characters read so far ;
22       * 		might be useful in certain stop conditions.
23       * @param userData Any useful data for building the target
24       * 		object. It is initialized to be empty by the
25       * 		scanner, and it can be updated by this method as
26       * 		one goes along. When the stop condition is realized,
27       * 		this method have to set the definitive target object
28       * 		that will be returned by the scanner
29       * 		(that can be <code>null</code>). Before the stop
30       * 		condition is realized, any intermediate value can be
31       * 		stored within.
32       * @param scanner The scanner that reads the input.
33       * 		According to the parsing strategy, if the sequence
34       * 		of characters involved in the stop condition (if any)
35       * 		don't have to be consumed, the relevant methods of
36       * 		the scanner should be involved.
37       *
38       * @return <code>true</code> to indicate that the current scan
39       * 		must stop, <code>false</code> if more characters have
40       * 		to be read.
41       *
42       * @throws IOException When the scanner cause an error.
43       */
44      boolean stopCondition(int sourceIndex, UserData<T> userData, Scanner scanner) throws IOException;
45  
46  }
