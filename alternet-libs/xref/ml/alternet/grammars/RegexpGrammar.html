42      Proxy ElementaryExp = $();
43  
44      boolean b1 = SimpleExp.is( ElementaryExp.oneOrMore() );

46      // mandatory groups are not processed as optional groups
47      // (optional groups must not fire characters as one goes along
48      // rather they have to ensure that the group does match)
49      Proxy ZeroOrMoreGroup = $();
50      Proxy OptionalGroup = $();
51      Proxy OneOrMoreGroup = $();
52      Proxy MandatoryGroup = $();
53      Rule Group = OptionalGroup.or(ZeroOrMoreGroup, OneOrMoreGroup);
54  
55      boolean b2 =  OptionalGroup.  is( LEFT_BRACE.seq(Exp, RIGHT_BRACE, OPTIONAL) ) // optional
56                  ^ ZeroOrMoreGroup.is( LEFT_BRACE.seq(Exp, RIGHT_BRACE, ZERO_OR_MORE) ) // optional
57                  ^ OneOrMoreGroup. is( LEFT_BRACE.seq(Exp, RIGHT_BRACE, ONE_OR_MORE ) ) // can fire characters
58                  ^ MandatoryGroup .is( LEFT_BRACE.seq(Exp, RIGHT_BRACE) ); // can fire characters

60      Token RANGE = is('-');
61      Proxy Chr = $();
62      Rule Range = Chr.seq(RANGE, Chr);
63  
64      Rule SetItem = Range.or(Chr);

67  /*
68      <RE>    ::=     <union> | <simple-RE>
69      <union>     ::= <RE> "|" <simple-RE>
70      <simple-RE>     ::=     <concatenation> | <basic-RE>
71      <concatenation>     ::= <simple-RE> <basic-RE>
72      <basic-RE>  ::= <star> | <plus> | <elementary-RE>
73      <star>  ::= <elementary-RE> "*"
74      <plus>  ::= <elementary-RE> "+"
75      <elementary-RE>     ::= <group> | <any> | <eos> | <char> | <set>
76      <group>     ::=     "(" <RE> ")"
77      <any>   ::=     "."
78      <eos>   ::=     "$"
79      <char>  ::=     any non metacharacter | "\" metacharacter
80      <set>   ::=     <positive-set> | <negative-set>
81      <positive-set>  ::=     "[" <set-items> "]"
82      <negative-set>  ::=     "[^" <set-items> "]"
83      <set-items>     ::=     <set-item> | <set-item> <set-items>
84      <set-items>     ::=     <range> | <char>
85      <range>     ::=     <char> "-" <char>
86  */
