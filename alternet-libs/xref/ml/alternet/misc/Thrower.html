3   /**
4    * A thrower can "cast" any exception as a RuntimeException
5    * without wrapping it.
6    *
7    * @author Philippe Poulard
8    */
9   public class Thrower {
10  
11      /**
12       * Throw an exception as a RuntimeException ;
13       * the thrown exception is NOT wrapped in a RuntimeException.
14       *
15       * This allow to throw a checked exception as an unchecked one.
16       *
17       * @param <T> Any return type
18       *
19       * @param e Any exception.
20       *
21       * @return <code>null</code> when the calling code is in a block
22       *      that require a value, it can return this call.
23       */
24      public static <T> T doThrow(Exception e) {
25          Thrower.<RuntimeException> throwAs(e);
26          return null;
27      }
28  
29      @SuppressWarnings("unchecked")
30      private static <E extends Exception> void throwAs(Exception e) throws E {
31          throw (E) e;
32      }
33  
34      /**
35       * Invoke a callable without throwing any checked exception : if the underlying
36       * callable throws a checked exception, it is cast to an unchecked exception.
37       *
38       * @param <T> Any return type
39       *
40       * @param callable The function to call
41       *
42       * @return The value.
43       */
44      public static <T> T safeCall(java.util.concurrent.Callable<T> callable) {
45          try {
46              return callable.call();
47          } catch (Exception e) {
48              return doThrow(e);
49          }
50      }
51  
52      /**
53       * Invoke a callable without throwing any checked exception : if the underlying
54       * callable throws a checked exception, it is cast to an unchecked exception.
55       *
56       * @param callable The function to call
57       */
58      public static void safeCall(Callable callable) {
59          try {
60              callable.call();
61          } catch (Exception e) {
62              doThrow(e);
63          }
64      }
65  
66      /**
67       * Like a callable, but without a return value,
68       * or like a java.lang.Runnable but with exception.
69       *
70       * @author Philippe Poulard
71       */
72      @FunctionalInterface
73      public interface Callable {
74  
75          /**
76           * Call something, or throws an exception if unable to do so.
77           *
78           * @throws Exception if unable to call.
79           */
80          void call() throws Exception;
81  
82      }
83  
84  }
