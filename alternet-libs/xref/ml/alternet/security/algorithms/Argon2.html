223         private void initMemory(Argon2 argon2) {
224             /* 2. Align memory size */
225             /* Minimum memoryBlocks = 8L blocks, where L is the number of lanes */
226             int memoryBlocks = argon2.getMemory();
227 
228             if (memoryBlocks < 2 * ARGON2_SYNC_POINTS * argon2.getLanes()) {
229                 memoryBlocks = 2 * ARGON2_SYNC_POINTS * argon2.getLanes();
230             }
231 
232             this.segmentLength = memoryBlocks / (argon2.getLanes() * ARGON2_SYNC_POINTS);
233             this.laneLength = segmentLength * ARGON2_SYNC_POINTS;
234             /* Ensure that all segments have equal length */
235             memoryBlocks = segmentLength * (argon2.getLanes() * ARGON2_SYNC_POINTS);
236 
237             this.memory = new Block[memoryBlocks];
238 
239             for(int i=0;i<memory.length;i++){
240                 memory[i] = new Block();
241             }
242         }

249         private void fillFirstBlocks(byte[] initialHash) {
250 
251             byte[] blockhash_bytes;
252 
253             for (int i=0;i<this.getLanes();i++) {
254 
255                 byte[] iBytes = LittleEndian.intToBytes(i);
256                 byte[] zeroBytes = LittleEndian.intToBytes(0);
257                 byte[] oneBytes = LittleEndian.intToBytes(1);
258 
259                 System.arraycopy(zeroBytes, 0, initialHash, ARGON2_PREHASH_DIGEST_LENGTH, 4);
260                 System.arraycopy(iBytes, 0, initialHash, ARGON2_PREHASH_DIGEST_LENGTH + 4, 4);
261                 blockhash_bytes = Functions.blake2bLong(initialHash, ARGON2_BLOCK_SIZE);
262                 int index = i * this.getLaneLength() + 0;
263                 this.memory[index].fromBytes(blockhash_bytes);
264 
265                 System.arraycopy(oneBytes, 0, initialHash, ARGON2_PREHASH_DIGEST_LENGTH, 4);
266                 blockhash_bytes = Functions.blake2bLong(initialHash, ARGON2_BLOCK_SIZE);
267                 this.memory[i * this.getLaneLength() + 1].fromBytes(blockhash_bytes);
268             }
269         }

275         public void finalize(Argon2 argon2) {
276 
277             if (argon2 != null) {
278 
279                 Block blockhash = new Block();
280                 blockhash.copyBlock(this.memory[this.getLaneLength() - 1]);
281 
282                 /* XOR the last blocks */
283                 for (int i=1; i<this.getLanes(); ++i) {
284                     int last_block_in_lane = i * this.getLaneLength() + (this.getLaneLength() - 1);
285                     blockhash.xorBlock(this.memory[last_block_in_lane]);
286                 }
287 
288                 /* Hash the result */
289                 byte[] blockhash_bytes = blockhash.toBytes();
290                 byte[] finalResult = Functions.blake2bLong(blockhash_bytes,
291                         argon2.getOutputLength());
292 
293                 argon2.setOutput(finalResult);
294 
295                 if(argon2.isClearMemory()) {
296                     this.clear();
297                 }
298             }
299         }

356     public static class Block {
357 
358         /* 128 * 8 Byte QWords */
359         public long[] v;
360 
361         public Block() {
362             v = new long[ARGON2_QWORDS_IN_BLOCK];
363         }
364 
365         public void fromBytes(byte[] input){
366             assert(input.length == ARGON2_BLOCK_SIZE);
367 
368             for(int i=0;i<v.length;i++){
369                 byte[] slice = Arrays.copyOfRange(input,i*8, (i+1)*8);
370                 v[i] = LittleEndian.bytesToLong(slice);
371             }
372         }
373 
374         public byte[] toBytes(){
375             byte[] result = new byte[ARGON2_BLOCK_SIZE];
376 
377             for(int i=0;i<v.length;i++){
378                 byte[] bytes = LittleEndian.longToBytes(v[i]);
379                 System.arraycopy(bytes, 0, result, i*bytes.length, bytes.length);
380             }
381 
382             return result;
383         }
384 
385         public void copyBlock(Block other){
386             System.arraycopy(other.v, 0, v, 0, v.length);
387         }
388 
389         public void xorBlock(Block other){
390             for(int i = 0; i< v.length; i++){
391                 v[i] = v[i] ^ other.v[i];
392             }
393         }
394 
395         @Override
396         public String toString() {
397             StringBuilder result = new StringBuilder();
398             for (long value : v) {
399                 result.append(StringUtil.getHex((LittleEndian.longToBytes(value)), false));
400             }
401 
402             return result.toString();
403         }

405         void clear() {
406             Arrays.fill(v, 0);
407         }

439         static void fillBlock(Block prevBlock, Block refBlock, Block nextBlock, boolean withXor) {
440 
441             Block blockR = new Block();
442             Block blockTmp = new Block();
443 
444             blockR.copyBlock(refBlock);
445             blockR.xorBlock(prevBlock);
446             blockTmp.copyBlock(blockR);
447 
448             if (withXor) {
449                 blockTmp.xorBlock(nextBlock);
450             }
451 
452             /* Apply Blake2 on columns of 64-bit words: (0,1,...,15) , then
453             (16,17,..31)... finally (112,113,...127) */
454             for (int i = 0; i < 8; ++i) {
455 
456                 Functions.roundFunction(blockR,
457                         16 * i, 16 * i + 1, 16 * i + 2,
458                         16 * i + 3, 16 * i + 4, 16 * i + 5,
459                         16 * i + 6, 16 * i + 7, 16 * i + 8,
460                         16 * i + 9, 16 * i + 10, 16 * i + 11,
461                         16 * i + 12, 16 * i + 13, 16 * i + 14,
462                         16 * i + 15
463                         );
464             }
465 
466             /* Apply Blake2 on rows of 64-bit words: (0,1,16,17,...112,113), then
467             (2,3,18,19,...,114,115).. finally (14,15,30,31,...,126,127) */
468             for (int i = 0; i < 8; i++) {
469 
470                 Functions.roundFunction(blockR,
471                         2 * i, 2 * i + 1, 2 * i + 16,
472                         2 * i + 17, 2 * i + 32, 2 * i + 33,
473                         2 * i + 48, 2 * i + 49, 2 * i + 64,
474                         2 * i + 65, 2 * i + 80, 2 * i + 81,
475                         2 * i + 96, 2 * i + 97, 2 * i + 112,
476                         2 * i + 113
477                         );
478 
479                 nextBlock.copyBlock(blockTmp);
480                 nextBlock.xorBlock(blockR);
481             }
482         }

487                 fillMemoryBlockSingleThreaded(instance);
488             } else {
489                 fillMemoryBlockMultiThreaded(instance);

493         private static void fillMemoryBlockSingleThreaded(Instance instance) {
494 
495             for (int i = 0; i < instance.getPasses(); i++) {
496                 for (int j = 0; j < ARGON2_SYNC_POINTS; j++) {
497                     for (int k = 0; k < instance.getLanes(); k++) {
498                         Position position = new Position(i, k, (byte) j, 0);
499                         Fillers.fillSegment(instance, position);
500                     }
501                 }
502             }
503         }
504 
505         private static void fillMemoryBlockMultiThreaded(Instance instance) {
506 
507             Thread[] threads = new Thread[instance.getLanes()];
508 
509             for (int i = 0; i < instance.getPasses(); i++) {
510                 for (int j = 0; j < ARGON2_SYNC_POINTS; j++) {
511 
512                     /* 2. Calling threads */
513                     for (int k = 0; k < instance.getLanes(); k++) {
514 
515                         /* 2.1 Join a thread if limit is exceeded */
516                         if (k >= instance.getThreads()) {
517                             try {
518                                 threads[k - instance.getThreads()].join();
519                             } catch (InterruptedException e) {
520                                 e.printStackTrace();
521                             }
522                         }
523 
524                         Position position = new Position(i, k, (byte) j, 0);
525 
526                         /* 2.2 Create thread */
527                         threads[k] = new Thread(() -> Fillers.fillSegment(instance, position));
528                         threads[k].start();
529                     }
530 
531                     /* 3. Joining remaining threads */
532                     for (int k = instance.getLanes() - instance.getThreads(); k < instance.getLanes(); k++) {
533                         try {
534                             threads[k].join();
535                         } catch (InterruptedException e) {
536                             e.printStackTrace();
537                         }
538                     }
539                 }
540             }
541         }

543         static void fillSegment(Instance instance, Position position) {
544             Block refBlock, currentBlock;
545             Block addressBlock = null, inputBlock = null, zeroBlock = null;
546             long pseudoRandom;
547             int refIndex, refLane;
548             int prevOffset, currentOffset;
549             int startingIndex;
550             boolean data_independent_addressing;
551 
552             if (instance == null) {
553                 return;
554             }
555 
556             data_independent_addressing =
557                     (instance.getType() == Type.argon2i) ||
558                     (instance.getType() == Type.argon2id && (position.pass == 0) &&
559                     (position.slice < Constants.ARGON2_SYNC_POINTS / 2));
560 
561             if (data_independent_addressing) {
562                 addressBlock = new Block();
563                 zeroBlock = new Block();
564                 inputBlock = new Block();
565 
566                 inputBlock.v[0] = position.pass;
567                 inputBlock.v[1] = position.lane;
568                 inputBlock.v[2] = position.slice;
569                 inputBlock.v[3] = instance.memory.length;
570                 inputBlock.v[4] = instance.getPasses();
571                 inputBlock.v[5] = instance.getType().ordinal();
572             }
573 
574             startingIndex = 0;
575 
576             if ((0 == position.pass) && (0 == position.slice)) {
577                 startingIndex = 2; /* we have already generated the first two blocks */
578 
579                 /* Don't forget to generate the first block of addresses: */
580                 if (data_independent_addressing) {
581                     nextAddresses(addressBlock, inputBlock, zeroBlock);
582                 }
583             }
584 
585             /* Offset of the current block */
586             currentOffset = position.lane * instance.getLaneLength() +
587                     position.slice * instance.getSegmentLength() + startingIndex;
588 
589             if (0 == currentOffset % instance.getLaneLength()) {
590                 /* Last block in this lane */
591                 prevOffset = currentOffset + instance.getLaneLength() - 1;
592             } else {
593                 /* Previous block */
594                 prevOffset = currentOffset - 1;
595             }
596 
597             for (int i = startingIndex; i < instance.getSegmentLength();
598                     ++i, ++currentOffset, ++prevOffset) {
599                 /*1.1 Rotating prev_offset if needed */
600                 if (currentOffset % instance.getLaneLength() == 1) {
601                     prevOffset = currentOffset - 1;
602                 }
603 
604 
605                 /* 1.2 Computing the index of the reference block */
606                 /* 1.2.1 Taking pseudo-random value from the previous block */
607                 if (data_independent_addressing) {
608                     if (i % Constants.ARGON2_ADDRESSES_IN_BLOCK == 0) {
609                         nextAddresses(addressBlock, inputBlock, zeroBlock);
610                     }
611                     pseudoRandom = addressBlock.v[i % Constants.ARGON2_ADDRESSES_IN_BLOCK];
612                 } else {
613                     pseudoRandom = instance.memory[prevOffset].v[0];
614                 }
615 
616                 /* 1.2.2 Computing the lane of the reference block */
617                 refLane = (int) (((pseudoRandom >>> 32)) % instance.getLanes());
618 
619 
620                 if ((position.pass == 0) && (position.slice == 0)) {
621                     /* Can not reference other lanes yet */
622                     refLane = position.lane;
623                 }
624 
625                 /* 1.2.3 Computing the number of possible reference block within the
626                  * lane.
627                  */
628 
629                 position.index = i;
630                 refIndex = indexAlpha(instance, position, pseudoRandom,
631                         refLane == position.lane);
632 
633 
634                 /* 2 Creating a new block */
635                 refBlock = instance.memory[(instance.getLaneLength()) * refLane + refIndex];
636                 currentBlock = instance.memory[currentOffset];
637 
638                 if (Constants.ARGON2_VERSION_10 == instance.getVersion()) {
639                     /* version 1.2.1 and earlier: overwrite, not XOR */
640                     Fillers.fillBlock(instance.memory[prevOffset], refBlock, currentBlock, false);
641                 } else {
642                     if (0 == position.pass) {
643                         Fillers.fillBlock(instance.memory[prevOffset], refBlock,
644                                 currentBlock, false);
645                     } else {
646                         Fillers.fillBlock(instance.memory[prevOffset], refBlock,
647                                 currentBlock, true);
648                     }
649                 }
650             }
651         }

653         private static void nextAddresses(Block address_block, Block input_block, Block zero_block) {
654             input_block.v[6]++;
655             Fillers.fillBlock(zero_block, input_block, address_block, false);
656             Fillers.fillBlock(zero_block, address_block, address_block, false);
657         }

659         private static int indexAlpha(Instance instance, Position position, long pseudoRand,
660                 boolean same_lane) {
661             /*
662              * Pass 0:
663              *      This lane : all already finished segments plus already constructed
664              * blocks in this segment
665              *      Other lanes : all already finished segments
666              * Pass 1+:
667              *      This lane : (SYNC_POINTS - 1) last segments plus already constructed
668              * blocks in this segment
669              *      Other lanes : (SYNC_POINTS - 1) last segments
670              */
671             int reference_area_size;
672             long relative_position;
673             int start_position, absolute_position;
674 
675             if (0 == position.pass) {
676                 /* First pass */
677                 if (0 == position.slice) {
678                     /* First slice */
679                     reference_area_size =
680                             position.index - 1; /* all but the previous */
681                 } else {
682                     if (same_lane) {
683                         /* The same lane => add current segment */
684                         reference_area_size =
685                                 position.slice * instance.getSegmentLength() +
686                                 position.index - 1;
687                     } else {
688                         reference_area_size =
689                                 position.slice * instance.getSegmentLength() +
690                                 ((position.index == 0) ? (-1) : 0);
691                     }
692                 }
693             } else {
694                 /* Second pass */
695                 if (same_lane) {
696                     reference_area_size = instance.getLaneLength() -
697                             instance.getSegmentLength() + position.index -
698                             1;
699                 } else {
700                     reference_area_size = instance.getLaneLength() -
701                             instance.getSegmentLength() +
702                             ((position.index == 0) ? (-1) : 0);
703                 }
704             }
705 
706 
707             /* 1.2.4. Mapping pseudoRand to 0..<reference_area_size-1> and produce
708              * relative position */
709 
710             /* long in java is a signed datatype
711              * we need to convert it to the unsigned value
712              */
713 
714             //            relative_position = pseudoRand & 0xFFFFFFFFL;
715             relative_position = pseudoRand << 32 >>> 32;
716             relative_position = relative_position * relative_position;
717             relative_position = relative_position >>> 32;
718                     relative_position = reference_area_size - 1 - (reference_area_size * relative_position >>> 32);
719 
720                     /* 1.2.5 Computing starting position */
721                     start_position = 0;
722 
723                     if (0 != position.pass) {
724                         start_position = (position.slice == Constants.ARGON2_SYNC_POINTS - 1)
725                                 ? 0
726                                         : (position.slice + 1) * instance.getSegmentLength();
727                     }
728 
729                     /* 1.2.6. Computing absolute position */
730                     absolute_position = (int) (start_position + relative_position) %
731                             instance.getLaneLength(); /* absolute position */
732 
733                     return absolute_position;
734         }

738     public static class Functions {
739 
740         /* H0
741             argon2 -> 64 byte (ARGON2_PREHASH_DIGEST_LENGTH)
742                     -> 72 byte (ARGON2_PREHASH_SEED_LENGTH)
743          */
744         public static byte[] initialHash(Argon2 argon2) {
745 
746             Blake2b.Param params = new Blake2b.Param()
747                     .setDigestLength(Constants.ARGON2_PREHASH_DIGEST_LENGTH);
748 
749             final Blake2b blake2b = Blake2b.Digest.newInstance(params);
750 
751             blake2b.update(LittleEndian.intToBytes(argon2.getLanes()));
752             blake2b.update(LittleEndian.intToBytes(argon2.getOutputLength()));
753             blake2b.update(LittleEndian.intToBytes(argon2.getMemory()));
754             blake2b.update(LittleEndian.intToBytes(argon2.getIterations()));
755             blake2b.update(LittleEndian.intToBytes(argon2.getVersion()));
756             blake2b.update(LittleEndian.intToBytes(argon2.getType().ordinal()));
757 
758             byte[] pwLengthBytes = LittleEndian.intToBytes(argon2.getPassword().length);
759             blake2b.update(pwLengthBytes);
760             if(argon2.getPassword() != null){
761                 blake2b.update(argon2.getPassword());
762                 argon2.clearPassword();
763             }
764 
765             blake2b.update(LittleEndian.intToBytes(argon2.getSalt().length));
766             if(argon2.getSalt() != null){
767                 blake2b.update(argon2.getSalt());
768             }
769 
770             blake2b.update(LittleEndian.intToBytes(argon2.getSecret() == null ? 0 : argon2.getSecret().length));
771             if(argon2.getSecret() != null){
772                 blake2b.update(argon2.getSecret());
773                 argon2.clearSecret();
774             }
775 
776             blake2b.update(LittleEndian.intToBytes(argon2.getAdditional() == null ? 0 : argon2.getAdditional().length));
777             if(argon2.getAdditional() != null){
778                 blake2b.update(argon2.getAdditional());
779                 argon2.clearAdditional();
780             }
781 
782             byte[] blake2hash = blake2b.digest();
783 
784             // at this point, blacke2b cleaned every sensible data
785 
786             assert(blake2hash.length == 64);
787 
788             byte[] result = new byte[ARGON2_PREHASH_SEED_LENGTH];
789             System.arraycopy(blake2hash, 0, result, 0, ARGON2_PREHASH_DIGEST_LENGTH);
790 
791             assert(result.length == 72);
792             assert(result[64] == 0);
793             assert(result[71] == 0);
794 
795             return result;
796         }

799         /* blake2_long - variable length hash function
800          * H'
801           (H0 || 0 || i) 72 byte -> 1024 byte
802          */
803         public static byte[] blake2bLong(byte[] input, int outlen) {
804 
805             byte[] result = new byte[outlen];
806             byte[] outlen_bytes = LittleEndian.intToBytes(outlen);
807 
808             int blake2BOutbytes = 64;
809 
810             if (outlen <= blake2BOutbytes) {
811                 Blake2b.Param params = new Blake2b.Param()
812                         .setDigestLength(outlen);
813 
814                 final Blake2b blake2b = Blake2b.Digest.newInstance(params);
815                 blake2b.update(outlen_bytes);
816                 blake2b.update(input);
817 
818                 result = blake2b.digest();
819 
820             } else {
821                 int toProduce;
822                 byte[] outBuffer;
823                 byte[] inBuffer = new byte[blake2BOutbytes];
824 
825                 Blake2b.Param params = new Blake2b.Param()
826                         .setDigestLength(blake2BOutbytes);
827 
828                 final Blake2b blake2b = Blake2b.Digest.newInstance(params);
829 
830                 blake2b.update(outlen_bytes);
831                 blake2b.update(input);
832 
833                 outBuffer = blake2b.digest();
834 
835                 System.arraycopy(outBuffer, 0, result, 0, blake2BOutbytes / 2);
836 
837                 int position = blake2BOutbytes / 2;
838                 toProduce = outlen - blake2BOutbytes / 2;
839 
840                 while (toProduce > blake2BOutbytes) {
841 
842                     System.arraycopy(outBuffer, 0, inBuffer, 0, blake2BOutbytes);
843 
844                     outBuffer = blake2b(inBuffer, blake2BOutbytes,
845                             null);
846 
847                     System.arraycopy(outBuffer, 0, result, position, blake2BOutbytes / 2);
848 
849 
850                     position += blake2BOutbytes / 2;
851                     toProduce -= blake2BOutbytes / 2;
852                 }
853 
854                 System.arraycopy(outBuffer, 0, inBuffer, 0, blake2BOutbytes);
855 
856                 outBuffer = blake2b(inBuffer, toProduce, null);
857                 System.arraycopy(outBuffer, 0, result, position, toProduce);
858             }
859 
860             assert(result.length == outlen);
861 
862             return result;
863         }

865         private static byte[] blake2b(byte[] input, int outlen, byte[] key){
866 
867             Blake2b.Param params = new Blake2b.Param()
868                     .setDigestLength(outlen);
869 
870             if(key != null)
871                 params.setKey(key);
872 
873             final Blake2b blake2b = Blake2b.Digest.newInstance(params);
874             blake2b.update(input);
875 
876             return blake2b.digest();
877         }

880                 int v0, int v1, int v2, int v3,
881                 int v4, int v5, int v6, int v7,
882                 int v8, int v9, int v10, int v11,

896         private static void G(Block block, int a, int b, int c, int d) {
897             fBlaMka(block, a, b);
898             rotr64(block, d, a, 32);
899 
900             fBlaMka(block, c, d);
901             rotr64(block, b, c, 24);
902 
903             fBlaMka(block, a, b);
904             rotr64(block, d, a, 16);
905 
906             fBlaMka(block, c, d);
907             rotr64(block, b, c, 63);
908         }

910         /*designed by the Lyra PHC team */
911         /* a <- a + b + 2*aL*bL
912          * + == addition modulo 2^64
913          * aL = least 32 bit */
914         private static void fBlaMka(Block block, int x, int y) {
915             final long m = 0xFFFFFFFFL;
916             long xy = (block.v[x] & m) * (block.v[y] & m);
917 
918             block.v[x] =  block.v[x] + block.v[y] + 2 * xy;
919         }
920 
921         private static void rotr64(Block block, int v, int w, long c) {
922             long temp = block.v[v] ^ block.v[w];
923             block.v[v] = (temp >>> c) | (temp << (64 - c));
924         }

976             static void validateInput(Argon2 argon2){

979                 if(argon2.getLanes() < MIN_PARALLELISM || argon2.getThreads() < MIN_PARALLELISM)
