28      /**
29       * Encrypts into crypt (Unix) code.
30       *
31       * @param salt the salt to be used
32       * @param keyword the key to be encrypted
33       * @return the encrypted bytes (without the salt)
34       */
35      public static byte[] encrypt(int salt, long keyword) {
36          long constdatablock = 0L; /* encryption constant */
37          byte[] cryptresult = new byte[11]; /* encrypted result */
38          long[] KS = des_setkey(keyword);
39          long rsltblock = des_cipher(constdatablock, salt, 25, KS);
40  
41          cryptresult[10] = (byte) ((((int) rsltblock) << 2) & 0x3f);
42          rsltblock >>= 4;
43          for (int i = 10; --i >= 0;) {
44              cryptresult[i] = (byte) (((int) rsltblock) & 0x3f);
45              rsltblock >>= 6;
46          }
47          return cryptresult;
48      }

123     static {
124 
125         byte[] perm = new byte[64];
126         byte[] temp = new byte[64];
127 
128         // PC1ROT - bit reverse, then PC1, then Rotate, then PC2
129         for (int i = 0; i < 64; i++) {
130             perm[i] = (byte) 0;
131         }
132 
133         for (int i = 0; i < 64; i++) {
134             int k;
135             if ((k = PC2[i]) == 0) continue;
136             k += Rotates[0] - 1;
137             if ((k % 28) < Rotates[0]) k -= 28;
138             k = PC1[k];
139             if (k > 0) {
140                 k--;
141                 k = (k | 0x07) - (k & 0x07);
142                 k++;
143             }
144             perm[i] = (byte) k;
145         }
146         init_perm(PC1ROT, perm, 8);
147 
148         // PC2ROT - PC2 inverse, then Rotate, then PC2
149         for (int j = 0; j < 2; j++) {
150             int k;
151             for (int i = 0; i < 64; i++) {
152                 perm[i] = temp[i] = 0;
153             }
154             for (int i = 0; i < 64; i++) {
155                 if ((k = PC2[i]) == 0) continue;
156                 temp[k - 1] = (byte) (i + 1);
157             }
158             for (int i = 0; i < 64; i++) {
159                 if ((k = PC2[i]) == 0) {
160                     continue;
161                 }
162                 k += j;
163                 if ((k % 28) <= j) {
164                     k -= 28;
165                 }
166                 perm[i] = temp[k];
167             }
168             init_perm(PC2ROT[j], perm, 8);
169         }
170 
171         // Bit reverse, intial permupation, expantion
172         for (int i = 0; i < 8; i++) {
173             for (int j = 0; j < 8; j++) {
174                 int k = (j < 2) ? 0 : IP[ExpandTr[i * 6 + j - 2] - 1];
175                 if (k > 32) {
176                     k -= 32;
177                 } else if (k > 0) {
178                     k--;
179                 }
180                 if (k > 0) {
181                     k--;
182                     k = (k | 0x07) - (k & 0x07);
183                     k++;
184                 }
185                 perm[i * 8 + j] = (byte) k;
186             }
187         }
188 
189         init_perm(IE3264, perm, 8);
190 
191         // Compression, final permutation, bit reverse
192         for (int i = 0; i < 64; i++) {
193             int k = IP[CIFP[i] - 1];
194             if (k > 0) {
195                 k--;
196                 k = (k | 0x07) - (k & 0x07);
197                 k++;
198             }
199             perm[k - 1] = (byte) (i + 1);
200         }
201 
202         init_perm(CF6464, perm, 8);
203 
204         // SPE table
205         for (int i = 0; i < 48; i++) {
206             perm[i] = P32Tr[ExpandTr[i] - 1];
207         }
208         for (int t = 0; t < 8; t++) {
209             for (int j = 0; j < 64; j++) {
210                 int k = (((j >> 0) & 0x01) << 5) | (((j >> 1) & 0x01) << 3)
211                         | (((j >> 2) & 0x01) << 2)
212                         | (((j >> 3) & 0x01) << 1)
213                         | (((j >> 4) & 0x01) << 0)
214                         | (((j >> 5) & 0x01) << 4);
215                 k = S[t][k];
216                 k = (((k >> 3) & 0x01) << 0) | (((k >> 2) & 0x01) << 1) | (((k >> 1) & 0x01) << 2) | (((k >> 0) & 0x01) << 3);
217                 for (int i = 0; i < 32; i++) {
218                     temp[i] = 0;
219                 }
220                 for (int i = 0; i < 4; i++) {
221                     temp[4 * t + i] = (byte) ((k >> i) & 0x01);
222                 }
223                 long kk = 0;
224                 for (int i = 24; --i >= 0;) {
225                     kk = ((kk << 1) | ((long) temp[perm[i] - 1]) << 32 | (temp[perm[i + 24] - 1]));
226                 }
227                 SPE[t][j] = to_six_bit(kk);
228             }
229         }
230     }

232     /**
233      * Returns the transposed and split code of a 24-bit code into a 4-byte
234      * code, each having 6 bits.
235      */
236     private static int to_six_bit(int num) {
237         return (((num << 26) & 0xfc000000) | ((num << 12) & 0xfc0000) | ((num >> 2) & 0xfc00) | ((num >> 16) & 0xfc));
238     }
239 
240     /**
241      * Returns the transposed and split code of two 24-bit code into two 4-byte
242      * code, each having 6 bits.
243      */
244     private static long to_six_bit(long num) {
245         return (((num << 26) & 0xfc000000fc000000L) | ((num << 12) & 0xfc000000fc0000L) | ((num >> 2) & 0xfc000000fc00L) | ((num >> 16) & 0xfc000000fcL));
246     }

248     /**
249      * Returns the permutation of the given 64-bit code with the specified
250      * permutataion table.
251      */
252     private static long perm6464(long c, long[][] p) {
253         long out = 0L;
254         for (int i = 8; --i >= 0;) {
255             int t = (int) (0x00ff & c);
256             c >>= 8;
257             long tp = p[i << 1][t & 0x0f];
258             out |= tp;
259             tp = p[(i << 1) + 1][t >> 4];
260             out |= tp;
261         }
262         return out;
263     }

265     /**
266      * Returns the permutation of the given 32-bit code with the specified
267      * permutataion table.
268      */
269     private static long perm3264(int c, long[][] p) {
270         long out = 0L;
271         for (int i = 4; --i >= 0;) {
272             int t = (0x00ff & c);
273             c >>= 8;
274             long tp = p[i << 1][t & 0x0f];
275             out |= tp;
276             tp = p[(i << 1) + 1][t >> 4];
277             out |= tp;
278         }
279         return out;
280     }

282     /**
283      * Returns the key schedule for the given key.
284      */
285     private static long[] des_setkey(long keyword) {
286         long K = perm6464(keyword, PC1ROT);
287         long[] KS = new long[16];
288         KS[0] = K & ~0x0303030300000000L;
289 
290         for (int i = 1; i < 16; i++) {
291             KS[i] = K;
292             K = perm6464(K, PC2ROT[Rotates[i] - 1]);
293 
294             KS[i] = K & ~0x0303030300000000L;
295         }
296         return KS;
297     }

299     /**
300      * Returns the DES encrypted code of the given word with the specified
301      * environment.
302      */
303     private static long des_cipher(long in, int salt, int num_iter, long[] KS) {
304         salt = to_six_bit(salt);
305         long L = in;
306         long R = L;
307         L &= 0x5555555555555555L;
308         R = (R & 0xaaaaaaaa00000000L) | ((R >> 1) & 0x0000000055555555L);
309         L = ((((L << 1) | (L << 32)) & 0xffffffff00000000L) | ((R | (R >> 32)) & 0x00000000ffffffffL));
310 
311         L = perm3264((int) (L >> 32), IE3264);
312         R = perm3264((int) (L & 0xffffffff), IE3264);
313 
314         while (--num_iter >= 0) {
315             for (int loop_count = 0; loop_count < 8; loop_count++) {
316                 long kp;
317                 long B;
318                 long k;
319 
320                 kp = KS[(loop_count << 1)];
321                 k = ((R >> 32) ^ R) & salt & 0xffffffffL;
322                 k |= (k << 32);
323                 B = (k ^ R ^ kp);
324 
325                 L ^= (SPE[0][(int) ((B >> 58) & 0x3f)] ^ SPE[1][(int) ((B >> 50) & 0x3f)]
326                       ^ SPE[2][(int) ((B >> 42) & 0x3f)]
327                       ^ SPE[3][(int) ((B >> 34) & 0x3f)]
328                       ^ SPE[4][(int) ((B >> 26) & 0x3f)]
329                       ^ SPE[5][(int) ((B >> 18) & 0x3f)]
330                       ^ SPE[6][(int) ((B >> 10) & 0x3f)] ^ SPE[7][(int) ((B >> 2) & 0x3f)]);
331 
332                 kp = KS[(loop_count << 1) + 1];
333                 k = ((L >> 32) ^ L) & salt & 0xffffffffL;
334                 k |= (k << 32);
335                 B = (k ^ L ^ kp);
336 
337                 R ^= (SPE[0][(int) ((B >> 58) & 0x3f)] ^ SPE[1][(int) ((B >> 50) & 0x3f)]
338                       ^ SPE[2][(int) ((B >> 42) & 0x3f)]
339                       ^ SPE[3][(int) ((B >> 34) & 0x3f)]
340                       ^ SPE[4][(int) ((B >> 26) & 0x3f)]
341                       ^ SPE[5][(int) ((B >> 18) & 0x3f)]
342                       ^ SPE[6][(int) ((B >> 10) & 0x3f)] ^ SPE[7][(int) ((B >> 2) & 0x3f)]);
343             }
344             // swap L and R
345             L ^= R;
346             R ^= L;
347             L ^= R;
348         }
349         L = ((((L >> 35) & 0x0f0f0f0fL) | (((L & 0xffffffff) << 1) & 0xf0f0f0f0L)) << 32 | (((R >> 35) & 0x0f0f0f0fL) | (((R & 0xffffffff) << 1) & 0xf0f0f0f0L)));
350 
351         L = perm6464(L, CF6464);
352 
353         return L;
354     }

356     /**
357      * Initializes the given permutation table with the mapping table.
358      */
359     private static void init_perm(long[][] perm, byte[] p, int chars_out) {
360         for (int k = 0; k < chars_out * 8; k++) {
361 
362             int l = p[k] - 1;
363             if (l < 0) continue;
364             int i = l >> 2;
365             l = 1 << (l & 0x03);
366             for (int j = 0; j < 16; j++) {
367                 int s = ((k & 0x07) + ((7 - (k >> 3)) << 3));
368                 if ((j & l) != 0x00) {
369                     perm[i][j] |= (1L << s);
370                 }
371             }
372         }
373     }
