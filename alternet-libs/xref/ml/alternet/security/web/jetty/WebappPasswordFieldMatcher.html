46  /**
47   * A matcher based on a Jetty web application.
48   *
49   * <p>The Web application has to be configured as follow :</p>
50   * <pre>
51   *&lt;context-param&gt;
52   *    &lt;param-name&gt;ml.alternet.security.web.config.formFields&lt;/param-name&gt;
53   *    &lt;param-value&gt;
54   *          /doRegister.html?pwd&amp;confirmPwd,
55   *          /doUpdatePassword.html?oldPwd&amp;newPwd&amp;confirmPwd
56   *    &lt;/param-value&gt;
57   *&lt;/context-param&gt;
58   *&lt;context-param&gt;
59   *    &lt;param-name&gt;ml.alternet.security.web.config.authenticationMethod&lt;/param-name&gt;
60   *    &lt;param-value&gt;Basic&lt;/param-value&gt;
61   *&lt;/context-param&gt;
62   * </pre>
63   * <p>The former context parameter consist of a comma-separated sequence of
64   * paths relative to the context path ; those paths are the target
65   * destination to which form data are posted, and that indicates in
66   * their query string the fields of the form parameters that contains
67   * the passwords to capture.</p>
68   * <p>The latter context parameter indicates the authentication method ;
69   * setting "Basic" indicates to capture passwords
70   * during HTTP Basic Authentication ; setting "Form" indicates to capture
71   * passwords during HTTP Form Authentication.</p>
72   *
73   * @author Philippe Poulard
74   */
75  public class WebappPasswordFieldMatcher implements PasswordFieldMatcher, FormLimit {

82      /**
83       * Create a password field matcher for a Jetty Webapp.
84       *
85       * @param wac The Jetty Webapp context.
86       */
87      public WebappPasswordFieldMatcher(WebAppContext wac) {
88          servletContextGetter = req -> wac.getServletContext();
89          formLimitGetter = () -> new FormLimit() {
90              @Override
91              public int getMaxFormContentSize() {
92                  return wac.getMaxFormContentSize();
93              }
94              @Override
95              public int getMaxFormKeys() {
96                  return wac.getMaxFormKeys();
97              }
98          };
99          addLifeCycleListener(wac);
100     }

102     /**
103      * Create a password field matcher for a Jetty Server.
104      *
105      * @param server The Jetty Server.
106      */
107     public WebappPasswordFieldMatcher(Server server) {
108         servletContextGetter = req -> {
109             // lookup for webapps in tree handlers and in beans
110             return lookupForWebApps(server)
111                 .sorted(Comparator.<WebAppContext> comparingInt(wac -> wac.getContextPath().length()).reversed())
112                  //                                          "/" is last, otherwise it always match
113                 .filter(wac -> {
114                     HttpServletRequest hreq = ((HttpServletRequest) req);
115                     if (hreq.getContextPath() != null && hreq.getContextPath().length() > 0) {
116                         // when WebappPasswordFieldMatcher.this.login(this, username, request)
117                         // is called, it appears that _pathInfo is null, but the context path is not
118                         return hreq.getContextPath().equals(wac.getContextPath());
119                     } else if (hreq.getPathInfo() != null) {
120                         // no other fields are available
121                         return hreq.getPathInfo().startsWith(wac.getContextPath());
122                     } else {
123                         return hreq.getRequestURI().startsWith(wac.getContextPath());
124                     }
125                 })
126                 .findFirst()
127                 .orElseThrow(() -> {
128                     // Uh ?
129                     LOG.warn("UNABLE TO FIND A WEB APP CONTEXT FOR THIS REQUEST", req);
130                     return new IllegalStateException("UNABLE TO FIND A WEB APP CONTEXT FOR THIS REQUEST " + req);
131                 })
132                 .getServletContext();
133         };
134         formLimitGetter = () -> new FormLimit() {
135             @Override
136             public int getMaxFormContentSize() {
137                 Integer mfcs = (Integer) server.getAttribute("org.eclipse.jetty.server.Request.maxFormContentSize");
138                 if (mfcs == null) {
139                     return -1;
140                 } else {
141                     return mfcs;
142                 }
143             }
144             @Override
145             public int getMaxFormKeys() {
146                 Integer mfk = (Integer) server.getAttribute("org.eclipse.jetty.server.Request.maxFormKeys");
147                 if (mfk == null) {
148                     return -1;
149                 } else {
150                     return mfk;
151                 }
152             }
153         };
154         server.addLifeCycleListener(new LifeCycle.Listener() {
155             @Override
156             public void lifeCycleStopping(LifeCycle event) { }
157             @Override
158             public void lifeCycleStopped(LifeCycle event) { }
159             @Override
160             public void lifeCycleStarting(LifeCycle event) {
161                 lookupForWebApps(server)
162                     .forEach(wac -> addLifeCycleListener(wac));
163             }
164             @Override
165             public void lifeCycleStarted(LifeCycle event) { }
166             @Override
167             public void lifeCycleFailure(LifeCycle event, Throwable cause) { }
168         });
169     }
170 
171     private void addLifeCycleListener(WebAppContext wac) {
172         wac.addLifeCycleListener(new LifeCycle.Listener() {
173             @Override
174             public void lifeCycleStopping(LifeCycle event) {
175                 ServletContext sc = wac.getServletContext();
176                 FormFieldConfiguration.reset(sc);
177                 AuthenticationMethod.reset(sc);
178             }
179             @Override
180             public void lifeCycleStopped(LifeCycle event) { }
181             @Override
182             public void lifeCycleStarting(LifeCycle event) {
183 //                am = getAuthenticationMethod();

185                     wac.getSecurityHandler().setAuthenticatorFactory(
186                         newAuthenticatorFactory()
187                     );
188 //                }
189             }
190             @Override
191             public void lifeCycleStarted(LifeCycle event) { }
192             @Override
193             public void lifeCycleFailure(LifeCycle event, Throwable cause) { }
194         });
195     }

197     private Stream<WebAppContext> lookupForWebApps(Server server) {
198         return Stream.concat(
199             server.getBeans(WebAppContext.class).stream(),
200             Stream.of(server.getHandler())
201                 .flatMap(h -> {
202                     if (h instanceof HandlerContainer) {
203                         return Arrays.stream(
204                             ((HandlerContainer) h).getChildHandlersByClass(WebAppContext.class)
205                         )
206                         .map(handler -> (WebAppContext) handler);
207                     } else if (h instanceof WebAppContext) {
208                         return Stream.of((WebAppContext) h);
209                     } else {
210                         return Stream.empty();
211                     }
212                 })
213         );
214     }

216     /**
217      * Login with the captured password.
218      *
219      * @param loginAuthenticator For retrieving the login service
220      * @param username The username.
221      * @param request The request that contains the captured password
222      * @return The user identity, or <code>null</code>.
223      */
224     public UserIdentity login(LoginAuthenticator loginAuthenticator, String username, ServletRequest request) {
225         // "password" contains "*****"
226         // the pwd has been captured before, let's retrieve it
227         Credentials credentials = getAuthenticationMethod(request).getCredentials(request);
228         if (credentials.getUserName() == null) {
229             credentials.withUser(username);
230         }
231         // use "pwd" instead of "password"
232         UserIdentity user = loginAuthenticator.getLoginService().login(username, credentials);
233         return user;
234     }

236     /**
237      * Return an authenticator that login with the captured
238      * Password class instead of the supplied password field
239      * that has been filled with '*'.
240      *
241      * @return By default, a subclass of BasicAuthenticator ;
242      *      see implementation for details.
243      */
244     protected Factory newAuthenticatorFactory() {
245         return new DefaultAuthenticatorFactory() {
246             @Override
247             public Authenticator getAuthenticator(Server server, ServletContext context,
248                     AuthConfiguration configuration, IdentityService identityService, LoginService loginService)
249             {
250                 // do the normal stuff, see the parent class
251 
252                 String auth = configuration.getAuthMethod();
253                 Authenticator authenticator = null;
254 
255                 if (auth == null || Constraint.__BASIC_AUTH.equalsIgnoreCase(auth)) {
256                     authenticator = new BasicAuthenticator() {
257                         @Override
258                         public UserIdentity login(String username, Object password, ServletRequest request) {
259                             UserIdentity user = WebappPasswordFieldMatcher.this.login(this, username, request);
260                             if (user != null) {
261                                 renewSession((HttpServletRequest) request, (request instanceof Request
262                                         ? ((Request) request).getResponse() : null));
263                                 // the parent class doesn't call that, but if it is not present
264                                 // the session is not created (and I don't know why)
265                                 ((HttpServletRequest) request).getSession(true); // ensure to have a session
266                                 return user;
267                             }
268                             return null;
269                         }
270                     };
271                 } else if (Constraint.__DIGEST_AUTH.equalsIgnoreCase(auth)) {
272                     authenticator = new DigestAuthenticator();
273                 } else if (Constraint.__FORM_AUTH.equalsIgnoreCase(auth)) {
274                     authenticator = new FormAuthenticator() {
275                         @Override
276                         public UserIdentity login(String username, Object password, ServletRequest request) {
277                             UserIdentity user = WebappPasswordFieldMatcher.this.login(this, username, request);
278                             if (user != null) {
279                                 renewSession((HttpServletRequest) request, (request instanceof Request
280                                         ? ((Request) request).getResponse() : null));
281                             }
282                             if (user != null) {
283                                 HttpSession session = ((HttpServletRequest) request).getSession(true);
284                                 // NOTE : "pwd" is invalidate() when leaving the stack
285                                 // "password" just contains "*****" and won't be usable for
286                                 //        session serialization/deserialization (used for login)
287                                 // IT IS A BAD PRACTICE TO SERIALIZE PASSWORDS (like Jetty can do)
288                                 Authentication cached = new SessionAuthentication(getAuthMethod(), user, password);
289                                 session.setAttribute(SessionAuthentication.__J_AUTHENTICATED, cached);
290                             }
291                             return user;
292                         }
293                     };
294                 } else if ( Constraint.__SPNEGO_AUTH.equalsIgnoreCase(auth) ) {
295                     authenticator = new SpnegoAuthenticator();
296                 } else if ( Constraint.__NEGOTIATE_AUTH.equalsIgnoreCase(auth) ) { // see Bug #377076
297                     authenticator = new SpnegoAuthenticator(Constraint.__NEGOTIATE_AUTH);
298                 } else if (Constraint.__CERT_AUTH.equalsIgnoreCase(auth)
299                         || Constraint.__CERT_AUTH2.equalsIgnoreCase(auth))
300                 {
301                     authenticator = new ClientCertAuthenticator();
302                 }
303                 return authenticator;
304             }
305         };
306     }

308     @Override
309     public Optional<List<String>> matches(HttpServletRequest request) {
310         // IMPL NOTE : the ServletContext can't be extract from the request
311         // because Jetty didn't set it yet
312         ServletContext sc = servletContextGetter.apply(request);
313         return FormFieldConfiguration.matches(sc, request);
314     }
315 
316     @Override
317     public AuthenticationMethod getAuthenticationMethod(ServletRequest request) {
318         ServletContext sc = servletContextGetter.apply(request);
319         return AuthenticationMethod.extract(sc);
320     }
