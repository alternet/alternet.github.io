4    * Little Endian subject to code inlining.

8   public class LittleEndian {
9   
10      public static int readInt(final byte[] b, int off) {
11          int v0
12              = (b [ off++ ] & 0xFF );
13          v0 |= (b [ off++ ] & 0xFF ) <<  8;
14          v0 |= (b [ off++ ] & 0xFF ) << 16;
15          v0 |= (b [ off   ]        ) << 24;
16          return v0;
17      }
18  
19      /** Little endian - byte[] to long */
20      public static long readLong(final byte[] b, int off) {
21          long v0
22              = ((long)b [ off++ ] & 0xFF );
23          v0 |= ((long)b [ off++ ] & 0xFF ) <<  8;
24          v0 |= ((long)b [ off++ ] & 0xFF ) << 16;
25          v0 |= ((long)b [ off++ ] & 0xFF ) << 24;
26          v0 |= ((long)b [ off++ ] & 0xFF ) << 32;
27          v0 |= ((long)b [ off++ ] & 0xFF ) << 40;
28          v0 |= ((long)b [ off++ ] & 0xFF ) << 48;
29          v0 |= ((long)b [ off   ]        ) << 56;
30          return v0;
31      }
32  
33      /** */
34      /** Little endian - long to byte[] */
35      public static final void writeLong(long v, final byte[] b, final int off) {
36          b [ off ]     = (byte) v; v >>>= 8;
37          b [ off + 1 ] = (byte) v; v >>>= 8;
38          b [ off + 2 ] = (byte) v; v >>>= 8;
39          b [ off + 3 ] = (byte) v; v >>>= 8;
40          b [ off + 4 ] = (byte) v; v >>>= 8;
41          b [ off + 5 ] = (byte) v; v >>>= 8;
42          b [ off + 6 ] = (byte) v; v >>>= 8;
43          b [ off + 7 ] = (byte) v;
44      }
45  
46      /** Little endian - int to byte[] */
47      public static void writeInt(int v, final byte[] b, final int off) {
48          b [ off ]     = (byte) v; v >>>= 8;
49          b [ off + 1 ] = (byte) v; v >>>= 8;
50          b [ off + 2 ] = (byte) v; v >>>= 8;
51          b [ off + 3 ] = (byte) v;
52      }
53  
54      public static long bytesToLong(byte[] b) {
55          return readLong(b, 0);
56      }
57  
58      public static byte[] intToBytes(int i) {
59          byte[] result = new byte[4];
60          writeInt(i, result, 0);
61          return result;
62      }
63  
64      public static byte[] longToBytes(long l) {
65          byte[] result = new byte[8];
66          writeLong(l, result, 0);
67          return result;
68      }
