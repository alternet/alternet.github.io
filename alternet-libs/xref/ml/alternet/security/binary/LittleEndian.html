3   /**
4    * Little Endian subject to code inlining.
5    *
6    * @author Philippe Poulard
7    */
8   public final class LittleEndian {
9   
10      private LittleEndian() { }
11  
12      /**
13       * Little endian - byte[] to int
14       *
15       * @param b The input byte
16       * @param off The offset where to start reading
17       * @return A little endian int
18       */
19      public static int readInt(final byte[] b, int off) {
20          int v0
21              =  b [ off++ ] & 0xFF;
22          v0 |= (b [ off++ ] & 0xFF ) <<  8;
23          v0 |= (b [ off++ ] & 0xFF ) << 16;
24          v0 |= (b [ off   ]        ) << 24;
25          return v0;
26      }
27  
28      /**
29       * Little endian - byte[] to long
30       *
31       * @param b The input byte
32       * @param off The offset where to start reading
33       * @return A little endian long
34       */
35      public static long readLong(final byte[] b, int off) {
36          long v0
37              =  (long) b [ off++ ] & 0xFF;
38          v0 |= ((long) b [ off++ ] & 0xFF ) <<  8;
39          v0 |= ((long) b [ off++ ] & 0xFF ) << 16;
40          v0 |= ((long) b [ off++ ] & 0xFF ) << 24;
41          v0 |= ((long) b [ off++ ] & 0xFF ) << 32;
42          v0 |= ((long) b [ off++ ] & 0xFF ) << 40;
43          v0 |= ((long) b [ off++ ] & 0xFF ) << 48;
44          v0 |= ((long) b [ off   ]        ) << 56;
45          return v0;
46      }
47  
48      /**
49       * Little endian - long to byte[]
50       *
51       * @param v The value
52       * @param b The output byte
53       * @param off The offset where to start writing
54       */
55      public static void writeLong(long v, final byte[] b, final int off) {
56          b [ off ]     = (byte) v; v >>>= 8;
57          b [ off + 1 ] = (byte) v; v >>>= 8;
58          b [ off + 2 ] = (byte) v; v >>>= 8;
59          b [ off + 3 ] = (byte) v; v >>>= 8;
60          b [ off + 4 ] = (byte) v; v >>>= 8;
61          b [ off + 5 ] = (byte) v; v >>>= 8;
62          b [ off + 6 ] = (byte) v; v >>>= 8;
63          b [ off + 7 ] = (byte) v;
64      }
65  
66      /**
67       * Little endian - int to byte[]
68       *
69       * @param v The value
70       * @param b The output byte
71       * @param off The offset where to start writing
72       */
73      public static void writeInt(int v, final byte[] b, final int off) {
74          b [ off ]     = (byte) v; v >>>= 8;
75          b [ off + 1 ] = (byte) v; v >>>= 8;
76          b [ off + 2 ] = (byte) v; v >>>= 8;
77          b [ off + 3 ] = (byte) v;
78      }
79  
80      /**
81       * Little endian - byte[] to long
82       *
83       * @param b The input byte
84       * @return A little endian long
85       */
86      public static long bytesToLong(byte[] b) {
87          return readLong(b, 0);
88      }
89  
90      /**
91       * Little endian - int to byte[]
92       *
93       * @param i The input int
94       * @return The bytes
95       */
96      public static byte[] intToBytes(int i) {
97          byte[] result = new byte[4];
98          writeInt(i, result, 0);
99          return result;
100     }
101 
102     /**
103      * Little endian - long to byte[]
104      *
105      * @param l The input long
106      * @return The bytes
107      */
108     public static byte[] longToBytes(long l) {
109         byte[] result = new byte[8];
110         writeLong(l, result, 0);
111         return result;
112     }
