11  import ml.alternet.security.auth.hashers.impl.PBKDF2Hasher;

16  /**
17   * LDAP hash formats specified by RFC 2307 including standard formats {MD5}, {SMD5}, {SHA}, {SSHA}
18   * and extensions. These schemes range from somewhat to very insecure, and should not be used except when required.
19   *
20   * <h1>Configuration</h1>
21   *
22   * You can examine the configuration of each value
23   * <a href="https://github.com/alternet/alternet.ml/blob/master/security-auth/src/main/java/ml/alternet/security/auth/hashers/CurlyBracesCryptFormatHashers.java">here</a>.
24   *
25   * @author Philippe Poulard
26   */
27  public enum CurlyBracesCryptFormatHashers implements Supplier<Hasher.Builder> {
28  
29      /**
30       * LDAP’s plain MD5 format.
31       *
32       * <p>"<tt>password</tt>" -&gt; "<tt>{MD5}X03MO1qnZdYdgyfeuILPmQ==</tt>"</p>
33       */
34      MD5 {
35          @Override
36          public Builder get() {
37              return Hasher.builder()
38                  .setClass(MessageHasher.class)
39                  .setAlgorithm("MD5")
40                  .setEncoding(BytesEncoder.base64)
41                  .setFormatter(CurlyBracesCryptFormat.CryptFormatter.INSTANCE);
42          }
43      },
44  
45      /**
46       * LDAP’s plain SHA1 format.
47       *
48       * <p>"<tt>password</tt>" -&gt; "<tt>{SHA}W6ph5Mm5Pz8GgiULbPgzG37mj9g=</tt>"</p>
49       */
50      SHA {
51          @Override
52          public Builder get() {
53              return Hasher.builder()
54                  .setClass(MessageHasher.class)
55                  .setAlgorithm("SHA1")
56                  .setEncoding(BytesEncoder.base64)
57                  .setFormatter(CurlyBracesCryptFormat.CryptFormatter.INSTANCE);
58          }
59      },
60  
61      /**
62       * LDAP’s salted MD5 format with 4 bytes salt.
63       *
64       * <p>"<tt>password</tt>" -&gt; "<tt>{SMD5}jNoSMNY0cybfuBWiaGlFw3Mfi/U=</tt>"</p>
65       */
66      SMD5 {
67          @Override
68          public Builder get() {
69              return Hasher.builder()
70                  .setClass(SaltedMessageHasher.class)
71                  .setAlgorithm("MD5")
72                  .setEncoding(BytesEncoder.base64)
73                  .setSaltByteSize(4)
74                  .setFormatter(CurlyBracesCryptFormat.SaltedCryptFormatter.INSTANCE);
75          }
76      },
77  
78      /**
79       * LDAP’s salted SHA1 format with 4 bytes salt.
80       *
81       * <p>"<tt>password</tt>" -&gt; "<tt>{SSHA}pKqkNr1tq3wtQqk+UcPyA3HnA2NsU5NJ</tt>"</p>
82       */
83      SSHA {
84          @Override
85          public Builder get() {
86              return Hasher.builder()
87                  .setClass(SaltedMessageHasher.class)
88                  .setAlgorithm("SHA1")
89                  .setEncoding(BytesEncoder.base64)
90                  .setSaltByteSize(4)
91                  .setFormatter(CurlyBracesCryptFormat.SaltedCryptFormatter.INSTANCE);
92          }
93      },
94  
95      /**
96       * PBKDF2 with SHA1 algorithm.
97       *
98       * <p>"<tt>password</tt>" -&gt; "<tt>{PBKDF2}131000$tLbWWssZ45zzfi9FiDEmxA$dQlpmhY4dGvmx4MOK/uOj/WU7Lg</tt>"</p>
99       */
100     PBKDF2 {
101         @Override
102         public Builder get() {
103             return Hasher.builder()
104                 .setClass(PBKDF2Hasher.class)
105                 .setAlgorithm("PBKDF2WithHmacSHA1")
106                 .setEncoding(BytesEncoder.abase64)
107                 .setSaltByteSize(24)
108                 .setHashByteSize(20)
109                 .setIterations(29000)
110                 .setFormatter(CurlyBracesCryptFormat.IterativeSaltedFormatter.INSTANCE);
111         }
112     },
113 
114     PBKDF2_SHA256 {
115         @Override
116         public Builder get() {
117             return PBKDF2.get()
118                 .setHashByteSize(32)
119                 .setAlgorithm("PBKDF2WithHmacSHA256");
120         }
121     },
122 
123     PBKDF2_SHA512 {
124         @Override
125         public Builder get() {
126             return PBKDF2.get()
127                 .setHashByteSize(64)
128                 .setAlgorithm("PBKDF2WithHmacSHA512");
129         }
130     },
131 
132     /**
133      * Alternate method for storing plaintext passwords by using
134      * the identifying prefix <tt>{plaintext}</tt>.
135      */
136     plaintext {
137         @SuppressWarnings({ "rawtypes", "unchecked" })
138         @Override
139         public Builder get() {
140             Builder b = PlainTextCryptFormat.get();
141             b.setFormatter(new CryptFormatterWrapper(b.getFormatter()));
142             return b;
143         }
144     };
145 
146     @Override
147     public abstract Hasher.Builder get();
148 
149     private CurlyBracesCryptFormatHashers() {
150         EnumUtil.replace(this, s -> s.replace("_", "-"));
151     }
152 
153 }
