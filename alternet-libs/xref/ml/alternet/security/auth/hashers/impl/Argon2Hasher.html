39      @Override
40      public Hasher configureWithCrypt(String crypt) {
41          // the hash part is used to set the hash byte size
42          Argon2Hasher hr = this;
43          CryptParts parts= hr.getConfiguration().getFormatter().parse(crypt, hr);
44          if (parts.hash != null) {
45              hr = (Argon2Hasher) hr.getBuilder().setHashByteSize(parts.hash.length)
46                  .use(null) // this avoid loooooooop
47                  .build();
48          }
49          return hr;
50      }

52      @Override
53      public byte[] encrypt(Credentials credentials, Argon2Parts parts) {
54          ByteBuffer bb;
55          try (Password.Clear pwd = credentials.getPassword().getClearCopy()) {
56              bb = SafeBuffer.encode(CharBuffer.wrap(pwd.get()), getConfiguration().getCharset());
57          }
58          byte[] passwordb = SafeBuffer.getData(bb);
59          Configuration conf = parts.hr.getConfiguration();
60  //        int outputLength = parts.hash == null ? conf.getHashByteSize() : parts.hash.length;
61          int outputLength = conf.getHashByteSize();
62          byte[] hash = Argon2Bridge.$.encrypt(passwordb, parts.salt, parts.memoryCost, parts.timeCost, parts.parallelism,
63                  Argon2Bridge.Type.valueOf(conf.getVariant()), parts.version,
64                  outputLength, parts.data, getSecret(parts.keyid));
65  
66          Arrays.fill(passwordb, (byte) 0);
67  
68          return hash;
69      }

156     /**
157      * The Argon2 hash format is defined by the argon2 reference implementation.
158      * It’s compatible with the PHC Format and Modular Crypt Format, and uses
159      * $argon2i$, $argon2d$ and $argon2id$ as it’s identifying prefixes for all
160      * its strings.
161      *
162      * An example hash (of password) is:
163      *
164      * <code>$argon2i$v=19$m=512,t=3,p=2$c29tZXNhbHQ$SqlVijFGiPG+935vDSGEsA</code>
165      *
166      * This string has the format <code>$argon2X$v=V$m=M,t=T,p=P[,data=DATA]$salt$digest</code>, where:
167      *
168      * <ul>
169      * <li>X is either i or d, depending on the argon2 variant (i in the example).</li>
170      * <li>V is an integer representing the argon2 revision. the value (when rendered into
171      * hexadecimal) matches the argon2 version (in the example, v=19 corresponds to 0x13,
172      * or Argon2 v1.3).</li>
173      * <li>M is an integer representing the variable memory cost, in kibibytes (512kib in the example).</li>
174      * <li>T is an integer representing the variable time cost, in linear iterations. (3 in the example).</li>
175      * <li>P is a parallelization parameter, which controls how much of the hash calculation is
176      * parallelization (2 in the example).</li>
177      * <li>DATA is an optional additional amount of data (omitted in the example)</li>
178      * <li>salt - this is the base64-encoded version of the raw salt bytes passed into the Argon2
179      * function (c29tZXNhbHQ in the example).</li>
180      * <li>digest - this is the base64-encoded version of the raw derived key bytes returned from
181      * the Argon2 function. Argon2 supports a variable checksum size, though the hashes
182      * will typically be 16 bytes, resulting in a 22 byte digest (SqlVijFGiPG+935vDSGEsA in the
183      * example).</li>
184      * </ul>
185      *
186      * All integer values are encoded uses ascii decimal, with no leading zeros.
187      * All byte strings are encoded using the standard base64 encoding, but without any trailing padding (“=”) chars.
188      *
189      * Note
190      *
191      * The v=version$ segment was added in Argon2 v1.3; older version Argon2 v1.0 hashes may not include this portion.
192      * The algorithm used by all of these schemes is deliberately identical and simple: The password is encoded
193      * into UTF-8 if not already encoded, and handed off to the Argon2 function. A specified number of bytes
194      * (16 byte default) returned result are encoded as the checksum.
195      */
196     public static final CryptFormatter<Argon2Parts> ARGON2_FORMATTER = new CryptFormatter<Argon2Hasher.Argon2Parts>() {
197 
198         @Override
199         public Argon2Parts parse(String crypt, Hasher hr) {
200             Argon2Parts parts = new Argon2Parts(hr);
201             String[] stringParts = crypt.split("\\$");
202             if (! stringParts[1].equals(parts.hr.getConfiguration().getVariant())) {
203                 hr = parts.hr.getBuilder().setVariant(stringParts[1]).build();
204                 parts.hr = hr;
205             }
206             int versionPresent = 0;
207             if (stringParts[2].startsWith("v=")) {
208                 versionPresent = 1;
209                 parts.version = Integer.parseInt(stringParts[2].substring(2));
210             }
211             BytesEncoding encoding = hr.getConfiguration().getEncoding();
212             if (stringParts.length > 2 + versionPresent && ! StringUtil.isVoid(stringParts[2 + versionPresent])) {
213                 String[] params = stringParts[2 + versionPresent].split(",");
214                 if (params.length > 0 && params[0].startsWith("m=")) {
215                     parts.memoryCost = Integer.parseInt(params[0].substring(2));
216                 }
217                 if (params.length > 1 && params[1].startsWith("t=")) {
218                     parts.timeCost = Integer.parseInt(params[1].substring(2));
219                 }
220                 if (params.length > 2 && params[2].startsWith("p=")) {
221                     parts.parallelism = Integer.parseInt(params[2].substring(2));
222                 }
223                 if (params.length > 3) {
224                     int keyPresent = 0;
225                     if (params[3].startsWith("keyid=")) {
226                         keyPresent = 1;
227                         parts.keyid = encoding.decode(params[3].substring(6));
228                     }
229                     if (params.length > 3+keyPresent && params[3+keyPresent].startsWith("data=")) {
230                         parts.data = encoding.decode(params[3+keyPresent].substring(5));
231                     }
232                 }
233             }
234             if (stringParts.length > 3 + versionPresent && ! StringUtil.isVoid(stringParts[3 + versionPresent])) {
235                 parts.salt = encoding.decode(stringParts[3 + versionPresent]);
236             }
237             if (stringParts.length > 4 + versionPresent && ! StringUtil.isVoid(stringParts[4 + versionPresent])) {
238                 parts.hash = encoding.decode(stringParts[4 + versionPresent]);
239             }
240             return parts;
241         }
242 
243         @Override
244         public String format(Argon2Parts parts) {
245             StringBuffer crypt = new StringBuffer(60);
246             crypt.append("$")
247                 .append(parts.hr.getConfiguration().getVariant())
248                 .append("$v=")
249                 .append(Integer.toString(parts.version))
250                 .append("$m=")
251                 .append(Integer.toString(parts.memoryCost))
252                 .append(",t=")
253                 .append(Integer.toString(parts.timeCost))
254                 .append(",p=")
255                 .append(Integer.toString(parts.parallelism));
256             BytesEncoding encoding = parts.hr.getConfiguration().getEncoding();
257             if (parts.keyid != null) {
258                 crypt.append(",keyid=")
259                 .append(encoding.encode(parts.keyid));
260             }
261             if (parts.data != null) {
262                 crypt.append(",data=")
263                     .append(encoding.encode(parts.data));
264             }
265             crypt.append("$");
266             crypt.append(encoding.encode(parts.salt));
267             if (parts.hash != null && parts.hash.length > 0) {
268                 crypt.append("$")
269                     .append(encoding.encode(parts.hash));
270             }
271             return crypt.toString();
272         }
273 
274         @Override
275         public CryptFormat getCryptFormat() {
276             return new ModularCryptFormat();
277         }
278     };

293         /**
294          * Computes an Argon2 hash.
295          *
296          * @param passwordb the password to encrypt
297          * @param salt the salt
298          * @param memoryCost the memory cost, -1 for the default
299          * @param timeCost the time cost, -1 for the default
300          * @param parallelism the parallelism, -1 for the default
301          * @param variant "argon2i", "argon2i", "argon2id"
302          * @param version v=19 corresponds to 0x13, or Argon2 v1.3 ; can also be 0x10
303          * @param outputLength -1 for the default which is 32 bytes
304          * @param additional Maybe null
305          * @param secret Maybe null
306          *
307          * @return The hash.
308          */
309         byte[] encrypt(byte[] passwordb, byte[] salt, int memoryCost, int timeCost, int parallelism,
310                 Type variant, int version, int outputLength, byte[] additional, byte[] secret);

313          * The singleton implementation.

315         static Argon2Bridge $ = argon2();

327     public static class Bridge implements Argon2Bridge {
328         @Override
329         public byte[] encrypt(byte[] passwordb, byte[] salt, int memoryCost, int timeCost, int parallelism,
330                 Type type, int version, int outputLength, byte[] additional, byte[] secret) {
331             Argon2 argon2 = new Argon2();
332 
333             if (memoryCost != -1) {
334                 argon2.setMemoryInKiB(memoryCost);
335             }
336             if (timeCost != -1) {
337                 argon2.setIterations(timeCost);
338             }
339             if (parallelism != -1) {
340                 argon2.setParallelism(parallelism);
341             }
342             argon2.setType(type);
343             if (version != -1) {
344                 argon2.setVersion(version);
345             }
346             if (outputLength != -1) {
347                 argon2.setOutputLength(outputLength);
348             }
349             if (additional != null) {
350                 argon2.setAdditional(additional);
351             }
352             if (secret != null) {
353                 argon2.setSecret(secret);
354             }
355             return argon2.hash(passwordb, salt);
356         }
