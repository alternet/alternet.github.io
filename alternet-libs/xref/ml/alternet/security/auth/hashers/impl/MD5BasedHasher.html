17  /**
18   * The MD5 based BSD password algorithm 1 and apr1 (Apache variant).
19   *
20   * Usually in the Modular Crypt Format :
21   * <ol>
22   * <li>Apache MD5 crypt format :<br>
23   *      <tt>$apr1$jgwedrkq$jzeetEHMGal5H0SUFDMEl1</tt></li>
24   * <li>Crypt MD5 :<br>
25   *      <tt>$1$3iuE5z/b$JHyXMzQOIq3cl6WlEMoZC.</tt></li>
26   * </ol>
27   *
28   * @author Philippe Poulard
29   */
30  public class MD5BasedHasher extends HasherBase<SaltedParts> {
31  
32      private static final String APR1 = "apr1";
33  
34      static boolean isApacheVariant(Hasher hr) {
35          return APR1.equals(hr.getConfiguration().getVariant());
36      }
37  
38      public MD5BasedHasher(Configuration config) {
39          super(config);
40      }
41  
42      @Override
43      public String getScheme() {
44          return "MD5";
45      }
46  
47      @Override
48      public byte[] encrypt(Credentials credentials, SaltedParts parts) {
49          byte b[] = MD5Crypt.encrypt(credentials.getPassword(), parts.salt, isApacheVariant(parts.hr));
50  
51          // bytes are not encoded in the original order but in that order :
52          byte[] order = {0, 6, 12,
53                          1, 7, 13,
54                          2, 8, 14,
55                          3, 9, 15,
56                          4, 10, 5,
57                          11};
58          // and 3 consecutive 8-bits gives 4 *reverted* 6-bits
59          // that is to say with the following mapping :
60          // 0       6       12       // source index
61          // ........________........ // 8-bits position
62          // abcdefghijklmnopqrstuvwx // 24 bits input (3 bytes)
63          // ++++++------++++++------ // 6 bits position marks (4 * 6-bits)
64          // stuvwxmnopqrghijklabcdef // 24 bits output (3 bytes)
65          // ........________........ // 8-bits position
66          // 0       1       2        // target index
67          byte[] result = new byte[16];
68          // process the 15 first bytes
69          for (int i= 0 ; i< 5; i++) {
70              // process 3 by 3
71              byte pos0 = order[i*3  ]; // starts with 0
72              byte pos1 = order[i*3+1]; // starts with 6
73              byte pos2 = order[i*3+2]; // starts with 12
74              // some masks are useless but ensure we don't miss a bit
75              // stuvwxmn
76              result[i*3  ] = (byte) ( ((b[pos2] << 2) & 0b11111100) | ((b[pos1] >> 2) & 0b00000011) );
77              // opqrghij
78              result[i*3+1] = (byte) ( ((b[pos1] << 6) & 0b11000000) | ((b[pos2] >> 2) & 0b00110000)
79                                     | ((b[pos0] << 2) & 0b00001100) | ((b[pos1] >> 6) & 0b00000011) );
80              // klabcdef
81              result[i*3+2] = (byte) ( ((b[pos1] << 2) & 0b11000000) | ((b[pos0] >> 2) & 0b00111111 ) );
82              // and so on with 1, 7, 13, you got it ?
83          }
84          // revert the 16th byte
85          result[15] = (byte) ( ((b[11] << 2)  & 0b11111100) | ((b[11] >> 6) & 0b00000011) );
86          return result;
87      }
88  
89      @Override
90      public SaltedParts initializeParts() {
91          SaltedParts parts = new SaltedParts(this);
92          parts.salt = genSalt();
93          return parts;
94      }
95  
96      static private final char[] SALTCHARS = BytesEncoder.ValueSpace.base64.get();
97          // "abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ1234567890+/"
98  
99      private byte[] genSalt() {
100         StringBuffer salt = new StringBuffer();
101         SecureRandom randgen = new SecureRandom();
102         while (salt.length() < 8) {
103             int index = (int) (randgen.nextFloat() * SALTCHARS.length);
104             salt.append(SALTCHARS[index]);
105         }
106         return salt.toString().getBytes(StandardCharsets.US_ASCII);
107     }
108 
109     /**
110      * The most popular formatter for this hasher is the Modular Crypt Format.
111      *
112      * <ul>
113      * <li>Apache MD5 crypt format :
114      * <pre>$apr1$jgwedrkq$jzeetEHMGal5H0SUFDMEl1</pre></li>
115      * <li>Crypt MD5 :
116      * <pre>$1$3iuE5z/b$JHyXMzQOIq3cl6WlEMoZC.</pre></li>
117      * </ul>
118      *
119      * @see ModularCryptFormatHashers
120      */
121     public static final CryptFormatter<SaltedParts> MD5CRYPT_FORMATTER = new CryptFormatter<SaltedParts>() {
122         @Override
123         public SaltedParts parse(String crypt, Hasher hr) {
124             String[] stringParts = crypt.split("\\$");
125             SaltedParts parts = new SaltedParts(hr);
126             if (! stringParts[1].equals(parts.hr.getConfiguration().getVariant())) {
127                 parts.hr = parts.hr.getBuilder().setVariant(stringParts[1]).build();
128             }
129             if (stringParts.length > 2 && ! StringUtil.isVoid(stringParts[2])) {
130                 parts.salt = stringParts[2].getBytes(StandardCharsets.US_ASCII);
131             }
132             if (stringParts.length > 3 && ! StringUtil.isVoid(stringParts[3])) {
133                 parts.hash = hr.getConfiguration().getEncoding().decode(stringParts[3]);
134             }
135             return parts;
136         }
137 
138         @Override
139         public String format(SaltedParts parts) {
140             boolean isApacheVariant = isApacheVariant(parts.hr);
141             StringBuffer buf = new StringBuffer(isApacheVariant ? 37 : 34);
142             buf.append('$');
143             buf.append(parts.hr.getConfiguration().getVariant());
144             buf.append("$");
145             buf.append(new String(parts.salt, StandardCharsets.US_ASCII));
146             buf.append('$');
147             buf.append(parts.hr.getConfiguration().getEncoding().encode(parts.hash));
148             return buf.toString();
149         }
150 
151 		@Override
152 		public CryptFormat getCryptFormat() {
153 			return new ModularCryptFormat();
154 		}
155 
156     };
