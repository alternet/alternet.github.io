147     @Override
148     public byte[] encrypt(Credentials credentials, WorkFactorSaltedParts parts) {
149         // SHA-256 or SHA-512
150         Algorithm algo = Algorithm.valueOf(parts.hr.getConfiguration().getAlgorithm());
151         byte bytes[] = SHA2Crypt.encrypt(
152             credentials.getPassword(),
153             parts.workFactor == -1 ? SHA2Crypt.ROUNDS_DEFAULT : parts.workFactor,
154             parts.salt,
155             algo.name(),
156             algo.blockSize()
157         );
158         // bytes are not encoded in the original order but in that order
159         return map(bytes, algo);
160     }

162     /**
163      * Map the bytes to the base 64 expected order.
164      *
165      * @param b The bytes to map.
166      * @param algo The algorithm.
167      *
168      * @return The remapped bytes.
169      */
170     protected byte[] map(byte[] b, Algorithm algo) {
171         // e) the base-64 encoded final C digest. The encoding used is as
172         // follows:
173         // [...]
174         //
175         // Each group of three bytes from the digest produces four
176         // characters as output:
177         //
178         // 1. character: the six low bits of the first byte
179         // 2. character: the two high bits of the first byte and the
180         // four low bytes from the second byte
181         // 3. character: the four high bytes from the second byte and
182         // the two low bits from the third byte
183         // 4. character: the six high bits from the third byte
184         //
185         // The groups of three bytes are as follows (in this sequence).
186         // These are the indices into the byte array containing the
187         // digest, starting with index 0. For the last group there are
188         // not enough bytes left in the digest and the value zero is used
189         // in its place. This group also produces only three or two
190         // characters as output for SHA-512 and SHA-512 respectively.
191 
192 
193         // 3 consecutive 8-bits gives 4 *reverted* 6-bits
194         // that is to say with the following mapping :
195         // 0       10      20       // source index (see SHA_256_ORDER above)
196         // ........________........ // 8-bits position
197         // abcdefghijklmnopqrstuvwx // 24 bits input (3 bytes)
198         // ++++++------++++++------ // 6 bits position marks (4 * 6-bits)
199         // stuvwxmnopqrghijklabcdef // 24 bits output (3 bytes)
200         // ........________........ // 8-bits position
201         // 0       1       2        // target index
202 
203         byte[] order = algo.order();
204         byte[] result = new byte[algo.blockSize()];
205 
206         for (int i= 0 ; i< order.length / 3; i++) {
207             // process 3 by 3
208             byte pos0 = order[i*3  ]; // starts with 0
209             byte pos1 = order[i*3+1]; // starts with 10
210             byte pos2 = order[i*3+2]; // starts with 20
211             byte v0 = pos0 == -1 ? 0 : b[pos0];
212             byte v1 = pos1 == -1 ? 0 : b[pos1];
213             byte v2 = b[pos2];
214             // some masks are useless but ensure we don't miss a bit
215             // stuvwxmn
216             result[i*3] = (byte) ( ((v2 << 2) & 0b11111100) | ((v1 >> 2) & 0b00000011) );
217             if (i*3+1 < result.length) {
218                 if (pos0 == -1) {
219                     // what is annoying is the processing of the end,
220                     // since 43 sextets (258 bits) doesn't fit in 32 bytes (256 bits)
221 
222                     // just keep the significative bits
223 
224                     // opqrijkl (abcdefgh are missing)
225                     result[i*3+1] = (byte) (  ((v1 << 6) & 0b11000000) | ((v2 >> 2) & 0b00110000)
226                             | ((v1 >> 4) & 0b00001111) );
227                 } else {
228                     // opqrghij
229                     result[i*3+1] = (byte) (  ((v1 << 6) & 0b11000000) | ((v2 >> 2) & 0b00110000)
230                                             | ((v0 << 2) & 0b00001100) | ((v1 >> 6) & 0b00000011) );
231                 }
232             }
233             if (i*3+2 < result.length) {
234                 // klabcdef
235                 result[i*3+2  ] = (byte) ( ((v1 << 2) & 0b11000000) | ((v0 >> 2) & 0b00111111 ) );
236             }
237             if (pos1 == -1) {
238                 // stuvwxqr (abcdefghijklmnop are missing)
239                 result[i*3] = (byte) ( ((v2 << 2) & 0b11111100) | ((v2 >> 6) & 0b00000011) );
240             }
241             // and so on with 21, 1, 11, you got it ? (see SHA_256_ORDER above)
242         }
243         return result;
244     }

254     static private final char[] SALTCHARS = BytesEncoder.ValueSpace.h64.get();
255         // "./0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz"
256 
257     private byte[] genSalt() {
258         StringBuffer salt = new StringBuffer();
259         SecureRandom randgen = new SecureRandom();
260         while (salt.length() < 8) {
261             int index = (int) (randgen.nextFloat() * SALTCHARS.length);
262             salt.append(SALTCHARS[index]);
263         }
264         return salt.toString().getBytes(StandardCharsets.US_ASCII);
265     }

267     /**
268      * SHA2 crypt formatter :
269      * <ul>
270      * <li><tt>$5$rounds=12345$q3hvJE5mn5jKRsW.$BbbYTFiaImz9rTy03GGi.Jf9YY5bmxN0LU3p3uI1iUB</tt></li>
271      * <li><tt>$6$49gH89TK$kt//rwoKf1ad/.hnthg363594OMwnM8Z4XScLZug4HdA36pw62AST6/kbirnypS5uzha83Ew2AmITy2HrCW3O0</tt></li>
272      * </ul>
273      */
274     public static final CryptFormatter<WorkFactorSaltedParts> SHA2CRYPT_FORMATTER = new CryptFormatter<WorkFactorSaltedParts>() {
275         @Override
276         public WorkFactorSaltedParts parse(String crypt, Hasher hr) {
277             String[] stringParts = crypt.split("\\$");
278             if ("5".equals(stringParts[1]) || "6".equals(stringParts[1])) {
279                 WorkFactorSaltedParts parts = new WorkFactorSaltedParts(hr);
280                 int pos = 2;
281                 if (stringParts[2].startsWith(SHA2Crypt.ROUNDS_PREFIX)) {
282                     pos++;
283                     parts.workFactor = Integer.parseInt(stringParts[2].substring(SHA2Crypt.ROUNDS_PREFIX.length()));
284                     parts.workFactor = Math.max(SHA2Crypt.ROUNDS_MIN, Math.min(SHA2Crypt.ROUNDS_MAX, parts.workFactor));
285                 } else {
286                     parts.workFactor = -1;
287                 }
288                 BytesEncoding encoding = hr.getConfiguration().getEncoding();
289                 parts.salt = stringParts[pos++].getBytes(StandardCharsets.US_ASCII);
290                 if (stringParts.length > pos) {
291                     parts.hash = encoding.decode(stringParts[pos++]);
292                 }
293                 return parts;
294             } else {
295                 throw new IllegalArgumentException("Unable to parse " + crypt);
296             }
297         }

299         @Override
300         public String format(WorkFactorSaltedParts parts) {
301             StringBuffer buf = new StringBuffer();
302             buf.append('$');
303             Algorithm algo = Algorithm .valueOf(parts.hr.getConfiguration().getAlgorithm());
304             buf.append(algo.ordinal());
305             buf.append("$");
306             if (parts.workFactor != -1) {
307                 buf.append(SHA2Crypt.ROUNDS_PREFIX);
308                 buf.append(parts.workFactor);
309                 buf.append("$");
310             }
311             buf.append(new String(parts.salt, StandardCharsets.US_ASCII));
312             buf.append('$');
313             BytesEncoding encoding = parts.hr.getConfiguration().getEncoding();
314             String string = encoding.encode(parts.hash);
315             buf.append(string);
316             return buf.toString();
317         }
