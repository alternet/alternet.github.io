19  /**
20   * The colon crypt formatter :
21   *
22   * <p>"<tt>PBKDF2:999:0314E17362D0D966C8F999A66045210DBE7EA897F024E07F:3AE344F7AB5AA17308A49FDAD997105340DD6E348FDF5623</tt>"</p>
23   *
24   * Although it is based on {@link WorkFactorSaltedParts}, the work factor may be omitted.
25   *
26   * @author Philippe Poulard
27   */
28  public class ColonCryptFormatter implements CryptFormatter<WorkFactorSaltedParts> {
29  
30      @Override
31      public WorkFactorSaltedParts parse(String crypt, Hasher hr) {
32          return Thrower.safeCall(() -> {
33              Hasher hasher = hr;
34              Scanner scanner = Scanner.of(crypt);
35              SchemeWithEncoding swe = new SchemeWithEncoding(scanner, c -> true, ':');
36              Optional<BytesEncoding> encoding = swe.encoding.get();
37              if (encoding.isPresent()) {
38                  hasher = hr.getBuilder().setEncoding(encoding.get())
39                      .setVariant("withEncoding")
40                      .build();
41              };
42              ensure(scanner.hasNextChar(':', true));
43              WorkFactorSaltedParts parts = new WorkFactorSaltedParts(hasher);
44              Number workFactor = scanner.nextNumber(NumberConstraint.INT_CONSTRAINT);
45              if (workFactor != null) {
46                  parts.workFactor = workFactor.intValue();
47                  if (scanner.hasNext()) {
48                      ensure(scanner.hasNextChar(':', true));
49                  }
50              }
51              if (scanner.hasNext()) {
52                  parts.salt = decode(hr, scanner, c -> c != ':');
53                  if (scanner.hasNext()) {
54                      ensure(scanner.hasNextChar(':', true));
55                      parts.hash = decode(hr, scanner, c -> true); // decode till the end
56                  }
57              }
58              return parts;
59          });
60      }
61  
62      @Override
63      public String format(WorkFactorSaltedParts parts) {
64          StringBuffer crypt = new StringBuffer(60);
65          BytesEncoding encoding = parts.hr.getConfiguration().getEncoding();
66          crypt.append(parts.hr.getConfiguration().getScheme());
67          if ("withEncoding".equals(parts.hr.getConfiguration().getVariant())) {
68              SchemeWithEncoding.code(encoding).ifPresent(s -> crypt.append('.').append(s));
69          }
70          if (parts.workFactor != -1) {
71              crypt.append(':').append(Integer.toString(parts.workFactor));
72          }
73          if (parts.salt != null && parts.salt.length > 0) {
74              crypt.append(':').append(encoding.encode(parts.salt));
75              if (parts.hash != null && parts.hash.length > 0) {
76                  crypt.append(':').append(encoding.encode(parts.hash));
77              }
78          }
79          return crypt.toString();
80      }
81  
82      @Override
83      public CryptFormat getCryptFormat() {
84          return new ColonCryptFormat();
85      }

87  }
