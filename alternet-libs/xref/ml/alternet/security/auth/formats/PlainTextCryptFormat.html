17  /**
18   * A fallback format for passwords stored in plain text ;
19   * DO NOT USE IN PRODUCTION.
20   *
21   * Can be use as the last crypt format to resolve.
22   *
23   * @author Philippe Poulard
24   */
25  @Singleton
26  public class PlainTextCryptFormat implements CryptFormat {
27  
28      public static Hasher.Builder get() {
29          return Hasher.builder()
30              .setClass(PlainTextHasher.class)
31              .setFormatter(PLAIN_FORMATTER);
32      }
33  
34      @Override
35      public Optional<Hasher.Builder> resolve(String crypt) {
36          return Optional.of(get());
37      }
38  
39      /**
40       * Just encode the chars with the charset supplied without hashing ;
41       * DO NOT USE IN PRODUCTION.
42       *
43       * @author Philippe Poulard
44       */
45      public static class PlainTextHasher extends HasherBase<CryptParts> {
46          public PlainTextHasher(Configuration conf) {
47              super(conf);
48          }
49          @Override
50          public String getScheme() {
51              return "[CLEAR PASSWORD]";
52          }
53          @Override
54          public byte[] encrypt(Credentials credentials, CryptParts parts) {
55              try (Password.Clear pwd = credentials.getPassword().getClearCopy()) {
56                  BytesEncoding encoding  = parts.hr.getConfiguration().getEncoding();
57                  if (encoding == null) {
58                      return SafeBuffer.getData(
59                          SafeBuffer.encode(
60                              CharBuffer.wrap(pwd.get()),
61                              parts.hr.getConfiguration().getCharset()
62                          )
63                      );
64                  } else {
65                      // prefer a byte encoder if one is supplied
66                      return encoding.decode(new String(pwd.get()));
67                  }
68              }
69          }
70          @Override
71          public CryptParts initializeParts() {
72              return new CryptParts(this);
73          }
74      }
75  
76      /**
77       * Convert string to bytes and vice-versa with the supplied
78       * bytes encoder or by default with the supplied charset.
79       */
80      private static final CryptFormatter<? extends CryptParts> PLAIN_FORMATTER = new CryptFormatter<CryptParts>() {
81          @Override
82          public CryptParts parse(String crypt, Hasher hr) {
83              CryptParts parts = new CryptParts(hr);
84              BytesEncoding encoding  = parts.hr.getConfiguration().getEncoding();
85              if (encoding == null) {
86                  parts.hash = SafeBuffer.getData(
87                      hr.getConfiguration().getCharset().encode(crypt)
88                  );
89              } else {
90                  // prefer a byte encoder if one is supplied
91                  parts.hash = encoding.decode(crypt);
92              }
93              return parts;
94          }
95          @Override
96          public String format(CryptParts parts) {
97              BytesEncoding encoding  = parts.hr.getConfiguration().getEncoding();
98              if (encoding == null) {
99                  return parts.hr.getConfiguration()
100                     .getCharset()
101                     .decode(ByteBuffer.wrap(parts.hash))
102                     .toString();
103             } else {
104                 // prefer a byte encoder if one is supplied
105                 return encoding.encode(parts.hash);
106             }
107         }
108 		@Override
109 		public CryptFormat getCryptFormat() {
110 			return new PlainTextCryptFormat();
111 		}
112     };
113 
114     /**
115      * @return ""

117     @Override
118     public String family() {
119         return "";
120     }
121 
122     /**
123      * @return "[CLEAR PASSWORD]"
124      */
125     @Override
126     public String infoTemplate() {
127         return "[CLEAR PASSWORD]";
128     }
