10  /**
11   * Maps a token value to a custom user node.
12   *
13   * A token marked as {@link Fragment} doesn't
14   * need a mapper.
15   *
16   * @author Philippe Poulard
17   *
18   * @param <Node> The type of the target node.
19   */
20  public interface TokenMapper<Node> {
21  
22      /**
23       * Transform a token to a node in the context
24       * of what was previously parsed in the stack,
25       * and what follows in the current rule.
26       *
27       * For example, if a rule defines a comma-separated
28       * list of digits, that the input is "1,2,3,4", and
29       * that the current token is "2", then the next elements
30       * are ",3,4" and the stack is "1,".
31       *
32       * Some elements may be consumed during the production
33       * of the target node.
34       *
35       * @param stack The stack contains source tokens
36       *      encountered so far but not yet processed.
37       *      The stack is a work area that may be useful
38       *      for the transformation of the token.
39       * @param token The actual token value to transform is
40       *      supplied by the parser.
41       * @param next The next elements in the boundaries
42       *      of the current rule.
43       * @return A custom user node, or <code>null</code>
44       *      to keep the token unchanged (in that case it
45       *      is supposed to be consumed elsewhere.
46       */
47      Node transform(
48              ValueStack<Value<Node>> stack,
49              TokenValue<?> token,
50              Deque<Value<Node>> next);
51  
52  }
