10  /**
11   * Maps a rule to a custom user node.
12   *
13   * A rule marked as {@link Fragment} doesn't
14   * need a mapper.
15   *
16   * @author Philippe Poulard
17   *
18   * @param <Node> The type of the target node.
19   */
20  public interface RuleMapper<Node> {
21  
22      /**
23       * Transform a rule to a node in the context
24       * of what was previously parsed in the stack,
25       * and what follows in the current rule.
26       *
27       * @param stack The stack contains source tokens
28       *      encountered so far but not yet processed.
29       *      The stack is a work area that may be useful
30       *      for the transformation of the token.
31       * @param rule The actual rule to transform is
32       *      supplied by the parser.
33       * @param args All the elements in the boundaries
34       *      of the current rule, that have been previously
35       *      transformed.
36       * @return A custom user node, or <code>null</code>
37       *      to discard the rule, which means that all
38       *      its arguments will be merged with the arguments
39       *      of the nested rule.
40       */
41      Node transform(
42              ValueStack<Value<Node>> stack,
43              Rule rule,
44              Deque<Value<Node>> args);
45  
46  }
