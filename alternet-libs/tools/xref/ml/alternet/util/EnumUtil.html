11  /**
12   * Enum-related utilities.
13   *
14   * @author Philippe Poulard
15   */
16  @Util
17  public class EnumUtil {
18  
19      /**
20       * Replace the value of an enum instance by another value.
21       *
22       * Helpful for enum values that are made of characters that
23       * are not Java names.
24       *
25       * <h3>Example</h3>
26       * An enum type made of values that contains a space.
27       *
28       * <pre>public enum MyEnum {
29       *    VALUE_1,
30       *    VALUE_2,
31       *    VALUE_3;
32       *
33       *    MyEnum() {
34       *        // replace the "_" in the name by a " "
35       *       EnumUtil.replace(MyEnum.class, this, s -&gt; s.replace('_', ' '));
36       *    }
37       *}</pre>
38       * Usage :
39       * <pre> MyEnum val = MyEnum.valueOf("VALUE 2");</pre>
40       *
41       * @param enumClass The enum class to patch
42       * @param enumValue The instance to change
43       * @param transformer Apply a transformation on the name of the instance
44       */
45      public static void replace(Class<? extends Enum<?>> enumClass, Object enumValue,
46              UnaryOperator<String> transformer)
47      {
48          try {
49              Field fieldName = enumClass.getSuperclass().getDeclaredField("name");
50              AccessController.doPrivileged((PrivilegedAction<Void>) (() -> {
51                  fieldName.setAccessible(true);
52                  return null;
53              }));
54              String value = (String) fieldName.get(enumValue);
55              value = transformer.apply(value);
56              fieldName.set(enumValue, value);
57              AccessController.doPrivileged((PrivilegedAction<Void>) (() -> {
58                  fieldName.setAccessible(false);
59                  return null;
60              }));
61          } catch (Exception e) {
62              Logger.getLogger(enumClass.getName())
63              .severe("Unable to access to \"name\" field of class " + enumClass.getName());
64              Thrower.doThrow(e);
65          }
66      }
67  
68      /**
69       * Replace the ordinal of an enum instance by another ordinal.
70       *
71       * @param enumClass The enum class to patch
72       * @param enumValue The instance to change
73       * @param transformer Apply a transformation on the ordinal of the instance
74       */
75      public static void reorder(Class<? extends Enum<?>> enumClass, Object enumValue,
76              UnaryOperator<Integer> transformer)
77      {
78          try {
79              Field fieldName = enumClass.getSuperclass().getDeclaredField("ordinal");
80              AccessController.doPrivileged((PrivilegedAction<Void>) (() -> {
81                  fieldName.setAccessible(true);
82                  return null;
83              }));
84              int value = (int) fieldName.get(enumValue);
85              value = transformer.apply(value);
86              fieldName.set(enumValue, value);
87              AccessController.doPrivileged((PrivilegedAction<Void>) (() -> {
88                  fieldName.setAccessible(false);
89                  return null;
90              }));
91          } catch (Exception e) {
92              Logger.getLogger(enumClass.getName())
93              .severe("Unable to access to \"ordinal\" field of class " + enumClass.getName());
94              Thrower.doThrow(e);
95          }
96      }
