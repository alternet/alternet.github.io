14      @Test
15      public void customProperties_shouldBe_unmarshalled() throws IOException, URISyntaxException {
16          Conf conf = Conf.unmarshall(org.example.conf.generated.test.Conf.class.getResourceAsStream("conf.properties"));
17  
18          assertThat(conf.gui.window.width).isEqualTo(500);
19          assertThat(conf.gui.window.height).isEqualTo((short) 300);
20  
21          assertThat(conf.db.account.login).isEqualTo("theLogin");
22          assertThat(conf.db.account.password).isEqualTo("thePassword".toCharArray());
23  
24      }

26      @Test
27      public void dynamicProperties_shouldBe_unmarshalled() throws IOException, URISyntaxException {
28          Conf conf = Conf.unmarshall(org.example.conf.generated.test.Conf.class.getResourceAsStream("conf-dynamic.properties"));
29  
30          assertThat(conf.gui.window.width).isEqualTo(500);
31          assertThat(conf.gui.window.height).isEqualTo((short) 300);
32  
33          // loaded from a separate file
34          assertThat(conf.db.account.login).isEqualTo("theLogin");
35          assertThat(conf.db.account.password).isEqualTo("thePassword".toCharArray());
36  
37          assertThat(conf.user).isNull();

40      @Test
41      public void defaultProperties_shouldBe_unmarshalled() throws IOException, URISyntaxException {
42          Conf conf = Conf.unmarshall(System.getProperties());
43          conf.update(org.example.conf.generated.test.Conf.class.getResourceAsStream("conf-dynamic.properties"));
44  
45          assertThat(conf.gui.window.width).isEqualTo(500);
46          assertThat(conf.gui.window.height).isEqualTo((short) 300);
47  
48          // loaded from a separate file
49          assertThat(conf.db.account.login).isEqualTo("theLogin");
50          assertThat(conf.db.account.password).isEqualTo("thePassword".toCharArray());
51  
52          assertThat(conf.user).isNotNull();
53          assertThat(conf.user.name).isEqualTo(System.getProperty("user.name"));
54          assertThat(conf.user.dir).isEqualTo(new File(System.getProperty("user.dir")));
55  
56      }
