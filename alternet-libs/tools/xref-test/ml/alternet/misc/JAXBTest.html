19  @Test
20  public class JAXBTest {
21  
22      public static class Item {
23  
24          public Item() { }
25  
26          public Item(int val) {
27              this.value = val;
28          }
29  
30          @XmlAttribute
31          public int value;
32  
33          @Override
34          public boolean equals(Object obj) {
35              return obj instanceof Item && ((Item) obj).value == this.value;
36          }
37  
38          @Override
39          public int hashCode() {
40              return this.value;
41          }
42  
43          @Override
44          public String toString() {
45              return "<item value=" + value + "/>";
46          }
47      }
48  
49      static QName ITEM = new QName("http://example.com/items", "item");
50      static Supplier<InputStream> LIST = () -> JAXBTest.class.getResourceAsStream("items.xml");
51  
52      @SuppressWarnings({ "rawtypes", "unchecked" })
53      public void xmlNotCached_Should_UnmarshallToItems() {
54          JAXBStream<Item> xml = new JAXBStream<Item>(LIST, ITEM, Item.class,
55              JAXBStream.CacheStrategy.noCache);
56          List<Item> items = xml.stream().collect(Collectors.toList());
57          assertThat(items).containsExactly(new Item(42), new Item(0), new Item(-24));
58          assertThat(xml.source).isInstanceOf(JAXBStream.StreamSource.class);
59          assertThat(((JAXBStream.StreamSource) xml.source).getLocation()).isEmpty(); // because JAXBTest.class.getResourceAsStream()
60          assertThat(xml.cacheStrategy).isSameAs(JAXBStream.CacheStrategy.noCache);
61      }
62  
63      public void xmlInMemoryCache_Should_UnmarshallToItems() {
64          JAXBStream<Item> xml = new JAXBStream<Item>(LIST, ITEM, Item.class,
65              JAXBStream.CacheStrategy.memoryCache);
66          assertThat(xml.source).isInstanceOf(JAXBStream.StreamSource.class);
67  
68          List<Item> items = xml.stream().collect(Collectors.toList());
69          assertThat(items).containsExactly(new Item(42), new Item(0), new Item(-24));
70          assertThat(xml.source).isInstanceOf(JAXBStream.ListSource.class);
71          assertThat(xml.cacheStrategy).isSameAs(JAXBStream.CacheStrategy.noCache);
72  
73          // check reading through the cache
74          List<Item> itemsInMemory = xml.stream().collect(Collectors.toList());
75          assertThat(itemsInMemory).containsExactly(new Item(42), new Item(0), new Item(-24));
76  
77          xml.clearCache();
78  
79          // ensure it was rebuilt properly
80          assertThat(xml.source).isInstanceOf(JAXBStream.StreamSource.class);
81          assertThat(xml.cacheStrategy).isSameAs(JAXBStream.CacheStrategy.memoryCache);
82  
83          // ensure it works fine again
84          List<Item> itemsCleared = xml.stream().collect(Collectors.toList());
85          assertThat(itemsCleared).containsExactly(new Item(42), new Item(0), new Item(-24));
86          assertThat(xml.source).isInstanceOf(JAXBStream.ListSource.class);
87          assertThat(xml.cacheStrategy).isSameAs(JAXBStream.CacheStrategy.noCache);
88  
89          List<Item> itemsClearedInMemory = xml.stream().collect(Collectors.toList());
90          assertThat(itemsClearedInMemory).containsExactly(new Item(42), new Item(0), new Item(-24));
91      }
92  
93      @SuppressWarnings({ "rawtypes", "unchecked" })
94      public void xmlInTmpFileCache_Should_UnmarshallToItems() throws URISyntaxException {
95          JAXBStream<Item> xml = new JAXBStream<Item>(LIST, ITEM, Item.class,
96              JAXBStream.CacheStrategy.localTmpFileCache);
97          assertThat(xml.source).isInstanceOf(JAXBStream.StreamSource.class);
98  
99          List<Item> items = xml.stream().collect(Collectors.toList());
100         assertThat(items).containsExactly(new Item(42), new Item(0), new Item(-24));
101         assertThat(xml.source).isInstanceOf(JAXBStream.StreamSource.class);
102         assertThat(xml.cacheStrategy).isSameAs(JAXBStream.CacheStrategy.noCache);
103         Optional<URL> tmp = ((JAXBStream.StreamSource) xml.source).getLocation();
104         assertThat(tmp).isPresent();
105         File file = new File(tmp.get().toURI());
106         // ensure TMP file exists
107         assertThat(file).exists();
108         assertThat(file.getName()).contains("JAXBStream_");
109 
110         // check reading through the cache
111         List<Item> itemsInFile = xml.stream().collect(Collectors.toList());
112         assertThat(itemsInFile).containsExactly(new Item(42), new Item(0), new Item(-24));
113 
114         xml.clearCache();
115         // ensure TMP file was deleted
116         assertThat(file).doesNotExist();
117 
118         // ensure it was rebuilt properly
119         assertThat(xml.source).isInstanceOf(JAXBStream.StreamSource.class);
120         assertThat(xml.cacheStrategy).isSameAs(JAXBStream.CacheStrategy.localTmpFileCache);
121         assertThat(((JAXBStream.StreamSource) xml.source).getLocation()).isEmpty(); // because JAXBTest.class.getResourceAsStream()
122 
123         // ensure it works fine again
124         List<Item> itemsCleared = xml.stream().collect(Collectors.toList());
125         assertThat(itemsCleared).containsExactly(new Item(42), new Item(0), new Item(-24));
126         assertThat(xml.source).isInstanceOf(JAXBStream.StreamSource.class);
127         assertThat(xml.cacheStrategy).isSameAs(JAXBStream.CacheStrategy.noCache);
128         Optional<URL> tmpClear = ((JAXBStream.StreamSource) xml.source).getLocation();
129         assertThat(tmpClear).isPresent();
130         assertThat(tmpClear.get().getPath()).contains("JAXBStream_");
131 
132         List<Item> itemsClearedInMemory = xml.stream().collect(Collectors.toList());
133         assertThat(itemsClearedInMemory).containsExactly(new Item(42), new Item(0), new Item(-24));
134     }
