1   package ml.alternet.misc;
2   
3   import static org.assertj.core.api.Assertions.assertThat;
4   
5   import org.testng.annotations.Test;
6   
7   import ml.alternet.misc.CharRange.Char;
8   import ml.alternet.misc.CharRange.Chars;
9   import ml.alternet.misc.CharRange.Range;
10  
11  @Test
12  public class CharRangeContainsTest {
13  
14      public void char_Should_containOnlyItsChar() {
15          Char c = new Char(true, 'a');
16          assertThat(c.contains('a')).isTrue();
17          assertThat(c.contains('A')).isFalse();
18      }
19  
20      public void rangeOfSingleChar_Should_containOnlyItsChar() {
21          Range r = new Range('a', 'a');
22          assertThat(r.contains('a')).isTrue();
23          assertThat(r.contains('A')).isFalse();
24      }
25  
26      public void range_Should_containOnlyItsChars() {
27          Range r = new Range('a', 'z');
28          assertThat(r.contains('a')).isTrue();
29          assertThat(r.contains('b')).isTrue();
30          assertThat(r.contains('z')).isTrue();
31          assertThat(r.contains('A')).isFalse();
32      }
33  
34      public void charsOfSingleChar_Should_containOnlyItsChar() {
35          Chars c = new Chars(true, "a");
36          assertThat(c.contains('a')).isTrue();
37          assertThat(c.contains('A')).isFalse();
38      }
39  
40      public void chars_Should_containOnlyTheirChars() {
41          Chars c = new Chars(true, "abcz");
42          assertThat(c.contains('a')).isTrue();
43          assertThat(c.contains('b')).isTrue();
44          assertThat(c.contains('z')).isTrue();
45          assertThat(c.contains('A')).isFalse();
46      }
47  
48      public void notChar_Should_containOnlyOtherChars() {
49          Char c = new Char(false, 'a');
50          assertThat(c.contains('A')).isTrue();
51          assertThat(c.contains('a')).isFalse();
52      }
53  
54      public void notCharsOfSingleChar_Should_containOtherChars() {
55          Chars c = new Chars(false, "a");
56          assertThat(c.contains('A')).isTrue();
57          assertThat(c.contains('a')).isFalse();
58      }
59  
60      public void notChars_Should_containOnlyOtherChars() {
61          Chars c = new Chars(false, "abcz");
62          assertThat(c.contains('a')).isFalse();
63          assertThat(c.contains('b')).isFalse();
64          assertThat(c.contains('z')).isFalse();
65          assertThat(c.contains('A')).isTrue();
66      }
67  
68  }
