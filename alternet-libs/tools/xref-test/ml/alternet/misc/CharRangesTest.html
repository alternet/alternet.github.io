4   
5   import java.util.Arrays;
6   import java.util.List;
7   import java.util.stream.Collectors;
8   
9   import org.testng.annotations.Test;
10  
11  import ml.alternet.misc.CharRange;
12  import ml.alternet.misc.CharRange.BoundRange;
13  import ml.alternet.misc.CharRange.Char;
14  import ml.alternet.misc.CharRange.Chars;
15  import ml.alternet.misc.CharRange.Range;
16  import ml.alternet.misc.CharRange.Ranges;
17  
18  @Test
19  public class CharRangesTest {
20  
21      public void unionOfOrderedChar_Should_giveOrderedRanges() {
22          CharRange c = new Range('A', 'Z').union(new Char(true, 'a'));
23  
24          assertThat(c.getClass()).isAssignableFrom(Ranges.class);
25  
26          List<BoundRange> charRanges = c.asIntervals().collect(Collectors.toList());
27          assertThat(charRanges).containsExactly(
28              new Range('A', 'Z'),
29              new Char(true, 'a')
30          );
31      }
32  
33      public void unionOfUnorderedChar_Should_giveOrderedRanges() {
34          // check order too
35          CharRange c = new Range('A', 'Z').union(new Chars(true, "9287315")).union(new Char(true, 'a'));
36  
37          assertThat(c.getClass()).isAssignableFrom(Ranges.class);
38  
39          List<BoundRange> charRanges = c.asIntervals().collect(Collectors.toList());
40          assertThat(charRanges).containsExactly(
41              new Range('1', '3'),
42              new Char(true, '5'),
43              new Range('7', '9'),
44              new Range('A', 'Z'),
45              new Char(true, 'a'));
46      }
47  
48      public void alphaChar_ShouldBe_orderedAfterNumericRange() {
49          BoundRange a = new Char(true, 'a');
50          BoundRange oneToThree = new Range('1', '3');
51          assertThat(a).isGreaterThan(oneToThree);
52      }
53  
54      public void alphaRange_ShouldBe_orderedAfterNumericChar() {
55          BoundRange one = new Char(true, '1');
56          BoundRange aTof = new Range('a', 'f');
57          assertThat(aTof).isGreaterThan(one);
58      }
59  
60      public void variousRanges_ShouldBe_sortedByStartChar() {
61          List<BoundRange> list = Arrays.asList(
62              new Char(true, 'a'),
63              new Char(true, '5'),
64              new Range('1', '3'),
65              new Range('A', 'Z'),
66              new Range('7', '9')
67          );
68          list = list.stream().sorted().collect(Collectors.toList());
69          assertThat(list).containsExactly(
70              new Range('1', '3'),
71              new Char(true, '5'),
72              new Range('7', '9'),
73              new Range('A', 'Z'),
74              new Char(true, 'a')
75          );
76      }
