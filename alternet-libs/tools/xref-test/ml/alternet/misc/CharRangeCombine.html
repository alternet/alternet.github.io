18  @Test
19  public class CharRangeCombine {
20  
21      public void anyCharsExceptAChar_Shoud_haveTheSameIntervalsThanNotAChar() {
22          Char a = CharRange.is('a');
23          CharRange anyExceptA = ANY.except(a);
24          Char notA = CharRange.isNot('a');
25          assertThat(anyExceptA.asIntervals().findFirst().get()).isEqualTo(notA.asIntervals().findFirst().get());
26          assertThat(notA).usingComparator(CHAR_RANGE_COMPARATOR).isEqualTo(anyExceptA);
27      }
28  
29      public void doubleNegateForChar_Shoud_giveTheSame() {
30          Char a = CharRange.is('a');
31          CharRange anyExceptA = ANY.except(a);
32          CharRange aAgain = ANY.except(anyExceptA);
33          assertThat(a).isEqualTo(aAgain);
34      }
35  
36      public void doubleNegateForNotChar_Shoud_giveTheSame() {
37          Char a = CharRange.isNot('a');
38          CharRange anyExceptA = ANY.except(a);
39          CharRange aAgain = ANY.except(anyExceptA);
40          assertThat(CHAR_RANGE_COMPARATOR.compare(a, aAgain)).isEqualTo(0);
41          assertThat(a).usingComparator(CHAR_RANGE_COMPARATOR).isEqualTo(aAgain);
42      }
43  
44      public void rangesWithExclusions_Shoud_beCut() {
45          CharRange range = new Range('A', 'Z')
46                  .union(new Chars(true, "9287315"))
47                  .union(new Char(true, 'a'))
48                  .union(new Char(true, 'h'))
49                  .union(new Range('m', 'x'))
50                  .except(new Char(true, 'M'))
51                  .except(new Range('w', 'z'))
52                  .except(new Range('f', 'p'))
53                  .except(new Char(true, '0'));
54  
55          assertThat(range.getClass()).isAssignableFrom(Ranges.class);
56  
57          List<BoundRange> charRanges = range.asIntervals().collect(Collectors.toList());
58          assertThat(charRanges).containsExactly(
59              new Range('1', '3'),
60              new Char(true, '5'),
61              new Range('7', '9'),
62              new Range('A', 'L'),
63              new Range('N', 'Z'),
64              new Char(true, 'a'),
65              new Range('q', 'v')
66          );
67      }
68  
69      public void rangesWithExclusionsAtOnce_Shoud_beCut() {
70          CharRange range = new Range('A', 'Z')
71                  .union(new Chars(true, "9287315"), new Char(true, 'a'), new Char(true, 'h'), new Range('m', 'x'))
72                  .except(new Char(true, 'M'), new Range('w', 'z'), new Range('f', 'p'), new Char(true, '0'));
73  
74          assertThat(range.getClass()).isAssignableFrom(Ranges.class);
75  
76          List<BoundRange> charRanges = range.asIntervals().collect(Collectors.toList());
77          assertThat(charRanges).containsExactly(
78              new Range('1', '3'),
79              new Char(true, '5'),
80              new Range('7', '9'),
81              new Range('A', 'L'),
82              new Range('N', 'Z'),
83              new Char(true, 'a'),
84              new Range('q', 'v')
85          );
86      }
