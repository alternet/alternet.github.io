10      BigInteger bigLong = BigInteger.valueOf(Long.MAX_VALUE);

14      @Test
15      public void parseNumber_Should_ParseBigInt() {
16          Number n = NumberUtil.parseNumber(twiceALongAsString, true, BigInteger.class);
17          Assertions.assertThat(n).isInstanceOf(BigInteger.class);
18          Assertions.assertThat(n).isEqualTo(twiceALong);
19      }
20  
21      @Test
22      public void parseNumber_Should_ParseNonFloatingPointBigIntToBigInt() {
23          Number n = NumberUtil.parseNumber(twiceALongAsString, false, BigInteger.class);
24          Assertions.assertThat(n).isInstanceOf(BigInteger.class);
25          Assertions.assertThat(n).isEqualTo(twiceALong);
26      }

28      @Test
29      public void parseNumber_Should_ParseBigIntToFloat() {
30          Number n = NumberUtil.parseNumber(twiceALongAsString);
31          Assertions.assertThat(n).isInstanceOf(Float.class);
32          Assertions.assertThat(n).isEqualTo(Float.parseFloat(twiceALongAsString));
33      }

35      @Test
36      public void parseNumber_Should_ParseMaxFloatToFloat() {

39          Assertions.assertThat(n).isEqualTo(Float.MAX_VALUE);

55      public void parseNumber_Should_ParseIntBiggerThanMaxDoubleToBigInt() {

57          Number n = NumberUtil.parseNumber(biggerThanADouble); // Double.MAX_VALUE
58                                                                // * 10 is a big
59                                                                // int

64      @Test
65      public void toDecimal_Should_ConvertMaxFloat() {

67          Assertions.assertThat(toDecimal("" + Float.MAX_VALUE)).isEqualTo(maxFloatAsString);
68      }

70      String toDecimal(String number) {
71          String[] parts = number.split("E");
72          int exp = Integer.parseInt(parts[1]);
73          parts = parts[0].split("\\.");
74          StringBuffer s = new StringBuffer(exp + 20);
75          s.append(parts[0]);
76          for (int i = 0; i < parts[1].length(); i++) {
77              s.append(parts[1].charAt(i));
78              exp--;
79          }
80          while (exp-- > 0) {
81              s.append('0');
82          }
83          return s.toString();
84      }
