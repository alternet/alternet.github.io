96      /**
97       * Return the singleton instance of the given grammar.
98       *
99       * @param grammar The grammar must be an interface.
100      *
101      * @return An instance of the given interface,
102      *      generated without methods implementation
103      *      that would be specified in the interface.
104      *
105      * @param <T> The target grammar interface.
106      */
107     @SuppressWarnings("unchecked")
108     public static <T extends Grammar> T $(Class<? extends Grammar> grammar) {
109         if (grammar.isInterface()) {
110             try {
111                 Grammar$ g = (Grammar$) BYTECODE_FACTORY.getInstance(grammar);
112                 g.grammar = grammar;
113                 g.log = Logger.getLogger(g.getGrammarName());
114                 g.log.setLevel(Level.ALL);
115                 g.log.info("Creating grammar " + g.getGrammarName());
116                 // before doing anything on the grammar,
117                 // ensure to prepare it before
118                 g.init();
119                 return (T) g;
120             } catch (NoSuchFieldException | SecurityException
121                     | InstantiationException | IllegalAccessException
122                     | ClassNotFoundException | IllegalArgumentException e)
123             {
124                 // Trouble shooting java.lang.IllegalAccessError:
125                 //        class a.b.c.D$E$ cannot access its superinterface a.b.c.D$E
126                 //        because a.b.c.D$E is not public
127                 return Thrower.doThrow(e);
128             }
129         } else {
130             throw new IllegalArgumentException(grammar.getName() + " is not an interface.");
131         }
132     }

134     /**
135      * Create either the singleton instance of the current Grammar
136      * or a new proxy Rule, according to how this method is invoked.
137      *
138      * <p>In the following sample code, the method <code>$()</code>
139      * is called twice, once on a rule field, then on the grammar
140      * singleton field :</p>
141      *
142      * <pre>import static ml.alternet.parser.Grammar.*;
143      *import ml.alternet.parser.Grammar;
144      *
145      *public interface Foo extends Grammar {
146      *
147      *    // create a proxy rule that will be initialized later
148      *    Rule MyRule = $();
149      *
150      *    // create the grammar instance to use for parsing an input
151      *    Foo $ = $(); // must be the last field
152      *
153      *}</pre>
154      *
155      * <p>In both cases, the caller of this method must be the
156      * actual grammar interface, that is to say you MUST NOT
157      * invoke that method outside for initializing a Rule or
158      * the Grammar instance elsewhere than in your grammar
159      * interface.</p>
160      *
161      * <h1>Grammar field</h1>
162      *
163      * <p><code>$()</code> return the singleton instance of the
164      * current grammar.</p>
165      *
166      * <p>NOTE : this method MUST BE called after all fields of
167      * the interface have been properly initialized ; typically
168      * this method is used to set the LAST field of the grammar
169      * by getting its instance.</p>
170      *
171      * <h1>Rule field</h1>
172      *
173      * <p><code>$()</code> return a new proxy rule, useful when a
174      * rule reference is expected while its definition will be
175      * specified later. It is useful when a rule is made of other
176      * rules/tokens that are defined later in the grammar.</p>
177      *
178      * <p>NOTE : all Rule fields in a grammar MUST NOT be null.</p>
179      *
180      * <p>When such proxy rule has been declared as a field in a
181      * grammar, it have to be initialized later in the grammar :</p>
182      * <ul>
183      *  <li>Either by declaring a STATIC method with the same name as
184      *      the field :
185      *      <pre>    static Rule MyRule() {
186      *        return SomeRule.seq(OtherRule);
187      *    }</pre></li>
188      *  <li>Or by declaring a supplier field with the same name as
189      *      the field prepend with "$" :
190      *      <pre>    Supplier&lt;Rule&gt; $MyRule = () -&gt; SomeRule.seq(OtherRule);</pre></li>
191      *  <li>Or by declaring a dummy field that perform the initialization :
192      *      <pre>    boolean b1 = ((Proxy) MyRule).is(
193      *        SomeRule.seq(OtherRule)
194      *    );</pre></li>
195      * </ul>
196      *
197      * @param <T> The target grammar interface OR a Rule.
198      *
199      * @return For the grammar field, return an instance of
200      *      the given interface, generated without methods
201      *      implementation that would be specified in the
202      *      interface ; for a rule field, return a new proxy
203      *      rule.
204      *
205      * @see ByteCodeFactory#getInstance(Class)
206      *
207      * @throws ClassCastException If the grammar interface defines
208      *      a rule field left <code>null</code>.
209      */
210     @SuppressWarnings("unchecked")
211     public static <T> T $() {
212         Class<? extends Grammar> g = GIF.getGrammarInterface();
213         // check preconditions : all Rules must be non-null
214         Field[] fields = g.getDeclaredFields();
215         for (int i = fields.length - 1 ; i >= 0; i--) { // start from the end for fail fast
216             Field f = fields[i];
217             try {
218                 // if it remains a rule not initialized, return a Proxy()
219                 if (Rule.class.isAssignableFrom(f.getType()) && f.get(null) == null) {
220                     return (T) new Grammar.Proxy();
221                 }
222             } catch (IllegalArgumentException | IllegalAccessException e) {
223                 Thrower.doThrow(e);
224             }
225         }
226         // otherwise return the Grammar instance
227         return $(g);
228     }

251     @Override
252     public Grammar init() { // see below INITIALIZERS
253         if (! this.init) {
254             // preconditions will be checked by RuleField on the first field browse
255             this.init = true;
256             // apply initializations in this order
257             // don't merge the loops in those methods !!!
258             // typically : naming is an important step that must be
259             //          performed BEFORE optimization and substitutions
260             try {
261                 // set the name of rules that are fields,
262                 // process fragments, and process proxies
263                 processFields();
264                 // resolve placeholders $self and $("namedRule")
265                 processPlaceHolders();
266                 // flatten an simplify
267                 processOptimizations();
268                 // substitutions and extensions
269                 processSubstitutions();
270                 // @Skip and @Drop
271                 processAnnotations();
272             } catch (IllegalArgumentException | IllegalAccessException e) {
273                 return Thrower.doThrow(e);
274             }
275         }
276         return this;
277     }

279     /**
280      * Return the main rule of the grammar.
281      *
282      * @return The main rule if one has been set
283      *      on the grammar or one of its interfaces.
284      *      ({@code Grammar.MainRule})
285      */
286     @Override
287     public java.util.Optional<Rule> mainRule() {
288         if (this.mainRule == null) {
289             // works on init() grammars only
290             this.mainRule = fields()
291                 .filter(f -> f.getAnnotation(MainRule.class) != null)
292                 .map(f -> (Rule) safeCall(() -> f.get(null)))
293                 .map(this::adopt) // if it comes from an inherited grammar
294                 .findFirst();
295         }
296         return this.mainRule;
297     }

299     /**
300      * Return the "tokenizer rule", that is to
301      * say all the non-fragment tokens explicitly
302      * specified by this grammar ({@code Grammar.Fragment}).
303      *
304      * @return The rule <code>(T1 | T2 | T3 ...)*</code>
305      *
306      * @see Grammar#tokenizer()
307      */
308     @Override
309     public Rule tokenizer() {
310         if (this.tokenizer == null) {
311             // works on init() grammars only
312             Stream<Token> tokens = fields()
313                 .filter(f ->   Token.class.isAssignableFrom(f.getType()) // keep tokens
314                             && f.getAnnotation(Fragment.class) == null)  // that ARE NOT fragments
315                 .map(f -> (Token) safeCall(() -> f.get(null)))
316                 .map(this::adopt) // if it comes from an inherited grammar
317                 .map(t -> (Token) t);
318             this.tokenizer = new Choice(tokens).zeroOrMore();
319         }
320         return this.tokenizer;
321     }

323     /**
324      * Parse an input with a rule of this grammar.
325      *
326      * @param scanner The input.
327      * @param handler The receiver.
328      * @param rule The rule to use for parsing the input.
329      * @param matchAll <code>true</code> to indicates that if
330      *      the input contains more characters at the end of
331      *      the parsing, an error will be reported, <code>false</code>
332      *      otherwise.
333      *
334      * @return <code>true</code> if the rule was matched,
335      *          <code>false</code> otherwise.
336      *
337      * @throws IOException When the input cause an error.
338      */
339     @Override
340     public boolean parse(Scanner scanner, EventsHandler handler, Rule rule, boolean matchAll) throws IOException {
341         Handler h = handler.asHandler();
342         // process substitutions if the rule belongs to another grammar
343         Rule r = adopt(rule);
344         log.fine(() -> "Parsing with rule " + r.toPrettyString() + "\n" + Dump.tree(r));
345 
346         Parser.Match match = r.parse(scanner, h);
347         // TODO : notification that characters are available
348         if (matchAll && scanner.hasNext()) {
349 //            handler.warning(); // TODO
350         }
351         return ! match.fail();
352     };

371         Field field;
372         Class<? extends Grammar> grammar;
373 
374         RuleField(Class<? extends Grammar> grammar, Field field) {
375             this.field = field;
376             this.grammar = grammar;
377         }
378 
379         Rule rule() {
380             try {
381                 Rule rule = (Rule) field.get(null);
382                 if (rule == null) {
383                     // case when Rule r = null;
384                     //        or TheGrammar $ = $(); is not the last field
385                     // -> abort
386                     throw new NullPointerException("A Rule or Token is 'null' in "
387                             + this.grammar + "\n or NOT initialized because the field " + field.getName()
388                             + " is declared AFTER getting the instance of the grammar with $() :"
389                             + "\n all Rule/Token fields MUST be declared before the Grammar field");
390                 } else {
391                     return rule;
392                 }
393             } catch (IllegalArgumentException | IllegalAccessException e) {
394                 return Thrower.doThrow(e);
395             }
396         }

422     private void processFields() throws IllegalArgumentException, IllegalAccessException {
423         getRuleFields().forEach(rf -> {
424             // 1) set the name to rules
425             Rule rule = rf.rule();
426             if (rule.isGrammarField()) { // if it has already a name...
427                 throw new IllegalArgumentException("The field \"" + rf.field.getName()
428                         + "\" is a direct reference of an existing named rule."
429                         + "\nPlease correct your grammar by defining your field as a proxy :"
430                         + "\nRule " + rf.field.getName() + " = is(" + rule.getName() + ");");
431             }
432             // set the name of the rule with the name it has in the interface
433             String name = rf.field.getName();
434             rule.setName(name);
435             log.finest(() -> "Setting rule/token name " + name );
436             if (rf.field.getAnnotation(Fragment.class) == null) {
437                 rule.setFragment(false);
438             } else { // assume true by default
439                 rule.setFragment(true);
440             }
441 
442             // 2) set fields declaration that are proxies
443             if (rule instanceof Proxy) {
444                 // lookup for a method that has the same name as the field
445                 // Proxy foo = proxy();
446                 Proxy proxy = (Proxy) rule;
447                 if (proxy.getComponent() == null) {
448                     try {
449                         // static Rule foo() {
450                         //     return SomeRule.zeroOrMore();
451                         // }
452                         Method method = this.grammar.getDeclaredMethod(name);
453                         try {
454                             // call the method
455                             // Rule rule = foo();
456                             Rule r = (Rule) method.invoke(proxy);
457                             // set it to the proxy
458                             proxy.is(r);
459                             log.finest(() -> "Setting proxy " + name + " to " + method);
460                         } catch (IllegalAccessException | IllegalArgumentException | InvocationTargetException e) {
461                             Thrower.doThrow(e);
462                         }
463                     } catch (NoSuchMethodException | SecurityException e) {
464                         // lookup for a field that has the same name as the field prepend with $
465                         try {
466                             // Supplier<Rule> $foo = () -> SomeRule.zeroOrMore();
467                             Field field = this.grammar.getDeclaredField('$' + name);
468                             try {
469                                 // get the field and get the rule
470                                 @SuppressWarnings("unchecked")
471                                 Rule r = ((Supplier<Rule>) field.get(null)).get();
472                                 // set it to the proxy
473                                 proxy.is(r);
474                                 log.finest(() -> "Setting proxy " + name + " to " + r.toString());
475                             } catch (IllegalArgumentException | IllegalAccessException e1) {
476                                 Thrower.doThrow(e);
477                             }
478                         } catch (NoSuchFieldException | SecurityException e1) {
479                             // ignore : the proxy is initialize by other mean
480                         }
481                     }
482                 }
483             }
484             if (rule instanceof Initializable) {
485                 ((Initializable<?>) rule).init();
486                 log.finest(() -> "Initializing rule/token " + name + " to " + rule.toString());
487             }
488 
489         });
490     }

492     // resolve placeholders $self and $("namedRule")
493     private void processPlaceHolders() {
494         getRuleFields().forEach(rf -> {
495             Rule rule = rf.rule();
496             rule.accept(new PlaceHolderResolver(rule));

499     private class PlaceHolderResolver extends TransformTrackingHost {
500         PlaceHolderResolver(Rule rule) {
501             super(rule);
502         }
503         @Override
504         public Rule apply(Rule rule) {
505             if (rule == Grammar.$self) {
506                 rule = getHostRule();
507                 {
508                     Rule r = rule;
509                     log.finest(() -> "Resolving $self to " + r.getName());
510                 }
511             } else if (rule instanceof Grammar.Proxy.Named) {
512                 Grammar.Proxy.Named proxy = (Grammar.Proxy.Named) rule;
513                 if (proxy.getProxyName() != null) {
514                     try {
515                         Field f = Grammar$.this.grammar.getDeclaredField(proxy.getProxyName());
516                         rule = (Rule) f.get(Grammar$.this.grammar);
517                         log.finest(() -> "Resolving $" + proxy.getProxyName() + " in "
518                                                                 + getHostRule().getName());
519                     } catch (NoSuchFieldException | SecurityException
520                             | IllegalArgumentException | IllegalAccessException e)
521                     {
522                         Thrower.doThrow(e);
523                     }
524                 }
525             }
526             return rule;
527         }

545     // when a grammar extends another grammar and extend or overload some fields
546     private void processSubstitutions() throws IllegalArgumentException, IllegalAccessException {
547         // 1) collect the substitutions
548         getRuleFields().forEach(rf -> {
549             // a substitution is either declared explicitly by @Replace
550             // or because a field has the same name of a field in an inherited grammar
551             Grammar.Replace replace = rf.field.getAnnotation(Grammar.Replace.class);
552             if (replace == null) { // field override ?
553                 // find a rule with the same name in the inherited interfaces
554                 ClassUtil.getClasses(this.grammar)
555                     .filter(i -> i != this.grammar && i != Grammar.class)
556                     .flatMap(c -> Arrays.asList(c.getDeclaredFields()).stream())
557                     .filter(f -> rf.rule().isGrammarField()
558                         && f.getName().equals(rf.rule().getName())
559                         && Rule.class.isAssignableFrom(f.getType()))
560                     .findFirst()
561                     .ifPresent(f -> addSubstitution(f, rf.rule()));
562                     // else it is not a replacement... go on
563             } else if (! replace.disable()) {
564                 // @Replace was set
565                 Stream<Class<?>> interfaces;
566                 if (replace.grammar() == Grammar.class) { // means : not specified
567                     // find a rule with the given name in the inherited interfaces
568                     interfaces = ClassUtil.getClasses(this.grammar)
569                         .filter(i -> i != this.grammar && i != Grammar.class);
570                 } else {
571                     // find a rule with the given name in the given interface
572                     interfaces = Stream.of(replace.grammar());
573                 }
574                 Field source = interfaces
575                     .flatMap(c -> Arrays.asList(c.getDeclaredFields()).stream())
576                     .filter(f -> f.getName().equals(replace.field())
577                         && Rule.class.isAssignableFrom(f.getType()))
578                     .findFirst() // with @Replace, we MUST find one
579                     .orElseThrow(() -> new NoSuchFieldError("Substitution not found "
580                                     + replace + " in " + this.grammar.getName()));
581                 addSubstitution(source, rf.rule());
582             } // else it is not a substitution
583         });
584 
585         // 2) log the registered substitutions
586         log.fine(() -> "Substitutions in " + getGrammarName() + " :" + (this.substitutions.isEmpty()
587             ? " NONE"
588             : this.substitutions.entrySet().stream()
589                 .map(e -> "\n   FROM " + Dump.getHash(e.getValue().from) + ' ' + e.getKey()
590                         + " = " + e.getValue().from.toPrettyString()
591                         + "\n     TO " + Dump.getHash(e.getValue().to) + ' ' + e.getValue().to
592                         + " = " + e.getValue().to.toPrettyString())
593                 .collect(Collectors.joining("")))
594         );
595 
596         // 3) apply the substitutions to all fields
597         if (! this.substitutions.isEmpty()) {
598             Set<Rule> traversed = new HashSet<>();
599             getRuleFields().forEach(rf -> {
600                 Rule rule = rf.rule();
601                 log.finest(() -> "Looking for substitution in " + Dump.getHash(rule) + ' ' + rf.field.getName());
602                 rule.accept(new SusbstitutionResolver(rule, traversed));
603             });
604         }
605     }

607     private class SusbstitutionResolver extends TransformTrackingHost {
608         SusbstitutionResolver(Rule start, Set<Rule> traversed) {
609             super(start);
610             this.traversed = traversed;
611         }
612         @Override
613         public Rule apply(Rule from) {
614             if (from.isGrammarField()) {
615                 Substitution s = substitutions.get(from.getName());
616                 if (s != null) {
617                     log.finest(() -> "Substitution found in " + Dump.getHash(getHostRule()) + ' '
618                         + getHostRule().getName()
619                         + " for " + Dump.getHash(from) + ' ' + from.getName() + " FROM "
620                         + Dump.getHash(s.from) + ' ' + s.from.getName() + " TO "
621                         + Dump.getHash(s.to) + ' ' + s.to.getName()
622                         + "\nTraversed : " + traversed.stream().map(r -> Dump.getHash(r) + ' '
623                         + r.getName()).collect(Collectors.joining(", "))
624                     );
625                     if (getHostRule() != s.to) // prevent substitution on extension
626                        // e.g.  Grammar g2 extends g1 {
627                        //          Rule r = g1.r.asToken(...);
628                     {
629                         return s.to; // from -> to
630                     } else { // unchanged
631                         return from;
632                     }
633                 }
634             }
635             return adopted.computeIfAbsent(from, r ->
636                 r instanceof TraversableRule.StandaloneRule
637                 ? r
638                 : (Rule) safeCall(r::clone)
639             );
640         }
641     }

643     // @Drop & @Skip

757     /**
758      * Adopt a rule of another grammar, that is to say apply the
759      * relevant substitutions if any.
760      *
761      * @param rule The original rule, should be a rule of this
762      *      grammar or an inherited grammar.
763      * @return The same rule if no modifications were expected,
764      *      a new rule otherwise.
765      */
766     @Override
767     public Rule adopt(Rule rule) {
768         if (rule == this.tokenizer) { // || rule == this.mainRule.orElse(null)) {
769             return rule;
770         }
771         if (rule.isGrammarField()) {
772             try {
773                 this.grammar.getDeclaredField(rule.getName());
774                 return rule; // rule already in grammar
775             } catch (NoSuchFieldException | SecurityException e) {
776                 return adoption(rule);
777             }
778         } else {
779             throw new IllegalArgumentException("Unable to adopt a rule that is not a grammar field");
780         }
781     }
782 
783     private Rule adoption(Rule rule) {
784         Rule adoptedRule = this.adopted.computeIfAbsent(rule, rul -> {
785             log.fine(() -> "Adopting " + rul + " in " + getGrammarName() + "\n" + Dump.tree(rul));
786             if (this.substitutions.isEmpty()) {
787                 return rule; // as-is
788             } else {
789                 // start with a clone...
790                 Substitution s = substitutions.get(rule.getName());
791                 Rule copy = s == null
792                   ? (Rule) safeCall(rule::clone)
793                   : s.to;
794                 // ...and perform substitutions
795                 copy.accept(new SusbstitutionResolver(copy, new HashSet<>()));
796                 log.fine(() -> "Adopted rule is " + copy.toPrettyString() + "\n" + Dump.tree(copy));
797                 return copy;
798             }
799         });
800         // add an entry for itself for a next lookup with the result
801         this.adopted.put(adoptedRule, adoptedRule);
802         return adoptedRule;
803     }
