3   import java.io.IOException;
4   import java.util.HashMap;
5   import java.util.Map;
6   import java.util.Optional;

8   import ml.alternet.parser.Grammar;
9   import ml.alternet.parser.handlers.TreeHandler;
10  import ml.alternet.scan.Scanner;
11  
12  /**
13   * Build an homogeneous AST from a grammar, tokens mapper,
14   * and rules mapper.
15   *
16   * @author Philippe Poulard
17   *
18   * @param <Node> The type of the target node.
19   */
20  public class NodeBuilder<Node> extends TreeHandler<Node, Node> implements Builder<Node>, Mappers<Node> {
21  
22      Grammar grammar;
23      Map<String, TokenMapper<Node>> tokenMapper = new HashMap<>();
24      Map<String, RuleMapper<Node>> ruleMapper = new HashMap<>();
25  
26      /**
27       * Create a node builder.
28       * By default, the tokens mapper extract the value of the tokens
29       * and the rules mapper does nothing.
30       *
31       * @param grammar The grammar used during parsing.
32       */
33      public NodeBuilder(Grammar grammar) {
34          this.grammar = grammar;
35      }
36  
37      @Override
38      public Optional<Node> parse(Scanner input, boolean matchAll) throws IOException {
39          if (this.grammar.parse(input, this, matchAll)) {
40              return Optional.of(this.get());
41          } else {
42              return Optional.empty();
43          }
44      }
45  
46      @Override
47      public Map<String, TokenMapper<Node>> getTokenMapper() {
48          return this.tokenMapper;
49      }
50  
51      @Override
52      public Map<String, RuleMapper<Node>> getRuleMapper() {
53          return this.ruleMapper;
54      }
55  
56  }
