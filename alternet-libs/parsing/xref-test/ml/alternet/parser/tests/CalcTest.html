32      public static HandlerAccumulator parse(String str, Class<? extends Grammar> g, Rule r) throws IOException {
33          Scanner scanner = new StringScanner(str);
34          HandlerAccumulator acc = new HandlerAccumulator();
35          Grammar sg = Grammar.$(g);
36          sg.parse(scanner, acc, r, true);
37          return acc;
38      }
39  
40      public static List<String> parseToAcc(String str, Class<? extends Grammar> g, Rule r) throws IOException {
41          HandlerAccumulator acc = parse(str, g, r);
42          StringListAccumulator sla = new StringListAccumulator();
43          acc.emitAll(sla);
44          return sla.result;
45      }

47      @Test
48      public void calcGrammarTokenizer_Should_containAllTokensExceptFragments() throws IOException {
49          System.out.println("calcGrammarTokenizer_Should_containAllTokensExceptFragments");
50          Rule rule = C.tokenizer();
51          Choice choice = (Choice) ((ZeroOrMore) rule).getComponent();
52          List<Rule> tokens = choice.getComponent();
53          assertThat(tokens).containsExactly(FUNCTION, LBRACKET, RBRACKET, RAISED,
54                  ADDITIVE, MULTIPLICATIVE, NUMBER, VARIABLE);
55          System.out.println("calcGrammarTokenizer_Should_containAllTokensExceptFragments");
56      }

58      @Test
59      public void calcGrammar_Should_consumeTokens() throws IOException {
60          System.out.println("calcGrammar_Should_consumeTokens");
61          List<String> res = parseToAcc("sin(x)*(1+var_12)", CalcGrammar.class, C.tokenizer());
62          assertThat(res).containsExactly("Function:sin", "String:(", "String:x", "String:)", "Multiplicative:*",
63                                  "String:(", "Number:1", "Additive:+", "String:var_12", "String:)");
64          System.out.println("calcGrammar_Should_consumeTokens");
