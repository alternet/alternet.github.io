19  /**
20   * Allow to test overload on CharTokens.
21   *
22   * Combined CharTokens are optimized by merging ; this test ensure that
23   * an overload can be taken into account.
24   */
25  public class CharOverloadTest {
26  
27      public interface Base extends Grammar {
28  
29          CharToken sep = Grammar.is('-');
30  
31          Token tok = Grammar.range('0', '9').union(sep);
32  
33          @MainRule
34          Rule rul = tok.oneOrMore();
35  
36          Base $ = Grammar.$();
37  
38      }
39  
40      public interface Ext extends Base {
41  
42          Token sep = Grammar.is('_');
43  
44          Ext $ = Grammar.$();
45  
46      }
47  
48      @Test
49      public void charOptimization_ShouldNot_affectCharOverload() throws IOException {
50          // first pass : Ext has not yet been initialized
51          String input = "2018-01-16";
52          TokensCollector<StringBuilder> handler = TokensCollector.newStringBuilderHandler();
53          Base.$.parse(Scanner.of(input), handler, true);
54          String result = handler.get().toString();
55          assertThat(result).isEqualTo(input);
56  
57          // 2nd pass : check Ext with its changes
58          input = "2018_01_16";
59          handler = TokensCollector.newStringBuilderHandler();
60          Ext.$.parse(Scanner.of(input), handler, true);
61          result = handler.get().toString();
62          assertThat(result).isEqualTo(input);
63  
64          // 3rd pass : ensure that Ext didn't affect Base
65          input = "2018-01-16";
66          handler = TokensCollector.newStringBuilderHandler();
67          Base.$.parse(Scanner.of(input), handler, true);
68          result = handler.get().toString();
69          assertThat(result).isEqualTo(input);
70  
71      }
72  
73      public interface RawBase extends Grammar {
74  
75          Token sep = Grammar.is('-');
76  
77          Token tok = Grammar.range('0', '9').or(Grammar.is(sep)).asToken();
78  
79          @MainRule
80          Rule rul = tok.oneOrMore();
81  
82          RawBase $ = Grammar.$();
83  
84      }
85  
86      public interface ExtRaw extends RawBase {
87  
88          Token sep = Grammar.is('_');
89  
90          ExtRaw $ = Grammar.$();
91  
92      }
93  
94      @Test
95      public void charOptimization_ShouldBe_blocked() throws IOException {
96          // first pass : Ext has not yet been initialized
97          String input = "2018-01-16";
98          TokensCollector<StringBuilder> handler = TokensCollector.newStringBuilderHandler();
99          RawBase.$.parse(Scanner.of(input), handler, true);
100         String result = handler.get().toString();
101         assertThat(result).isEqualTo(input);
102 
103         // 2nd pass : check Ext with its changes
104         input = "2018_01_16";
105         handler = TokensCollector.newStringBuilderHandler();
106         ExtRaw.$.parse(Scanner.of(input), handler, true);
107         result = handler.get().toString();
108         assertThat(result).isEqualTo(input);
109 
110         // 3rd pass : ensure that Ext didn't affect Base
111         input = "2018-01-16";
112         handler = TokensCollector.newStringBuilderHandler();
113         RawBase.$.parse(Scanner.of(input), handler, true);
114         result = handler.get().toString();
115         assertThat(result).isEqualTo(input);

117     }
