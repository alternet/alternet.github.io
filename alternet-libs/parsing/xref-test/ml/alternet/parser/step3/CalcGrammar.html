8   import ml.alternet.parser.Grammar;
9   import ml.alternet.parser.Grammar.WhitespacePolicy;
10  
11  @WhitespacePolicy
12  public interface CalcGrammar extends Grammar {
13  
14      // FUNCTION ::= 'sin' | 'cos' | 'exp' | 'ln' | 'sqrt'
15      enum Function implements EvaluableFunction {
16          sin {
17              @Override
18              public double eval(double value) {
19                  return Math.sin(value);
20              }
21          }, cos {
22              @Override
23              public double eval(double value) {
24                  return Math.cos(value);
25              }
26          }, exp {
27              @Override
28              public double eval(double value) {
29                  return Math.exp(value);
30              }
31          }, ln {
32              @Override
33              public double eval(double value) {
34                  return Math.log(value);
35              }
36          }, sqrt {
37              @Override
38              public double eval(double value) {
39                  return Math.sqrt(value);
40              }
41          };
42      }
43      Token FUNCTION = is(Function.class);
44  
45      // LBRACKET ::= '('
46      @Fragment Token LBRACKET = is('(');
47      // RBRACKET ::= ')'
48      @Fragment Token RBRACKET = is(')');
49      // RAISED ::= '^'
50      Token RAISED = is('^');
51  
52      // ADDITIVE ::= '+' | '-'
53      enum Additive {
54          PLUS("+"), MINUS("-");
55          Additive(String str) {
56              replace(Additive.class, this, s -> str);
57          }
58      }
59      Token ADDITIVE = is(Additive.class);
60  
61      // MULTIPLICATIVE ::= '*' | '/'
62      enum Multiplicative {
63          MULT("*"), DIV("/");
64          Multiplicative(String str) {
65              replace(Multiplicative.class, this, s -> str);
66          }
67      }
68      Token MULTIPLICATIVE = is(Multiplicative.class);
69  
70      // DIGIT ::= [0-9]
71      @WhitespacePolicy(preserve=true)
72      @Fragment Token DIGIT = range('0', '9').asNumber();
73  
74      // NUMBER ::= DIGIT+
75      Token NUMBER = DIGIT.oneOrMore()
76              .asNumber();
77  
78      // UPPERCASE ::= [A-Z]
79      @WhitespacePolicy(preserve=true)
80      @Fragment Token UPPERCASE = range('A', 'Z');
81      // LOWERCASE ::= [a-z]
82      @WhitespacePolicy(preserve=true)
83      @Fragment Token LOWERCASE = range('a', 'z');
84  
85      @WhitespacePolicy(preserve=true)
86      @Fragment Token UNDERSCORE = is('_');
87  
88      // VARIABLE ::= ([a-z] | [A-Z]) ([a-z] | [A-Z] | DIGIT | '_')*
89      Token VARIABLE = (LOWERCASE).or(UPPERCASE).seq(
90              (LOWERCASE).or(UPPERCASE).or(DIGIT).or(UNDERSCORE).zeroOrMore() )
91              .asToken();
92  
93      // Value ::= NUMBER | VARIABLE
94      @Fragment Rule Value = NUMBER.or(VARIABLE);
95  
96      // Expression ::= Sum
97  //    Proxy Sum = proxy();
98      @MainRule @Fragment Rule Expression = $("Sum");
99  
100     // Argument ::= FUNCTION Argument | Value | '(' Expression ')'
101     @Fragment Rule Argument = ( FUNCTION.seq($self) ).or(Value).or( LBRACKET.seq( Expression, RBRACKET ) );
102 
103     @Fragment Proxy SignedFactor = $();
104     // Factor ::= Argument ('^' SignedFactor)?
105     Rule Factor = Argument.seq( RAISED.seq(SignedFactor).optional() );
106 
107     // Product ::= Factor (MULTIPLICATIVE SignedFactor)*
108     Rule Product = Factor.seq(MULTIPLICATIVE.seq(SignedFactor).zeroOrMore());
109 
110     // SignedTerm ::= ADDITIVE? Product
111     @Fragment Rule SignedTerm = ADDITIVE.optional().seq(Product);
112 
113     // Sum ::= SignedTerm (ADDITIVE Product)*
114     Rule Sum = SignedTerm.seq(ADDITIVE.seq(Product).zeroOrMore());
115 
116     boolean init =
117     // SignedFactor ::= ADDITIVE? Factor
118     SignedFactor.is(
119         ADDITIVE.optional().seq(Factor)
120     );
121 
122     CalcGrammar Calc = $();
123 
124 }
