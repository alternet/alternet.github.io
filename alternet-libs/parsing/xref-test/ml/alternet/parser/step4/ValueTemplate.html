7   public interface ValueTemplate extends Grammar {
8   
9       @Fragment Token LCB = is('{');
10      @Fragment Token RCB = is('}');
11  
12      Token ESCAPE_LCB = is("{{"); // a double {{ is an escape for {
13      Token ESCAPE_RCB = is("}}"); // a double }} is an escape for }
14  
15      //   EXPRESSION ::= Calc
16      Token EXPRESSION = is(
17              Calc.$,
18              () -> new ExpressionBuilder());
19  
20      //   Text ::= ('{{' | '}}' | !('{' & '}'))*
21      Rule Text = ESCAPE_LCB.or(ESCAPE_RCB, isNot(LCB, RCB)).zeroOrMore();
22  
23      //   ValueTemplate ::= ( Text EXPRESSION? )*
24      @MainRule
25      Rule ValueTemplate = Text.seq(LCB.seq(EXPRESSION.optional(), RCB).optional()).zeroOrMore();
26  
27      ValueTemplate $ = $();
28  
29  }
