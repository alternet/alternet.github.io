12  /**
13   * <pre># Challenge Basic
14   * WWW-Authenticate: Basic realm="FooCorp"
15   *
16   * # Challenge OAuth 2.0 après l'envoi d'un token expiré
17   * WWW-Authenticate: Bearer realm="FooCorp", error=invalid_token, error_description="The access token has expired"</pre>
18   *
19   * @author Philippe Poulard
20   */
21  public interface WAuth extends Grammar {
22  
23  /*
24      # from RFC-2617 (HTTP Basic and Digest authentication)
25  
26      challenge      = auth-scheme 1*SP 1#auth-param
27      auth-scheme    = token
28      auth-param     = token "=" ( token | quoted-string )
29  
30      # from RFC2616 (HTTP/1.1)
31  
32      token          = 1*<any CHAR except CTLs or separators>
33      separators     = "(" | ")" | "<" | ">" | "@"
34                     | "," | ";" | ":" | "\" | <">
35                     | "/" | "[" | "]" | "?" | "="
36                     | "{" | "}" | SP | HT
37      quoted-string  = ( <"> *(qdtext | quoted-pair ) <"> )
38      qdtext         = <any TEXT except <">>
39      quoted-pair    = "\" CHAR
40  */
41  
42      @Fragment Token SEPARATORS = isOneOf("()<>@,;:\\\"/[]?={} \t");
43      @Fragment Token CTRLS = range(0, 31).union(127); // octets 0 - 31 and DEL (127)
44  
45      @WhitespacePolicy(preserve=true)
46      @Fragment Token TOKEN_CHAR = isNot(SEPARATORS, CTRLS);
47  
48      @WhitespacePolicy
49      Token TOKEN = TOKEN_CHAR.oneOrMore().asToken();
50  
51      @Fragment Token DOUBLE_QUOTE = is('"');
52      @Fragment Token BACKSLASH = is('\\')
53              .drop();
54  
55      @WhitespacePolicy(preserve=true)
56      @Fragment Token QdText = isNot(DOUBLE_QUOTE);
57  
58      @WhitespacePolicy(preserve=true)
59      @Fragment Token QuotedPair = BACKSLASH.seq($any).asToken();
60  
61      Token QuotedString = DOUBLE_QUOTE.drop().seq(
62              QuotedPair.or(QdText).zeroOrMore(),
63              DOUBLE_QUOTE.drop())
64          .asToken();
65  
66      Token ParameterValue = TOKEN.or(QuotedString).asToken();
67  
68      @Fragment Token EQUAL = is('=');
69  
70      // Parameter ::= TOKEN     EQUAL  ParameterValue
71      Rule Parameter = TOKEN.seq(EQUAL, ParameterValue);
72  
73      @WhitespacePolicy
74      @Fragment Token COMMA = is(',');
75  
76      // Parameters ::= Parameter (COMMA Parameter?)*
77      Rule Parameters = Parameter
78              .seq(COMMA.seq(Parameter.optional()).zeroOrMore());
79  
80      // Challenge ::= TOKEN     Parameters
81      @MainRule
82      Rule Challenge = TOKEN.seq(Parameters);
83  
84      WAuth $ = $();
85  
86  }
