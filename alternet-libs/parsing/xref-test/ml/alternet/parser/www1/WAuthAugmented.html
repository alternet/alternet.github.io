23  public interface WAuthAugmented extends WAuth {
24  
25      TypedToken<Parameter> Parameter = WAuth.Parameter
26          .asToken(tokens ->
27              new Parameter(
28                      tokens.getFirst().getValue(), // TOKEN
29                      tokens.getLast().getValue()   // ParameterValue
30          ));
31  
32      // Parameters ::= Parameter (COMMA Parameter?)*
33      TypedToken<List<Parameter>> Parameters = WAuth.Parameters
34          .asToken(tokens ->
35                  tokens.stream()
36                      // drop ","
37                      .filter(t -> t.getRule() != COMMA)
38                      // extract the value as a Parameter
39                      .map(t -> (Parameter) t.getValue())
40                      .collect(toList())
41          );
42  
43      @MainRule
44      TypedToken<Challenge> Challenge = WAuth.Challenge
45          .asToken(tokens -> new Challenge(
46                  tokens.removeFirst().getValue(),
47                  tokens.removeFirst().getValue())
48              );

52      class Parameter {
53  
54          public String name;
55          public String value;
56  
57          public Parameter(String name, String value) {
58              this.name = name;
59              this.value = value;
60          }
61  
62          @Override
63          public boolean equals(Object obj) {
64              if (obj instanceof Parameter) {
65                  Parameter p = (Parameter) obj;
66                  return p.name.equals(this.name)
67                          && p.value.equals(this.value);
68              } else return false;
69          }
70  
71          @Override
72          public int hashCode() {
73              return Objects.hash(this.name, this.value);
74          }
75  
76          @Override
77          public String toString() {
78              return name + "=" + value;
79          }
80  
81      }
82  
83      class Challenge {
84  
85          public String scheme;
86          public List<Parameter> parameters;
87  
88          public Challenge(String scheme, List<Parameter> parameters) {
89              this.scheme = scheme;
90              this.parameters = parameters;
91          }
92  
93      }
