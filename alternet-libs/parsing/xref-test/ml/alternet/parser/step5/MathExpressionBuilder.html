36      /**
37       * A factor expression is made of multiplicative (ร|รท) terms.
38       */
39      static class MathProduct implements NumericExpression {
40  
41          List<Term<MathMultiplicative>> arguments = new ArrayList<>();
42  
43          public MathProduct(List<Term<MathMultiplicative>> arguments) {
44              this.arguments = arguments;
45          }
46  
47          @Override
48          public Number eval(Map<String, Number> variables) {
49              double product = this.arguments.stream()
50                  .reduce(1d,
51                      (val, term) -> term.operation == MathMultiplicative.DIV ?
52                              val / term.term.eval(variables).doubleValue()
53                          :   val * term.term.eval(variables).doubleValue(),
54                      (t1, t2) -> t1 * t2);
55              return product;
56          }
57  
58          @Override
59          public String toString() {
60              return arguments.get(0).term
61                  + arguments.stream()
62                  .skip(1)
63                  .map(t -> "" + t.operation + t.term)
64                  .collect(Collectors.joining());
65          }
66  
67      }

69      enum MathRules implements RuleMapper<NumericExpression> {
70  
71          Product( (stack, rule, args) -> {
72              // Product ::= Factor (MULTIPLICATIVE SignedFactor)*
73              if (args.size() == 1) {
74                  // a single term is not a product
75                  return args.pollFirst().getTarget();
76              } else {
77                  // assume x to be *x, because the product will start by 1*x
78                  Term<MathMultiplicative> factor = new Term<>(MathMultiplicative.MULT, args.removeFirst().getTarget());
79                  List<Term<MathMultiplicative>> arguments = new LinkedList<>();
80                  arguments.add(factor);
81                  args.stream()
82                      // next arguments are all Term<Multiplicative>
83                      .map(v -> (Term<MathMultiplicative>) v.getTarget())
84                      .forEachOrdered(arguments::add);
85                  return new MathProduct(arguments);
86              }
87          });
88  
89          static {
90              EnumUtil.extend(ExpressionBuilder.CalcRules.class);
91          }
92  
93          RuleMapper<NumericExpression> rm;
94  
95          MathRules(ExpressionBuilder.CalcRules cr) {
96              this.rm = cr::transform;
97          }
98  
99          MathRules(RuleMapper<NumericExpression> rm) {
100             this.rm = rm;
101         }
102 
103         @Override
104         public NumericExpression transform(ValueStack<Value<NumericExpression>> stack, Rule rule, Deque<Value<NumericExpression>> args) {
105             return rm.transform(stack, rule, args);
106         }

110     enum MathTokens implements TokenMapper<NumericExpression> {
111 
112         MULTIPLICATIVE( (stack, token, next) -> {
113             // e.g. a ร b
114             MathMultiplicative op = token.getValue(); // ร | รท
115             // ร is always followed by an argument
116             NumericExpression arg = next.pollFirst().getTarget(); // b argument
117             Term<MathMultiplicative> term = new Term<>(op, arg);
118             return term;
119         });
