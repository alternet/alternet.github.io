18  public interface Math extends Calc {

20      // MULTIPLICATIVE ::= '×' | '÷'
21      enum MathMultiplicative {
22          MULT("×"), DIV("÷");
23          MathMultiplicative(String str) {
24              EnumUtil.replace(this, s -> str);
25          }
26      }
27      // same name than in CalcGrammar -> automatic replacement
28      Token MULTIPLICATIVE = is(MathMultiplicative.class);
29  
30      // FUNCTION ::= 'sin' | 'cos' | 'exp' | 'ln' | 'sqrt' | 'asin' | 'acos'
31      enum MathFunction implements EvaluableFunction {
32          sin(Calc.Function.sin::eval),
33          cos(Calc.Function.cos::eval),
34          exp(Calc.Function.exp::eval),
35          ln(Calc.Function.ln::eval),
36          sqrt(Calc.Function.sqrt::eval),
37          asin(n -> java.lang.Math.asin(n.doubleValue())),
38          acos(n -> java.lang.Math.acos(n.doubleValue()));
39  
40          java.util.function.Function<java.lang.Number, java.lang.Number> fun;
41  
42          @Override
43          public java.lang.Number eval(java.lang.Number value) {
44              return fun.apply(value);
45          }
46  
47          MathFunction(java.util.function.Function<java.lang.Number, java.lang.Number> fun) {
48              this.fun = fun;
49          }
50      }
51  
52      Token FUNCTION = is(MathFunction.class);
53  
54      // VARIABLE ::= [A-Z] ([A-Z] | DIGIT | '_')*
55      Token VARIABLE = UPPERCASE.seq(
56              UPPERCASE.or(DIGIT, UNDERSCORE).zeroOrMore() )
57              .asToken();
