5   import static ml.alternet.util.EnumUtil.replace;

7   import ml.alternet.parser.Grammar;
8   
9   public interface Calc extends Grammar {
10  
11      // FUNCTION ::= 'sin' | 'cos' | 'exp' | 'ln' | 'sqrt'
12      enum Function {
13          sin, cos, exp, ln, sqrt;
14      }
15      Token FUNCTION = is(Function.class);
16  
17      Token LBRACKET = is('(');
18      Token RBRACKET = is(')');
19      Token RAISED = is('^');
20  
21      // ADDITIVE ::= '+' | '-'
22      enum Additive {
23          PLUS("+"), MINUS("-");
24          Additive(String str) {
25              replace(Additive.class, this, s -> str);
26          }
27      }
28      Token ADDITIVE = is(Additive.class);
29  
30      // MULTIPLICATIVE ::= '*' | '/'
31      enum Multiplicative {
32          MULT("*"), DIV("/");
33          Multiplicative(String str) {
34              replace(Multiplicative.class, this, s -> str);
35          }
36      }
37      Token MULTIPLICATIVE = is(Multiplicative.class);
38  
39      // DIGIT ::= [0-9]
40      @Fragment Token DIGIT = range('0', '9').asNumber();
41  
42      // NUMBER ::= DIGIT+
43      Token NUMBER = DIGIT.oneOrMore()
44              .asNumber();
45  
46      // UPPERCASE ::= [A-Z]
47      @Fragment Token UPPERCASE = range('A', 'Z');
48      // LOWERCASE ::= [a-z]
49      @Fragment Token LOWERCASE = range('a', 'z');
50  
51      @Fragment Token UNDERSCORE = is('_');
52  
53      // VARIABLE ::= ([a-z] | [A-Z]) ([a-z] | [A-Z] | DIGIT | '_')*
54      Token VARIABLE = LOWERCASE.or(UPPERCASE).seq(
55              LOWERCASE.or(UPPERCASE, DIGIT, UNDERSCORE).zeroOrMore() )
56              .asToken();
57  
58      // Value ::= NUMBER | VARIABLE
59      @Fragment Rule Value = NUMBER.or(VARIABLE);
60  
61      @MainRule Proxy Expression = $();
62      // Argument ::= Value | FUNCTION Argument | '(' Expression ')'
63      @Fragment Rule Argument = Value.or( FUNCTION.seq($self) ).or( LBRACKET.seq( Expression, RBRACKET ) );
64  
65      Proxy SignedFactor = $();
66      // Factor ::= Argument ('^' SignedFactor)?
67      Rule Factor = Argument.seq( RAISED.seq(SignedFactor).optional() );
68  
69      // SignedFactor ::= ADDITIVE? Factor
70      boolean b2 = SignedFactor.is(
71          ADDITIVE.optional().seq(Factor)
72      );
73  
74      Proxy TermOp = $();
75      // Term ::= Factor TermOp
76      Rule Term = Factor.seq(TermOp);
77      // TermOp ::= (MULTIPLICATIVE SignedFactor TermOp)?
78      boolean b3 = TermOp.is(
79          MULTIPLICATIVE.seq(SignedFactor, TermOp).optional()
80      );
81  
82      // SignedTerm ::= ADDITIVE? Term
83      Rule SignedTerm = ADDITIVE.optional().seq(Term);
84  
85      // SumOp ::= (ADDITIVE Term SumOp)?
86      Rule SumOp = ADDITIVE.seq(Term, $self).optional();
87      // Expression ::= SignedTerm SumOp
88      static Rule Expression() {
89          return SignedTerm.seq(SumOp);
90      }
91  
92      Calc $ = $(Calc.class);
93  
94  }
