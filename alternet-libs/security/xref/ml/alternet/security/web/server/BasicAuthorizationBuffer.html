9   /**
10   * Extract a password of an HTTP Basic Authorization header,
11   * and replace it with '*'.
12   *
13   * <p>When creating, one must specify the boundaries of the
14   * portion of the buffer to analyze, and what this portion
15   * contains (all the HTTP headers, only the Authorization header,
16   * or only its value).</p>
17   *
18   * @author Philippe Poulard
19   */
20  public abstract class BasicAuthorizationBuffer {
21  
22      /**
23       * The scope of the lookup.
24       *
25       * @author Philippe Poulard
26       */
27      public enum Scope {
28          /** The buffer contains all the HTTP headers. */
29          Headers,
30          /** The buffer contains the line "Authorization: Basic [base64Credential]" */
31          AuthorizationHeader,
32          /** The buffer contains the value of the authorization header "Basic [base64Credential]" */
33          AuthorizationHeaderValue
34      }
35  
36      int position;
37      int limit;
38      Scope scope;
39  
40      /**
41       * Create a buffer.
42       *
43       * @param scope Indicates what is delimited in the buffer :
44       *      all the headers, just the Authorization header,
45       *      or simply its value.
46       * @param position The start offset of the bytes.
47       * @param limit The end offset of the bytes.
48       */
49      public BasicAuthorizationBuffer(Scope scope, int position, int limit) {
50          this.position = position;
51          this.limit = limit;
52          this.scope = scope;
53      }
54  
55      /**
56       * Get the byte at the index specified.
57       *
58       * @param i The actual index.
59       * @return The byte.
60       */
61      public abstract byte get(int i);
62  
63      /**
64       * Set a byte at the index specified.
65       *
66       * @param i The actual index.
67       * @param b The byte to set.
68       */
69      public abstract void set(int i, byte b);
70  
71      /**
72       * Log informations for debugging.
73       *
74       * @param msg The debug message.
75       */
76      public abstract void debug(String msg);
77  
78      // states of the parser
79      public static final int READ_AUTH = 0;
80      public static final int READ_UNTIL_CRLF = 1;
81      public static final int READ_OWS_BASIC = 2;
82      public static final int READ_BASIC = 3;
83      public static final int READ_CRED = 4;
84      public static final int READ_LF = 5;
85      public static final int READ_LAST_LF = 6;
86  
87      private static final String AUTHORIZATION = "authorization:";
88      private static final String BASIC = "basic ";
89  
90      int auth = 0;
91      int basicPos = 0;
92      int state = READ_AUTH;
93      int credStart = -1;
94      int credEnd = -1;
95  
96      /**
97       * Find the boundaries of the base64 credentials in the buffer.
98       *
99       * <p>String to find in the buffer : "Authorization: Basic [base64Credential]"
100      * (or just "Basic [base64Credential]" according to the scope)</p>
101      * <pre>    header-field = field-name ":" OWS field-value OWS</pre>
102      * <p>OWS means "optional whitespace" and header fields are delimited using CRLF.</p>
103      *
104      * @return <code>true</code> if the credentials have been found,
105      *         <code>false</code> otherwise.
106      */
107     public boolean findCredentialsBoundaries() {
108         if (scope == Scope.AuthorizationHeaderValue) {
109             state = READ_BASIC; // start here
110         }
111         int pos;
112         loop: for (pos = position ; pos < limit ; pos++) {
113             char c = (char) ( get(pos) & 0x7f );
114             c = Character.toLowerCase(c);
115             switch (state) {
116             case READ_AUTH:
117                 if (c == AUTHORIZATION.charAt(auth)) {
118                     auth++;
119                     if (auth == AUTHORIZATION.length()) {
120                         state = READ_OWS_BASIC;
121                     }
122                     break;
123                 } else if (scope == Scope.AuthorizationHeaderValue || scope == Scope.AuthorizationHeader) {
124                     return false;
125                 } else if (c == '\r') {
126                     auth = 0;
127                     state = READ_LAST_LF;
128                     break;
129                 } else {
130                     auth = 0;
131                     state = READ_UNTIL_CRLF;
132                     break;
133                 }
134             case READ_UNTIL_CRLF:
135                 if (c == '\r') {
136                     state = READ_LF;
137                 }
138                 break;
139             case READ_OWS_BASIC:
140                 if (c == ' ') {
141                     break;
142                 } else {
143                     state = READ_BASIC;
144                     // do NOT break
145                 }
146             case READ_BASIC:
147                 if (c == BASIC.charAt(basicPos)) {
148                     basicPos++;
149                     if (basicPos == BASIC.length()) {
150                         state = READ_CRED;
151                         credStart = pos + 1;
152                     }
153                 } else if (scope == Scope.AuthorizationHeaderValue || scope == Scope.AuthorizationHeader) {
154                     return false;
155                 } else {
156                     basicPos = 0;
157                     state = READ_UNTIL_CRLF;
158                 }
159                 break;
160             case READ_CRED:
161                 // TODO : handle EOF ?
162                 if (c == ' ') {
163                     credEnd = pos;
164                     break loop;
165                 } else if (c == '\r') {
166                     credEnd = pos;
167                     state = READ_LF;
168                 }
169                 break;
170             case READ_LF:
171                 if (c != '\n') {
172                     // error will be raised normally
173                     credEnd = -1;
174                     break loop;
175                 }
176                 if (credEnd > 0) {
177                     break loop;
178                 }
179                 state = READ_AUTH;
180                 break;
181             case READ_LAST_LF:
182                 if (c != '\n') {
183                     // error will be raised normally
184                     credEnd = -1;
185                 }
186                 // end of headers
187                 break loop;
188             default:
189             }
190         }
191         if (scope == Scope.AuthorizationHeaderValue || scope == Scope.AuthorizationHeader) {
192             // we weren't able to handle end of line
193             credEnd = pos; // the last pos known is the end
194             return true;
195         }
196         return credEnd != -1;
197     }

199     /**
200      * Replace the raw password in the buffer with '*'
201      * and reencode the credentials in Base64.
202      *
203      * <p>This method have to be called if {@link #findCredentialsBoundaries()}
204      * has returned <code>true</code>.</p>
205      *
206      * @param passwordManager Allow to create a secure password.
207      *
208      * @return The credentials with the captured password and the user name.
209      */
210     public Credentials replace(PasswordManager passwordManager) {
211         Credentials credentials = new Credentials();
212         byte[] encodedCred = new byte[credEnd - credStart];
213         for (int i = 0 ; i < encodedCred.length ; i++) {
214             encodedCred[i] = get(credStart + i);
215         }
216         byte[] decodedCred = Base64.getDecoder().decode(encodedCred);
217         // "encodedCred" will be cleaned later
218 
219         char[] password = null;
220         int pwdStart = -1;
221         for (int i = 0 ; i < decodedCred.length ; i++) {
222             if (pwdStart == -1) {
223                 if (decodedCred[i] == ':') {
224                     credentials.withUser(new String(decodedCred, 0, i));
225                     pwdStart = i + 1;
226                     password = new char[decodedCred.length - pwdStart];
227                 }
228             } else {
229                 password[i - pwdStart] = (char) decodedCred[i];
230                 decodedCred[i] = '*';
231                 // this is the pwd value that will be available later in HttpField
232             }
233         }
234         byte[] reEncodedCred = Base64.getEncoder().encode(decodedCred);
235         // not sure whether "reEncodedCred" has the same length of "encodedCred"
236         for (int i = 0 ; i < encodedCred.length ; i++) {
237             encodedCred[i] = ' '; // clean intermediate data
238             byte b = ' '; // fill with blanks if "reEncodedCred" was too short
239             if (i < reEncodedCred.length) {
240                 b = reEncodedCred[i];
241             }
242             set(credStart + i, b);
243         }
244         debug("HTTP Headers successfully parsed");
245         // now, we have it, and the buffer is filled with "*****" instead
246         Password pwd = passwordManager.newPassword(password);
247         credentials.withPassword(pwd);
248         return credentials;
249     }
