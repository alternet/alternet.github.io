54      @Path("/")
55      public static class ExampleRequest {
56  
57          @Path("{example}")
58          @POST
59          @Produces(MediaType.APPLICATION_XML)
60          public Data example(@PathParam("example") String example,
61                  @FormParam("pwd") Password pwd, // here is the trick !!!
62                  @FormParam("pwd") String rawValue,
63                  @FormParam("who") String whoValue,
64                  @Context HttpServletRequest req) {
65  
66              // this attribute MUST have been set in the server settings
67              RESTServerTestHarness<?> that = (RESTServerTestHarness<?>) req.getServletContext().getAttribute(RESTServerTestHarness.class.getName());
68  
69              // the result of the request to return
70              Data data = new Data(pwd, rawValue);
71  
72              // collect some tests before returning
73  
74              name.to().RESTPasswordConverter_Should_SupplyPassword();
75              that.serverTests.put(name.ref(),
76                      () -> Assertions.assertThat(pwd)
77                      .as("REST server : REST password converter should supply password")
78                      .isNotNull()
79              );
80  
81              PasswordState state  = pwd.state();
82              name.to().RESTPasswordConverter_Should_SupplyValidPassword();
83              that.serverTests.put(name.ref(),
84                      () -> Assertions.assertThat(state)
85                      .as("REST server : REST password converter should supply valid password")
86                      .isSameAs(PasswordState.Valid)
87              );
88  
89              char[] clearPwd = pwd.getClearCopy().get();
90              char[] reqPwd = Passwords.getPasswords(req, "pwd").getClearCopy().get();
91              name.to().RESTPasswordConverter_Should_ConvertRequestPassword();
92              that.serverTests.put(name.ref(),
93                      () -> Assertions.assertThat(clearPwd)
94                      .as("REST server : REST password converter should convert request password")
95                      .isEqualTo(reqPwd)
96              );
97  
98              PasswordState who = Passwords.getPasswords(req, "who").state();
99              name.to().unrelatedFields_ShouldNot_BeCaptured();
100             that.serverTests.put(name.ref(),
101                     () -> Assertions.assertThat(who)
102                     .as("REST server : unrelated fields should not be captured")
103                     .isSameAs(PasswordState.Empty)
104             );
105 
106             name.to().unrelatedFields_Should_HaveTheirValueUnchanged();
107             that.serverTests.put(name.ref(),
108                     () -> Assertions.assertThat(whoValue)
109                     .as("REST server : unrelated fields should have their value unchanged")
110                     .isEqualTo("azerty")
111             );
112 
113             name.to().rawValue_ShouldBe_FilledWithStars();
114             that.serverTests.put(name.ref(),
115                     () -> Assertions.assertThat(rawValue)
116                     .as("REST server : raw value should be filled with '*'")
117                     .matches("^\\*+$") // "*****"
118             );
119 
120             try {
121                 // the pwd received (and escaped) by the server is : da_AcTu%40%7C+P%40zzm0R%7C%29
122                 String clearUnsafePwd = URLEncoder.encode(new String(clearPwd), "UTF-8");
123                 name.to().rawValueAndPassword_Should_HaveTheSameLength();
124                 that.serverTests.put(name.ref(),
125                         () -> Assertions.assertThat(rawValue.length())
126                         .as("REST server : raw value and password should have the same length")
127                         .isEqualTo(clearUnsafePwd.length())
128                 );
129             } catch (UnsupportedEncodingException e) {
130                 throw new RuntimeException(e);
131             }
132             return data;
133         }
134     }

140     @Test(priority=1)
141     public void passwords_ShouldBe_capturedByServer() throws Exception {
142         String uri = "http://localhost:" + port + contextPath + "/Test";
143         Client client = ClientBuilder.newClient();
144         String unsafePwd = "da_AcTu@| P@zzm0R|)";
145         Form form = new Form();
146         form.param("who", "azerty");
147         form.param("pwd", unsafePwd);
148         Response response = client.target(uri).request().post(Entity.form(form));
149         Data data = response.readEntity(Data.class);
150 
151         Assertions.assertThat(data.pwd)
152             .as("REST client : raw value should be filled with '*'")
153             .matches("^\\*+$"); // "*****"
154         Assertions.assertThat(data.pwd.length())
155             .as("REST client : raw value and password should have the same length")
156             // the pwd received (and escaped) by the server is : da_AcTu%40%7C+P%40zzm0R%7C%29
157             .isEqualTo(URLEncoder.encode(unsafePwd, "UTF-8").length());
158         Assertions.assertThat(data.clearValue)
159             .as("REST client : clear value and password should be the same")
160             .isEqualTo(unsafePwd.toCharArray());
161     }
