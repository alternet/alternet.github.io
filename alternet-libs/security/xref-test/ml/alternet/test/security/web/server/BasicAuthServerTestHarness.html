50          @Path("{example}")
51          @GET
52          @Produces(MediaType.APPLICATION_XML)
53          public Data example(@PathParam("example") String example,
54                  @Context HttpHeaders httpHeaders,
55                  @Context HttpServletRequest req) throws UnsupportedEncodingException {
56  
57              // this attribute MUST have been set in the server settings
58              BasicAuthServerTestHarness<?> that = (BasicAuthServerTestHarness<?>) req.getServletContext().getAttribute(BasicAuthServerTestHarness.class.getName());
59  
60              List<String> authHeaders = httpHeaders.getRequestHeader(HttpHeaders.AUTHORIZATION);
61  
62              // collect some tests before returning
63  
64              name.to().HTTPAuthenticationMissing();
65              that.serverTests.put(name.ref(),
66                  () -> Assertions.assertThat(authHeaders.size())
67                  .isNotNull()
68                  .as("REST server : HTTP Authentication missing")
69              );
70  
71              String authHeader = authHeaders.get(0);
72  
73              name.to().HTTPBasicAuthenticationMissing();
74              that.serverTests.put(name.ref(),
75                  () -> Assertions.assertThat(authHeader.startsWith("Basic "))
76                  .isNotNull()
77                  .as("REST server : HTTP Basic Authentication missing")
78              );
79  
80              String[] cred = new String(
81                  DatatypeConverter.parseBase64Binary(
82                      authHeader.substring("Basic ".length())
83                  ), "ISO-8859-1").split(":");
84  
85              // the result of the request to return
86              Data data = new Data(cred[0], cred[1]);
87  
88              name.to().credential_Should_containTheRightUserName();
89              that.serverTests.put(name.ref(),
90                  () -> Assertions.assertThat(cred[0])
91                  .isEqualTo("who")
92                  .as("REST server : credential should contain the right user name")
93              );
94  
95              name.to().passwordHeader_ShouldBe_FilledWithStars();
96              that.serverTests.put(name.ref(),
97                  () -> Assertions.assertThat(cred[1])
98                  .as("REST server : password header should be filled with '*'")
99                  .matches("^\\*+$") // "*****"
100             );
101 
102             boolean uira = req.isUserInRole("admin");
103             name.to().user_ShouldHave_AdminRole();
104             that.serverTests.put(name.ref(),
105                 () -> Assertions.assertThat(uira)
106                 .as("REST server : user should have \"admin\" role")
107                 .isTrue()
108             );
109 
110             boolean uirc = req.isUserInRole("customer");
111             name.to().user_ShouldNotHave_CustomerRole();
112             that.serverTests.put(name.ref(),
113                 () -> Assertions.assertThat(uirc)
114                 .as("REST server : user should not have \"customer\" role")
115                 .isFalse()
116             );
117 
118             Credentials credentials = (Credentials) req.getAttribute(Passwords.BASIC_AUTH_ATTRIBUTE_KEY);
119 
120             name.to().request_Should_SupplyPassword();
121             that.serverTests.put(name.ref(),
122                 () -> Assertions.assertThat(credentials)
123                 .as("REST server : request should supply password")
124                 .isNotNull()
125             );
126 
127             Password pwd = credentials.getPassword();
128             PasswordState state  = pwd.state();
129             name.to().request_Should_SupplyValidPassword();
130             that.serverTests.put(name.ref(),
131                 () -> Assertions.assertThat(state)
132                 .as("REST server : request should supply valid password")
133                 .isSameAs(PasswordState.Valid)
134             );
135 
136             char[] clearPwd = pwd.getClearCopy().get();
137             name.to().password_Should_HaveTheRigthValue();
138             that.serverTests.put(name.ref(),
139                 () -> Assertions.assertThat(clearPwd)
140                 .as("REST server : password should have the rigth value")
141                 .isEqualTo(that.unsafePwd.toCharArray())
142             );
143 
144             return data;
145         }

152     @Test(priority=1)
153     public void authPassword_ShouldBe_capturedByServer() throws Exception {
154         String uri = "http://localhost:" + port + contextPath + "/Test";
155         Client client = ClientBuilder.newClient();
156         String unsafeCred = userName + ":" + unsafePwd;
157         Response response = client.target(uri)
158             .request()
159             .header(HttpHeaders.AUTHORIZATION, "Basic " +
160                 DatatypeConverter.printBase64Binary(unsafeCred.getBytes("UTF-8")))
161             .get();
162         Data data = response.readEntity(Data.class);
163 
164         Assertions.assertThat(data.pwd)
165             .as("REST client : password header should be filled with '*'")
166             .matches("^\\*+$"); // "*****"
167         Assertions.assertThat(data.user)
168             .as("REST client : user should be the same")
169             .isEqualTo("who");
170 
171         Map<String,NewCookie> newCookies = response.getCookies();
172         NewCookie session = newCookies.get("JSESSIONID");
173         Assertions.assertThat(session).isNotNull();
174     }
