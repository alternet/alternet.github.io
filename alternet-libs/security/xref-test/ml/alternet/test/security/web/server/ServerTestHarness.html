5   import java.util.concurrent.atomic.AtomicInteger;

7   import org.testng.annotations.AfterClass;
8   import org.testng.annotations.BeforeClass;

10  /**
11   * Base class for tests to run in a Web container.
12   *
13   * @author Philippe Poulard
14   *
15   * @param <T> The Web server.
16   */
17  public abstract class ServerTestHarness<T> {
18  
19      static AtomicInteger PORT = new AtomicInteger(18641);
20  
21      public int port = PORT.getAndIncrement();
22  
23      public T server;
24  
25      public String userName = "who";
26      public String unsafePwd = "da_AcTu@| P@zzm0R|)";
27      public String resourceBase = new File(System.getProperty("java.io.tmpdir")).getAbsolutePath();
28  
29      /** Tests registered inside the server but to run outside the server. */
30      protected ConcurrentHashMap<String, Runnable> serverTests = new ConcurrentHashMap<String, Runnable>();
31  
32      @BeforeClass
33      public final void startServer() throws Exception {
34          if (server == null) {
35              doStartServer();
36          }
37      }
38  
39      @AfterClass
40      public final void stopServer() throws Exception {
41          if (server != null) {
42              doStopServer();
43              server = null;
44          }
45      }
46  
47      public abstract void doStartServer() throws Exception;
48  
49      public abstract void doStopServer() throws Exception;
50  
51  }
