19  /**
20   * The Unix Crypt hasher.
21   *
22   * @author Philippe Poulard
23   */
24  public class UnixCryptHasher extends HasherBase<SaltedParts> {
25  
26      /**
27       * The builder for this hasher.
28       *
29       * @return The Unix Crypt hasher builder.
30       */
31  
32      /**
33       * UnixCryptHasher constructor.
34       *
35       * @param config The configuration of this hasher.
36       */
37      public UnixCryptHasher(Configuration config) {
38          super(config);
39      }
40  
41      /**
42       * The Unix Crypt formatter : 2 characters salt followed by the hash,
43       * each character encoded in the h64 encoder value space.
44       */
45      public static final CryptFormatter<SaltedParts> UNIXCRYPT_FORMATTER = new CryptFormatter<SaltedParts>() {
46          @Override
47          public SaltedParts parse(String crypt, Hasher hr) {
48              SaltedParts parts = new SaltedParts(hr);
49              BytesEncoding encoding = hr.getConfiguration().getEncoding();
50              String b64 = new String(BytesEncoder.ValueSpace.valueSpace(encoding));
51              // characters are individually mapped to the value space
52              parts.salt = new byte[2];
53              parts.salt[0] = (byte) b64.indexOf(crypt.charAt(0));
54              parts.salt[1] = (byte) b64.indexOf(crypt.charAt(1));
55              parts.hash = new byte[11];
56              for (int i = 2 ; i < 13 ; i++) {
57                  parts.hash[i-2] = (byte) b64.indexOf(crypt.charAt(i));
58              }
59              return parts;
60          }
61  
62          @Override
63          public String format(SaltedParts parts) {
64              byte[] b = new byte[13];
65              b[0] = parts.salt[0];
66              b[1] = parts.salt[1];
67              System.arraycopy(parts.hash, 0, b, 2, 11);
68              return new String(b, 0, 13, StandardCharsets.US_ASCII);
69          }
70  
71          @Override
72          public CryptFormat getCryptFormat() {
73              return new UnixCryptFormat();
74          }
75      };
76  
77      @Override
78      public String getScheme() {
79          return "CRYPT";
80      }
81  
82      @Override
83      public byte[] encrypt(Credentials credentials, SaltedParts parts) {
84          long keyword = 0L;
85          int salt = (parts.salt[1] << 6) | parts.salt[0];
86          try (Password.Clear clear = credentials.getPassword().getClearCopy()) {
87              byte[] pwd = SafeBuffer.getData(
88                  SafeBuffer.encode(
89                      CharBuffer.wrap(clear.get()), getConfiguration().getCharset()
90                  )
91              );
92              for (int i = 0; i < 8; i++) {
93                                                                  // cast byte to int
94                  keyword = (keyword << 8) | ((i < pwd.length) ? 2 * (pwd[i]  &  0xff) : 0);
95              }
96          }
97          try {
98              return UnixCrypt.encrypt(salt, keyword);
99          } finally {
100             keyword = 0L;
101         }
102     }
103 
104     @Override
105     public SaltedParts initializeParts() {
106         SaltedParts parts = new SaltedParts(this);
107         SecureRandom rand = new SecureRandom();
108         parts.salt = new byte[2];
109         rand.nextBytes(parts.salt);
110         // bytes are not in the right range
111         parts.salt[0] %= 64;
112         parts.salt[1] %= 64;
113         return parts;
114     }
