128     @DataProvider(name="pwdWithLogRounds")
129     Object[][] getPwdWithLogRounds() {
130         String[][] pwd = getPwd();
131         Object[][] o = new Object[pwd.length *9][2];
132         for (int i = 4; i <= 12; i++) {
133             for (int j =0 ; j < pwd.length; j++) {
134                 int k = (i-4) * pwd.length + j;
135                 o[k][0] = i;
136                 o[k][1] = pwd[j][0];
137             }
138         }
139         return o;
140     }

157     /**
158      * Test method for 'BCrypt.hashpw(String, String)'
159      * @throws InvalidAlgorithmParameterException
160      */
161     @Override
162     @Test(dataProvider="creds")
163     public void crypts_should_match(String plain, String salt, String expected) throws InvalidAlgorithmParameterException {
164         Credentials cred = Credentials.fromPassword(plain.toCharArray());
165         String hashed = encrypt(cred, salt);
166         if (salt.startsWith("$2b$")) {
167             // because $2b$ parse with $2b$ but format to $2a$
168             hashed = hashed.substring("$2b$".length());
169             expected = expected.substring("$2b$".length());
170         }
171         Assertions.assertThat(hashed).isEqualTo(expected);
172 
173         if (salt.startsWith("$2a$")) {
174             // check that BCrypt behaviour is the same
175             String hashedChecked = BCrypt.hashpw(plain, salt);
176             Assertions.assertThat(hashedChecked).isEqualTo(expected);
177         }
178     }

190     /**
191      * Test method for 'BCrypt.checkpw(String, String)'
192      * expecting success
193      * @throws InvalidAlgorithmParameterException
194      */
195     @Override
196     @Test(dataProvider="goodCreds")
197     public void checkPassword_should_success(String plain, String expected) throws InvalidAlgorithmParameterException {
198         Credentials cred = Credentials.fromPassword(plain.toCharArray());
199         Hasher hr = resolve(expected).build();
200         Assertions.assertThat(hr.check(cred, expected)).isTrue();
201 
202         if (expected.startsWith("$2a$")) {
203             // check that BCrypt behaviour is the same
204             Assertions.assertThat(BCrypt.checkpw(plain, expected)).isTrue();
205         }
206     }

208     /**
209      * Test method for 'BCrypt.gensalt(int)'
210      * @throws InvalidAlgorithmParameterException
211      */
212     @Test(dataProvider="pwdWithLogRounds")
213     public void logRound_should_affectCrypt(int logRounds, String plain) throws InvalidAlgorithmParameterException {
214         Hasher hr = ModularCryptFormatHashers.$2a$.get().setLogRounds(logRounds).build();
215         Credentials cred = Credentials.fromPassword(plain.toCharArray());
216         String hashed1 = hr.encrypt(cred);
217         String hashed2 = encrypt(cred, hashed1);
218         Assertions.assertThat(hashed1).isEqualTo(hashed2);
219 
220         // check that the log round was in the crypt
221         BCryptHasher bcrypt = newHasher(hashed1);
222         WorkFactorSaltedParts parts = (WorkFactorSaltedParts) bcrypt.getConfiguration().getFormatter()
223             .parse(hashed1, bcrypt);
224         Assertions.assertThat(parts.workFactor).isEqualTo(logRounds);
225     }

227     /**
228      * Test for correct hashing of non-US-ASCII passwords
229      * @throws InvalidAlgorithmParameterException
230      */
231     @Test
232     public void internationalChars_should_beHashedCorrectly() throws InvalidAlgorithmParameterException {
233         String pw1 = "\u2605\u2605\u2605\u2605\u2605\u2605\u2605\u2605";
234         String pw2 = "????????";
235         Credentials cred1 = Credentials.fromPassword(pw1.toCharArray());
236         Credentials cred2 = Credentials.fromPassword(pw2.toCharArray());
237 
238         String h1 = hr.encrypt(cred1);
239         Assertions.assertThat(hr.check(cred2, h1)).isFalse();
240 
241         String h2 = hr.encrypt(cred2);
242         Assertions.assertThat(hr.check(cred1, h2)).isFalse();
243     }
