25  /**
26   * The tests that show on BASIC authentication that the password is captured by
27   * Jetty (and encrypted), and that the raw value of the password in the
28   * incoming buffer is replaced with '*'.
29   *
30   * @author Philippe Poulard
31   */
32  @Test
33  public class BasicAuthTest extends BasicAuthServerTestHarness<Server> {
34  
35      static {
36          name = Methref.on(BasicAuthTest.class);
37      }
38  
39      @Override
40      @Test(enabled = false)
41      public void doStopServer() throws Exception {
42          server.stop();
43          server.destroy();
44          server = null;
45      }
46  
47      @Override
48      @Test(enabled = false)
49      public void doStartServer() throws Exception {
50          WebAppContext wac = new WebAppContext();
51          // a reference to this test
52          wac.setAttribute(BasicAuthServerTestHarness.class.getName(), this);
53  
54          AltHttpConnectionFactory cf = new AltHttpConnectionFactory(wac);
55  
56          server = new Server();
57          ServerConnector connector=new ServerConnector(server, cf);
58          connector.setPort(port);
59          server.setConnectors(new Connector[]{connector});
60  
61          ServletHolder sh = new ServletHolder("JERSEY_SERVLET", ServletContainer.class);
62          String classes = ExampleRequest.class.getName() + " ";
63          sh.setInitParameter(ServerProperties.PROVIDER_CLASSNAMES, classes);
64  
65          ServletHandler servletHandler = new ServletHandler();
66          servletHandler.addServletWithMapping(sh, "/*");
67  
68          wac.setServletHandler(servletHandler);
69          wac.setResourceBase(resourceBase);
70          wac.setContextPath(contextPath);
71  
72          // configure the security (standard fashion)
73          Constraint constraint = new Constraint();
74          constraint.setName(Constraint.__BASIC_AUTH);
75          constraint.setRoles(new String[]{"customer","admin"});
76          constraint.setAuthenticate(true);
77  
78          AltMappedLoginService loginService = new AltMappedLoginService();
79          loginService.setName("realm");
80          loginService.setCryptFormats(
81                  ModularCryptFormat.class.getName(),
82                  CurlyBracesCryptFormat.class.getName(),
83                  PlainTextCryptFormat.class.getName()
84          );
85          // store a user with its crypt
86          String crypt = ModularCryptFormatHashers.$2$.get().build()
87              .encrypt(unsafePwd.toCharArray());
88          loginService.putUser(userName, crypt, new String[] {"admin"});
89  
90          ConstraintSecurityHandler security = new ConstraintSecurityHandler();
91          security.setAuthMethod(Constraint.__BASIC_AUTH);
92          security.setLoginService(loginService);
93          ConstraintMapping cm = new ConstraintMapping();
94          cm.setConstraint(constraint);
95          cm.setPathSpec("/*");
96          security.setConstraintMappings(new ConstraintMapping[]{cm});
97  
98          wac.setSecurityHandler(security);
99          wac.setInitParameter(Config.AUTH_METHOD_INIT_PARAM, "Basic");
100 
101         server.setHandler(wac);
102 
103         server.start();
104 //        server.dumpStdErr();
105     }
