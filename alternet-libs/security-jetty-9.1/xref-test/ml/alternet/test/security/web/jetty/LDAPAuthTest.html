58      @Override
59      public String checkSession(HttpServletRequest req) {
60          SessionAuthentication sauth = (SessionAuthentication) req.getSession().getAttribute(SessionAuthentication.__J_AUTHENTICATED);
61          name.to().sessionAuthentication_ShouldBe_Define();
62          serverTests.put(name.ref(),
63                  () -> Assertions.assertThat(sauth)
64                  .as("REST server : session authentication should be defined")
65                  .isNotNull()
66                  );
67  
68          try {
69              // unable to check that sauth._credentials is "******" since it is a private field
70              Field field = SessionAuthentication.class.getDeclaredField("_credentials");
71              field.setAccessible(true);
72              String credentials = "" + field.get(sauth);
73              return credentials;
74          } catch (NoSuchFieldException | SecurityException | IllegalArgumentException | IllegalAccessException e) {
75              return Thrower.doThrow(e);
76          }
77      }

96      @Override
97      @Test(enabled = false)
98      public void doStartServer() throws Exception {
99          this.userName = "lsimpson";
100 
101         conf = Configuration.getConfiguration();
102         setJaasConfiguration();
103 
104         WebAppContext wac = new WebAppContext();
105         // a reference to this test
106         wac.setAttribute(FormAuthServerTestHarness.class.getName(), this);
107 
108         server = new Server();
109         AltHttpConnectionFactory cf = new AltHttpConnectionFactory(server);
110 
111         ServerConnector connector=new ServerConnector(server, cf);
112         connector.setPort(port);
113         server.setConnectors(new Connector[]{connector});
114 
115         ldapServer = new LdapServer();
116         ldapServer.setPort(10389);
117         ldapServer.setBaseDn("dc=alternet,dc=ml");
118         String ldifFile = LdapServer.class.getResource("users.ldif").getFile();
119         ldapServer.setLdifFiles(ldifFile);
120         server.addBean(ldapServer);
121 
122         ServletHolder sh = new ServletHolder("JERSEY_SERVLET", ServletContainer.class);
123         String classes = ExampleRequest.class.getName() + " ";
124         sh.setInitParameter(ServerProperties.PROVIDER_CLASSNAMES, classes);
125 
126         ServletHandler servletHandler = new ServletHandler();
127         servletHandler.addServletWithMapping(sh, "/protected/*");
128 
129         wac.setServletHandler(servletHandler);
130         wac.setResourceBase(resourceBase);
131         wac.setContextPath(contextPath);
132 
133         // configure the security (standard fashion)
134         Constraint constraint = new Constraint();
135         constraint.setName(Constraint.__FORM_AUTH);
136         constraint.setRoles(new String[]{"customer","admin"});
137         constraint.setAuthenticate(true);
138 
139         JAASLoginService loginService = new JAASLoginService();
140         loginService.setName("Alternet Realm");
141         loginService.setLoginModuleName("ldap");
142 
143         ConstraintSecurityHandler security = new ConstraintSecurityHandler();
144         security.setRealmName("Alternet Realm");
145         security.setLoginService(loginService);
146         security.setAuthMethod(Constraint.__FORM_AUTH);
147         ConstraintMapping cm = new ConstraintMapping();
148         cm.setConstraint(constraint);
149         cm.setPathSpec("/protected/*");
150         security.setConstraintMappings(new ConstraintMapping[]{cm});
151         security.setInitParameter(FormAuthenticator.__FORM_LOGIN_PAGE, "/login.html");
152         security.setInitParameter(FormAuthenticator.__FORM_ERROR_PAGE, "/authFail.html");
153 
154         wac.setSecurityHandler(security);
155         wac.setInitParameter(Config.AUTH_METHOD_INIT_PARAM, "Form");
156 
157         server.setHandler(wac);
158 
159         server.start();
160         //        server.dumpStdErr();
161         System.out.println("WEBAPPS :" + server.getBeans(WebAppContext.class));
162     }

164     protected void setJaasConfiguration() {
165         Configuration.setConfiguration(new Configuration(){
166             @Override
167             public AppConfigurationEntry[] getAppConfigurationEntry(String name){
168                 if ("ldap".equals(name)) {
169                     final Map<String,Object> options=new HashMap<String,Object>();
170 
171                     options.put("useSSL", "false");
172                     options.put("debug","true");
173                     options.put("contextFactory","com.sun.jndi.ldap.LdapCtxFactory");
174                     options.put("hostname","localhost");
175                     options.put("port","10389");
176 
177                     options.put("bindDn","ou=people,dc=alternet,dc=ml");
178                     options.put("bindPassword","secret");
179                     options.put("authenticationMethod", "simple");
180 
181                     options.put("forceBindingLogin", "false");
182                     options.put(CryptFormat.class.getName(),
183                             ModularCryptFormat.class.getName()
184                             + " , " + CurlyBracesCryptFormat.class.getName()
185                             + "," + PlainTextCryptFormat.class.getName());
186 
187                     options.put("userBaseDn", "ou=people,dc=alternet,dc=ml");
188                     options.put("userRdnAttribute", "uid");
189                     options.put("userIdAttribute", "uid");
190                     options.put("userPasswordAttribute", "userPassword");
191                     options.put("userObjectClass", "inetOrgPerson");
192 
193                     options.put("roleBaseDn", "ou=groups,dc=alternet,dc=ml");
194                     options.put("roleNameAttribute", "cn");
195                     options.put("roleMemberAttribute", "member");
196                     options.put("roleObjectClass", "groupOfNames");
197 
198                     AppConfigurationEntry appConf = new AppConfigurationEntry(AltLdapLoginModule.class.getName(), LoginModuleControlFlag.REQUIRED, options);
199                     return new AppConfigurationEntry[]{ appConf };
200                 } else {
201                     return null;
202                 }
203             }
204         });
205     }
