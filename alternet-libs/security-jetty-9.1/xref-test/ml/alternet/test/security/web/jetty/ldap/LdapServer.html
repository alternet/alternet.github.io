20  /**
21   * An embedded LDAP server that can be attached to the Jetty life-cycle.
22   *
23   * Typical usage :
24   * <pre>
25   * &lt;?xml version="1.0"?&gt;
26   * &lt;!DOCTYPE Configure PUBLIC "-//Jetty//Configure//EN" "http://www.eclipse.org/jetty/configure_9_2.dtd"&gt;
27   * &lt;Configure id="Server" class="org.eclipse.jetty.server.Server"&gt;
28   *   &lt;Call name="addBean"&gt;
29   *     &lt;Arg&gt;
30   *       &lt;New class="ml.alternet.test.security.web.jetty.ldap.LdapServer"&gt;
31   *         &lt;Set name="port"&gt;&lt;Property name="ldap.port" default="10389"/&gt;&lt;/Set&gt;
32   *         &lt;Set name="baseDn"&gt;&lt;Property name="ldap.baseDn" default="dc=alternet,dc=ml"/&gt;&lt;/Set&gt;
33   *         &lt;Set name="ldifFiles"&gt;&lt;Property name="ldap.ldifFiles"/&gt;&lt;/Set&gt;
34   *       &lt;/New&gt;
35   *     &lt;/Arg&gt;
36   *   &lt;/Call&gt;
37   * &lt;/Configure&gt;
38   * </pre>
39   * 
40   * <h1>NOTE</h1>
41   * This LDAP server relies on <a href="https://www.ldap.com/unboundid-ldap-sdk-for-java">UnboundID LDAP SDK for Java</a>.
42   * It appears that hash passwords are not supported by this version,
43   * therefore the configuration (typically in <code>ldap.conf</code> file) should indicate
44   * <code>forceBindingLogin="false"</code> (which means that the LDAP server will sent back
45   * the password field and checking it with the clear password won't be performed by the LDAP server)
46   * rather than <code>forceBindingLogin="true"</code>
47   * (which means that the clear password will be sent to the LDAP server that will check if they
48   * match the password field, therefore the password field stored in LDAP can't be a hash,
49   * it must be a clear plaintext password only).
50   * In a new release these may change, <a href="https://github.com/pingidentity/ldapsdk/issues/32">as mentioned here</a>
51   *
52   * @author Philippe Poulard
53   */

155     @Override
156     protected void doStart() throws Exception {
157         final InMemoryListenerConfig listenerConfig = InMemoryListenerConfig.createLDAPConfig("default", port);
158         final InMemoryDirectoryServerConfig config = new InMemoryDirectoryServerConfig(new DN(baseDn));
159         config.setListenerConfigs(listenerConfig);
160         if (! StringUtil.isVoid(bindDn) && ! StringUtil.isVoid(bindPassword)) {
161             config.addAdditionalBindCredentials(bindDn, bindPassword);
162         }
163         ldapServer = new InMemoryDirectoryServer(config);
164         ldapServer .startListening();
165         for (final String ldifFile : ldifFiles) {
166             loadData(ldifFile);
167         }
168     }
169 
170     /**
171      * Load LDIF records from a file to seed the LDAP directory.
172      *
173      * @param server   The embedded LDAP directory server.
174      * @param ldifFile The LDIF resource or file from which LDIF records will be loaded.
175      * @throws com.unboundid.ldif.LDIFException     If there was an error in the LDIF data.
176      * @throws com.unboundid.ldap.sdk.LDAPException If there was a problem loading the LDIF records into the LDAP directory.
177      * @throws java.io.IOException                  If there was a problem reading the LDIF records from the file.
178      */
179     public void loadData(String ldifFile)
180             throws LDIFException, LDAPException, IOException {
181         if (ldifFile != null && !ldifFile.isEmpty()) {
182             InputStream inputStream = new FileInputStream(ldifFile);
183             try {
184                 final LDIFReader ldifReader = new LDIFReader(inputStream);
185                 for (LDIFChangeRecord changeRecord = ldifReader.readChangeRecord(true);
186                                       changeRecord != null;
187                                       changeRecord = ldifReader.readChangeRecord(true)) {
188                     changeRecord.processChange(ldapServer.getConnection());
189                 }
190             } finally {
191                 inputStream.close();
192             }
193         }
194     }
195 
196     @Override
197     protected void doStop() throws Exception {
198         this.ldapServer.shutDown(true);
199         this.ldapServer = null;
200     }
