18  /**
19   * The tests that show that the passwords are captured by
20   * Jetty (and encrypted), and that the raw value of the password in the
21   * incoming buffer is replaced with '*'.
22   *
23   * @author Philippe Poulard
24   */
25  public class RESTTest extends RESTServerTestHarness<Server> {
26  
27      static {
28          name = Methref.on(RESTTest.class);
29      }
30  
31      @Override
32      @Test(enabled = false)
33      public void doStopServer() throws Exception {
34          server.stop();
35          server.destroy();
36          server = null;
37      }
38  
39      @Override
40      @Test(enabled = false)
41      public void doStartServer() throws Exception {
42          if (server == null) {
43              WebAppContext wac = new WebAppContext();
44              // a reference to this test
45              wac.setAttribute(RESTServerTestHarness.class.getName(), this);
46  
47              AltHttpConnectionFactory cf = new AltHttpConnectionFactory(wac);
48  
49              server = new Server();
50              ServerConnector connector=new ServerConnector(server, cf);
51              connector.setPort(port);
52              server.setConnectors(new Connector[]{connector});
53  
54              ServletHolder sh = new ServletHolder("JERSEY_SERVLET", ServletContainer.class);
55              String classes = ExampleRequest.class.getName() + " ";
56              sh.setInitParameter(ServerProperties.PROVIDER_CLASSNAMES, classes);
57              sh.setInitParameter("javax.ws.rs.Application", RestApp.class.getName());
58  
59              ServletHandler servletHandler = new ServletHandler();
60              servletHandler.addServletWithMapping(sh, "/*");
61  
62              wac.setServletHandler(servletHandler);
63              wac.setResourceBase(resourceBase);
64              wac.setContextPath(contextPath);
65              // configure for handling the "pwd" field when POSTing on the "/Test" path
66              wac.setInitParameter(Config.FORM_FIELDS_INIT_PARAM, "/Test?pwd");
67  
68              server.setHandler(wac);
69  
70              server.start();
71  //            server.dumpStdErr();
72          }
73      }
74  
75  }
