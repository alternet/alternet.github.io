23  /**
24   * A scanner can read Unicode characters from an input
25   * stream under conditions.
26   *
27   * <p>Convenient methods are available for
28   * testing for example whether the next content
29   * is a number or not, for reading characters,
30   * strings and numbers, pick a value
31   * from a set of possible strings or Enum values,
32   * and even pick the next character if it belongs
33   * to a range of characters (possibly built with
34   * union and exclusions of other ranges).</p>
35   *
36   * <p>Additional classes can help to read an
37   * object under constraints, for example "get
38   * the next number that has less than 5 digits",
39   * or the next number that fit in a specific type.</p>
40   *
41   * <p>It is a progressive scanner in the sense that
42   * the next content is inherently independent of
43   * what was read before. To hold a context and go
44   * back when successive items doesn't satisfy a
45   * specific grammar, the user will have to set markers.</p>
46   *
47   * <p>The {@link #mark()}, {@link #consume()}, and
48   * {@link #cancel()} methods can help to read some
49   * characters and go back to the marked position.
50   * Several successive positions can be marked without
51   * canceling or consuming the previous ones ;
52   * as marks are stacked, it falls to the user
53   * to apply the appropriate number of cancel + consume
54   * calls.</p>
55   *
56   * <p>A specific device is available for single Unicode
57   * characters :
58   * just use {@link #hasNextChar(int, boolean)},
59   * {@link #hasNextChar(String, boolean)}, or get it simply
60   * with {@link #lookAhead()} ; you don't need to set a mark
61   * for testing the next character to read.</p>
62   *
63   * <p>It is markable-reentrant that is to say
64   * that once marked, it can used itself for further
65   * scanning with or without setting new marks.</p>
66   *
67   * @author Philippe Poulard
68   */
69  public abstract class Scanner implements Trackable, Rewindable {

157     /**
158      * Append the next string in the given buffer.
159      *
160      * <p>The string stops at the first separator specified
161      * by the stop condition of the constraint or at the
162      * end of the stream. Each character to append is
163      * filtered by the constraint (for example to allow
164      * escaping characters).</p>
165      *
166      * <p>The characters involved in the stop condition
167      * can be preserved (they will be the next
168      * characters to read) or not according to the
169      * stop condition.</p>
170      * <p>WARNING: It is the responsibility
171      * of the {@link StringConstraint#stopCondition(int, int, Scanner)}
172      * method to mark and reset the scanner if necessary.</p>
173      *
174      * <p>In any case the characters involved in the stop condition are not
175      * appended to the buffer.</p>
176      *
177      * @param constraint The non-<code>null</code> constraint that the
178      *         string to read has to satisfy.
179      * @param buf The buffer to which the next string will be appended.
180      *
181      * @return The number of characters appended to the buffer.
182      *
183      * @throws IOException When an I/O error occur.
184      */
185     public int nextString( StringConstraint constraint, StringBuilder buf ) throws IOException {
186         int targetLength = 0;
187         this.state.sourceIndex = 0;
188         while ( ! this.state.end && ! constraint.stopCondition( this.state.sourceIndex, targetLength, this ) ) {
189             this.state.sourceIndex++;
190             targetLength += constraint.append( this.state.sourceIndex, targetLength, this, buf );
191             this.state.source.read();
192         }
193         return targetLength;
194     }

238      *      {@code false} to stop the stream.

241     public IntStream nextChars(IntPredicate predicate) {
242         return StreamSupport.intStream(new Spliterators.AbstractIntSpliterator(
243                 Long.MAX_VALUE, Spliterator.ORDERED | Spliterator.NONNULL | Spliterator.IMMUTABLE )
244         {
245 
246             @Override
247             public boolean tryAdvance(IntConsumer action) {
248                 if (hasNext() && predicate.test(lookAhead())) {
249                     action.accept(lookAhead());
250                     Thrower.safeCall(() -> read());
251                     return true;
252                 } else {
253                     return false;
254                 }
255             }
256         }, false);

260     /**
261      * Skip the next string (under constraint).
262      *
263      * <p>The string stops at the first separator specified
264      * by the stop condition of the constraint or at the
265      * end of the stream.</p>
266      *
267      * <p>The characters involved in the stop condition
268      * can be preserved (they will be the next
269      * characters to read) or not according to the
270      * stop condition.</p>
271      * <p>WARNING: It is the responsibility
272      * of the {@link StringConstraint#stopCondition(int, int, Scanner)}
273      * method to mark and reset the scanner if necessary.</p>
274      *
275      * @param constraint The constraints to apply on the string.
276      *
277      * @return The number of characters read, excluding those
278      *         involved in the stop condition.
279      *
280      * @throws IOException When an I/O error occur.
281      *
282      * @see #nextString(StringConstraint, StringBuilder)
283      */
284     public int skipNextString( StringConstraint constraint ) throws IOException {
285         this.state.sourceIndex = 0;
286         while ( ! this.state.end && ! constraint.stopCondition( this.state.sourceIndex, 0, this ) ) {
287             this.state.sourceIndex++;
288             this.state.source.read();
289         }
290         return this.state.sourceIndex;
291     }

347     /**
348      * Read the next number.
349      *
350      * @param constraint The non-<code>null</code> constraint that the
351      *         number to read has to satisfy.
352      *
353      * @return The next number read, or <code>null</code> if none
354      *         found under the constraints.
355      *
356      * @throws IOException When an I/O error occur.
357      *
358      * @see NumberUtil#parseNumber(String, boolean, Class)
359      */
360     public Number nextNumber( NumberConstraint constraint ) throws IOException {
361         StringBuffer buf = new StringBuffer();
362         this.state.source.mark();
363         boolean isFloatingPoint = parseNumber( buf, constraint );
364         // now, the buffer contains the expected number as a string
365         // just return the right class
366         if ( buf.length() > 0 ) {
367             try {
368                 String number = buf.toString();
369                 Number n = NumberUtil.parseNumber(
370                     number,
371                     isFloatingPoint,
372                     constraint.getNumberType()
373                 );
374                 this.state.source.consume();
375                 return n;
376             } catch ( NumberFormatException nfe ) {
377                 // ooops !
378                 this.state.source.cancel();
379                 return null;
380             }
381         } else {
382             this.state.source.cancel();
383             // no number found
384             return null;
385         }
386     }

388     /**
389      * Parse a number under constraint.
390      *
391      * @param buf The target buffer to fill with the number.
392      *         '+' and leading zeroes are discarded.
393      * @param constraint The non-<code>null</code> constraint that the
394      *         number to read has to satisfy.
395      *
396      * @return <code>true</code> if the parsed number is a floating number,
397      *         <code>false</code> otherwise.
398      *
399      * @throws IOException When the scanner fails to read the input.
400      */
401     private boolean parseNumber( StringBuffer buf, NumberConstraint constraint ) throws IOException {
402 // TODO :
403 //        Class clazz = constraint.getClass();
404 //        if ( clazz != null ) {
405 //            constraint = new CompositeNumberConstraint( constraint, clazz );
406 //        }
407         this.state.sourceIndex = 0;
408         int length = 0;
409         int dotIndex = -1;
410         int exponentIndex = -1;
411         if ( constraint.stopCondition(buf, this.state.sourceIndex, dotIndex, exponentIndex, this) ) {
412             return false;
413         }
414         char prev = (char) -1;
415         boolean isFloatingPoint = false;
416         char sign = (char) nextChar( "-+", true );
417         switch ( sign ) {
418         case '-':
419             buf.append( sign );
420             length++;
421             this.state.sourceIndex++;
422         case '+':
423             prev = sign;
424         }
425         if ( constraint.stopCondition(buf, this.state.sourceIndex, dotIndex, exponentIndex, this) ) {
426             return isFloatingPoint;
427         }
428         for ( int c = lookAhead() ; c >= '0' && c <= '9' ; c = lookAhead() ) {
429             // skip leading zeroes
430             // FIXME : wouldn't it be better to count leading zeroes and pass them to the constraint ?
431             if ( prev == '0' && ( length == 1 || (length == 2 && sign == '-' ) ) ) {
432                 buf.setCharAt( length - 1, (char) c ); // c is a BMP codepoint
433             } else {
434                 length++;
435                 buf.append( (char) c );
436             }
437             this.state.source.read();
438             this.state.sourceIndex++;
439             if ( constraint.stopCondition(buf, this.state.sourceIndex, dotIndex, exponentIndex, this) ) {
440                 return isFloatingPoint;
441             }
442             prev = (char) c;
443         }
444         if ( constraint.stopCondition(buf, this.state.sourceIndex, dotIndex, exponentIndex, this) ) {
445             return isFloatingPoint;
446         }
447         if ( hasNextChar( '.', true ) ) {
448             // FIXME : set the flag only when a non-zero digit is encountered ?
449             isFloatingPoint = true;
450             buf.append('.');
451             this.state.sourceIndex++;
452             dotIndex = this.state.sourceIndex;
453             if ( constraint.stopCondition(buf, this.state.sourceIndex, dotIndex, exponentIndex, this) ) {
454                 return isFloatingPoint;
455             }
456             for ( int c = lookAhead() ; c >= '0' && c <= '9' ; c = lookAhead() ) {
457                 length++;
458                 buf.append( (char) c ); // c is a BMP codepoint
459                 this.state.source.read();
460                 this.state.sourceIndex++;
461                 if ( constraint.stopCondition(buf, this.state.sourceIndex, dotIndex, exponentIndex, this) ) {
462                     return isFloatingPoint;
463                 }
464             }
465         }
466         if ( constraint.stopCondition(buf, this.state.sourceIndex, dotIndex, exponentIndex, this) ) {
467             return isFloatingPoint;
468         }
469         if ( hasNextChar( "eE", true ) ) {
470             isFloatingPoint = true;
471             buf.append('e');
472             this.state.sourceIndex++;
473             exponentIndex = this.state.sourceIndex;
474             sign = (char) nextChar( "-+", true );
475             if ( constraint.stopCondition(buf, this.state.sourceIndex, dotIndex, exponentIndex, this) ) {
476                 return isFloatingPoint;
477             }
478             if ( sign != IOUtil.EOF ) {
479                 buf.append( sign );
480                 this.state.sourceIndex++;
481             }
482             if ( constraint.stopCondition(buf, this.state.sourceIndex, dotIndex, exponentIndex, this) ) {
483                 return isFloatingPoint;
484             }
485             for ( int c = lookAhead() ; c >= '0' && c <= '9' ; c = lookAhead() ) {
486                 buf.append( (char) c ); // c is a BMP codepoint
487                 this.state.source.read();
488                 this.state.sourceIndex++;
489                 if ( constraint.stopCondition(buf, this.state.sourceIndex, dotIndex, exponentIndex, this) ) {
490                     return isFloatingPoint;
491                 }
492             }
493         }
494         return isFloatingPoint;
495     }

709     /**
710      * Test whether or not the next string in the input is those given.
711      * When found it can be consumed or not.
712      *
713      * @param string The string to test.
714      * @param consume <code>true</code> if the string found have to be
715      *         consumed, <code>false</code> otherwise.
716      *
717      * @return <code>true</code> if the string matches the input,
718      *         <code>false</code> otherwise.
719      *
720      * @throws IOException When an I/O error occur.
721      */
722     public boolean hasNextString(CharSequence string, boolean consume) throws IOException {
723         if ( string == null ) {
724             return true;
725         } else if ( this.state.end ) {
726             return false;
727         } else {
728             int len = string.length();
729             if ( len == 0 ) {
730                 return true;
731             } else {
732                 int c = this.state.next;
733                 if ( c == codePointAt(string, 0) ) {
734                     int start = Character.charCount(c);
735                     if ( len > start ) {
736                         this.state.source.mark();
737                         for ( int i = start ; i < len ; i++ ) {
738                             this.state.source.read();
739                             if ( this.state.end ) {
740                                 this.state.source.cancel();
741                                 return false;
742                             }
743                             c = codePointAt(string, i);
744                             if ( c != this.state.next ) {
745                                 this.state.source.cancel();
746                                 return false;
747                             } else if (Character.isSupplementaryCodePoint(c)) {
748                                 i++;
749                             }
750                         }
751                         if ( consume ) {
752                             this.state.source.consume();
753                             this.state.source.read();
754                         } else {
755                             this.state.source.cancel();
756                         }
757                     } else {
758                         if ( consume ) {
759                             this.state.source.read();
760                         }
761                     }
762                     return true;
763                 } else {
764                     return false;
765                 }
766             }
767         }
768     }

887     static int codePointAt(CharSequence string, int index) {
888         char c1 = string.charAt(index++);
889         if (Character.isHighSurrogate(c1)) {
890             char c2 = string.charAt(index);
891             if (Character.isLowSurrogate(c2)) {
892                 return Character.toCodePoint(c1, c2);
893             }
894         }
895         return c1;
896     }
