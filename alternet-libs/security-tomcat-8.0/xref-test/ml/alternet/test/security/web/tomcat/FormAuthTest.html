30  /**
31   * The tests that show on Form authentication that the password is captured by
32   * Tomcat (and encrypted), and that the raw value of the password in the
33   * incoming buffer is replaced with '*'.
34   *
35   * @author Philippe Poulard
36   */
37  public class FormAuthTest extends FormAuthServerTestHarness<Tomcat> implements TomcatSupplier {
38  
39      static {
40          name = Methref.on(FormAuthTest.class);
41      }
42  
43      /**
44       * Tomcat specific test
45       */
46      @Override
47      public String checkSession(HttpServletRequest req) {
48          HttpSession session = req.getSession();
49          name.to().sessionAuthentication_ShouldBe_Define();
50          serverTests.put(name.ref(),
51              () -> Assertions.assertThat(session)
52              .as("REST server : session authentication should be defined")
53              .isNotNull()
54          );
55          return null; // Tomcat remove the password from the session notes
56      }
57  
58      @Override
59      @Test(enabled = false)
60      public void doStopServer() throws Exception {
61          server.stop();
62          server.destroy();
63          server = null;
64      }
65  
66      @Override
67      @Test(enabled = false)
68      public void doStartServer() throws Exception {
69          server = get();
70  
71          // use the specific connector
72          Connector connector = new Connector("ml.alternet.security.web.tomcat.AltProtocolHandler");
73  //        Connector connector = new Connector("HTTP/1.1");
74          connector.setPort(port);
75          // set in "tomcatProtocol" what you would have set in the  Connector constructor
76          connector.setProperty("tomcatProtocol", "HTTP/1.1");
77          connector.setProperty("passwordManager", "ml.alternet.security.impl.StrongPasswordManager");
78          server.getService().addConnector(connector);
79          server.setConnector(connector);
80  
81          // create a REST webapp
82          String resourceBase = new File("src/main/resources/").getAbsolutePath();
83          Context wac = server.addContext(contextPath, resourceBase);
84          wac.setUseHttpOnly(false);
85          Wrapper servlet = Tomcat.addServlet(wac, "JERSEY_SERVLET", new ServletContainer());
86          String classes = ExampleRequest.class.getName() + " ";
87          servlet.addInitParameter(ServerProperties.PROVIDER_CLASSNAMES, classes);
88          wac.addServletMapping("/protected/*", "JERSEY_SERVLET");
89          wac.addServletMapping("/go/*", "JERSEY_SERVLET");
90          // a reference to this test
91          wac.getServletContext().setAttribute(FormAuthServerTestHarness.class.getName(), this);
92  
93          @SuppressWarnings("unused")
94          Wrapper defaultServlet = Tomcat.addServlet(wac, "default", new DefaultServlet());
95          wac.addServletMapping("/login.html", "default");
96  
97          // configure the security (standard fashion)
98          wac.setPreemptiveAuthentication(true);
99          wac.setPrivileged(true);
100         wac.addSecurityRole("customer");
101         wac.addSecurityRole("admin");
102         wac.setLoginConfig(new LoginConfig("FORM", "realm", "/login.html", "/authFail.html"));
103         SecurityConstraint security = new SecurityConstraint();
104         security.addAuthRole("admin");
105         security.setAuthConstraint(true);
106         SecurityCollection coll = new SecurityCollection();
107         coll.addMethod("GET");
108         coll.addPattern("/protected/*");
109         security.addCollection(coll);
110         wac.addConstraint(security);
111 
112         // specific security config
113         ApplicationParameter initParam = new ApplicationParameter();
114         initParam.setName(Config.AUTH_METHOD_INIT_PARAM);
115         initParam.setValue("Form");
116         wac.addApplicationParameter(initParam);
117 
118         AuthenticatorBase auth = new AltFormAuthenticator();
119         auth.setSecurePagesWithPragma(false);
120         auth.setChangeSessionIdOnAuthentication(true);
121         auth.setAlwaysUseSession(true);
122         ((StandardContext) wac).addValve(auth);
123 
124         MappedRealm realm = new MappedRealm();
125         realm.putUser(userName, unsafePwd.toCharArray(), new String[] {"admin"});
126         // specific conf
127         realm.setCredentialHandler(new AltCredentialHandler());
128         wac.setRealm(realm);
129 
130         server.start();
131 
132     }
