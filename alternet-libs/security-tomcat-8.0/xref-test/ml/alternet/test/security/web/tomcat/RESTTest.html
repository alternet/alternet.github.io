19  /**
20   * The tests that show that the passwords are captured by
21   * Jetty (and encrypted), and that the raw value of the password in the
22   * incoming buffer is replaced with '*'.
23   *
24   * @author Philippe Poulard
25   */
26  public class RESTTest extends RESTServerTestHarness<Tomcat> implements TomcatSupplier {
27  
28      static {
29          name = Methref.on(RESTTest.class);
30      }
31  
32      @Override
33      @Test(enabled = false)
34      public void doStopServer() throws Exception {
35          server.stop();
36          server.destroy();
37          server = null;
38      }
39  
40      @Override
41      @Test(enabled = false)
42      public void doStartServer() throws Exception {
43          server = get();
44          server.getEngine().setName(server.getEngine().getName() + "-" + Objects.hashCode(server.getEngine()));
45  
46          // use the specific connector
47          Connector connector = new Connector("ml.alternet.security.web.tomcat.AltProtocolHandler");
48          connector.setPort(port);
49          // set in "tomcatProtocol" what you would have set in the  Connector constructor
50          connector.setProperty("tomcatProtocol", "HTTP/1.1");
51          connector.setProperty("passwordManager", "ml.alternet.security.impl.StrongPasswordManager");
52          server.getService().addConnector(connector);
53          server.setConnector(connector);
54  
55          // create a REST webapp
56          Context wac = server.addContext(contextPath, resourceBase);
57          Wrapper servlet = Tomcat.addServlet(wac, "JERSEY_SERVLET", new ServletContainer());
58          String classes = ExampleRequest.class.getName() + " ";
59          servlet.addInitParameter(ServerProperties.PROVIDER_CLASSNAMES, classes);
60          servlet.addInitParameter("javax.ws.rs.Application", RestApp.class.getName());
61          wac.addServletMapping("/*", "JERSEY_SERVLET");
62  
63          // a reference to this test
64          wac.getServletContext().setAttribute(RESTServerTestHarness.class.getName(), this);
65  
66          // configure for handling the "pwd" field when POSTing on the "/Test" path
67          ApplicationParameter initParam = new ApplicationParameter();
68          initParam.setName(Config.FORM_FIELDS_INIT_PARAM);
69          initParam.setValue("/Test?pwd");
70          wac.addApplicationParameter(initParam);
71  
72          server.start();
73      }
74  
75  }
