12  import org.apache.catalina.realm.GenericPrincipal;
13  import org.apache.catalina.realm.RealmBase;
14  
15  class MappedRealm extends RealmBase {
16  
17      Hasher hr = ModularCryptFormat.Hashers.$2$.get(); // TODO Hasher conf
18  
19      private final Map<String,GenericPrincipal> principals = new HashMap<>();
20  
21      public void putUser(String userName, char[] password, String[] roles)
22              throws InvalidAlgorithmParameterException {
23  //        String crypt = Hasher.getDefault().encrypt(password);
24          String crypt = hr.encrypt(password);
25          GenericPrincipal principal = new GenericPrincipal(userName, crypt, Arrays.asList(roles));
26          principals.put(userName, principal);
27      }
28  
29      @Override
30      protected String getName() {
31          return MappedRealm.class.getName();
32      }
33  
34      @Override
35      protected String getPassword(String username) {
36          GenericPrincipal principal =  principals.get(username);
37          if (principal != null) {
38              return principal.getPassword();
39          } else {
40              return null;
41          }
42      }
43      @Override
44      protected Principal getPrincipal(String username) {
45          return principals.get(username);
46      }
47  }
